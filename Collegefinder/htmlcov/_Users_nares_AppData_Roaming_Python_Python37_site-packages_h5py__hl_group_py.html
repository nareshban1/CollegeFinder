417 def copy(self, source, dest, name=None, 

418 shallow=False, expand_soft=False, expand_external=False, 

419 expand_refs=False, without_attrs=False): 

420 """Copy an object or group. 

421 

422 The source can be a path, Group, Dataset, or Datatype object. The 

423 destination can be either a path or a Group object. The source and 

424 destinations need not be in the same file. 

425 

426 If the source is a Group object, all objects contained in that group 

427 will be copied recursively. 

428 

429 When the destination is a Group object, by default the target will 

430 be created in that group with its current name (basename of obj.name). 

431 You can override that by setting "name" to a string. 

432 

433 There are various options which all default to "False": 

434 

435 - shallow: copy only immediate members of a group. 

436 

437 - expand_soft: expand soft links into new objects. 

438 

439 - expand_external: expand external links into new objects. 

440 

441 - expand_refs: copy objects that are pointed to by references. 

442 

443 - without_attrs: copy object without copying attributes. 

444 

445 Example: 

446 

447 >>> f = File('myfile.hdf5') 

448 >>> f.listnames() 

449 ['MyGroup'] 

450 >>> f.copy('MyGroup', 'MyCopy') 

451 >>> f.listnames() 

452 ['MyGroup', 'MyCopy'] 

453 

454 """ 

455 with phil: 

456 if isinstance(source, HLObject): 

457 source_path = '.' 

458 else: 

459 # Interpret source as a path relative to this group 

460 source_path = source 

461 source = self 

462 

463 if isinstance(dest, Group): 

464 if name is not None: 

465 dest_path = name 

466 else: 

467 # copy source into dest group: dest_name/source_name 

468 dest_path = pp.basename(h5i.get_name(source[source_path].id)) 

469 

470 elif isinstance(dest, HLObject): 

471 raise TypeError("Destination must be path or Group object") 

472 else: 

473 # Interpret destination as a path relative to this group 

474 dest_path = dest 

475 dest = self 

476 

477 flags = 0 

478 if shallow: 

479 flags |= h5o.COPY_SHALLOW_HIERARCHY_FLAG 

480 if expand_soft: 

481 flags |= h5o.COPY_EXPAND_SOFT_LINK_FLAG 

482 if expand_external: 

483 flags |= h5o.COPY_EXPAND_EXT_LINK_FLAG 

484 if expand_refs: 

485 flags |= h5o.COPY_EXPAND_REFERENCE_FLAG 

486 if without_attrs: 

487 flags |= h5o.COPY_WITHOUT_ATTR_FLAG 

488 if flags: 

489 copypl = h5p.create(h5p.OBJECT_COPY) 

490 copypl.set_copy_object(flags) 

491 else: 

492 copypl = None 

493 

494 h5o.copy(source.id, self._e(source_path), dest.id, self._e(dest_path), 

495 copypl, base.dlcpl)

510 def visit(self, func): 

511 """ Recursively visit all names in this group and subgroups (HDF5 1.8). 

512 

513 You supply a callable (function, method or callable object); it 

514 will be called exactly once for each link in this group and every 

515 group below it. Your callable must conform to the signature: 

516 

517 func(<member name>) => <None or return value> 

518 

519 Returning None continues iteration, returning anything else stops 

520 and immediately returns that value from the visit method. No 

521 particular order of iteration within groups is guaranteed. 

522 

523 Example: 

524 

525 >>> # List the entire contents of the file 

526 >>> f = File("foo.hdf5") 

527 >>> list_of_names = [] 

528 >>> f.visit(list_of_names.append) 

529 """ 

530 with phil: 

531 def proxy(name): 

532 """ Call the function with the text name, not bytes """ 

533 return func(self._d(name)) 

534 return h5o.visit(self.id, proxy)

541 group below it. Your callable must conform to the signature: 

542 

543 func(<member name>, <object>) => <None or return value> 

544 

545 Returning None continues iteration, returning anything else stops

548 

549 Example: 

550
