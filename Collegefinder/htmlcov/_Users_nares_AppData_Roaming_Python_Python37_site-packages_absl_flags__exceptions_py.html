44class DuplicateFlagError(Error): 

45 """Raised if there is a flag naming conflict.""" 

46 

47 @classmethod 

48 def from_flag(cls, flagname, flag_values, other_flag_values=None): 

49 """Creates a DuplicateFlagError by providing flag name and values. 

50 

51 Args: 

52 flagname: str, the name of the flag being redefined. 

53 flag_values: FlagValues, the FlagValues instance containing the first 

54 definition of flagname. 

55 other_flag_values: FlagValues, if it is not None, it should be the 

56 FlagValues object where the second definition of flagname occurs. 

57 If it is None, we assume that we're being called when attempting 

58 to create the flag a second time, and we use the module calling 

59 this one as the source of the second definition. 

60 

61 Returns: 

62 An instance of DuplicateFlagError. 

63 """ 

64 first_module = flag_values.find_module_defining_flag( 

65 flagname, default='<unknown>') 

66 if other_flag_values is None: 

67 second_module = _helpers.get_calling_module() 

68 else: 

69 second_module = other_flag_values.find_module_defining_flag( 

70 flagname, default='<unknown>') 

71 flag_summary = flag_values[flagname].help 

72 msg = ("The flag '%s' is defined twice. First from %s, Second from %s. " 

73 "Description from first occurrence: %s") % ( 

74 flagname, first_module, second_module, flag_summary) 

75 return cls(msg)

82class UnrecognizedFlagError(Error): 

83 """Raised when a flag is unrecognized. 

84 

85 Attributes: 

86 flagname: str, the name of the unrecognized flag. 

87 flagvalue: The value of the flag, empty if the flag is not defined. 

88 """ 

89 

90 def __init__(self, flagname, flagvalue='', suggestions=None): 

91 self.flagname = flagname 

92 self.flagvalue = flagvalue 

93 if suggestions: 

94 # Space before the question mark is intentional to not include it in the 

95 # selection when copy-pasting the suggestion from (some) terminals. 

96 tip = '. Did you mean: %s ?' % ', '.join(suggestions) 

97 else: 

98 tip = '' 

99 super(UnrecognizedFlagError, self).__init__( 

100 'Unknown command line flag \'%s\'%s' % (flagname, tip))
