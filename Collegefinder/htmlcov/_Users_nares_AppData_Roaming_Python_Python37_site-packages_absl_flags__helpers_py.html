46# 

47# We lean towards a high threshold because we tend to be matching a phrase,

90def get_module_object_and_name(globals_dict): 

91 """Returns the module that defines a global environment, and its name. 

92 

93 Args: 

94 globals_dict: A dictionary that should correspond to an environment 

95 providing the values of the globals. 

96 

97 Returns: 

98 _ModuleObjectAndName - pair of module object & module name. 

99 Returns (None, None) if the module could not be identified. 

100 """ 

101 name = globals_dict.get('__name__', None) 

102 module = sys.modules.get(name, None) 

103 # Pick a more informative name for the main module. 

104 return _ModuleObjectAndName(module, 

105 (sys.argv[0] if name == '__main__' else name))

108def get_calling_module_object_and_name(): 

109 """Returns the module that's calling into this module. 

110 

111 We generally use this function to get the name of the module calling a 

112 DEFINE_foo... function. 

113 

114 Returns: 

115 The module object that called into this one. 

116 

117 Raises: 

118 AssertionError: Raised when no calling module could be identified. 

119 """ 

120 for depth in range(1, sys.getrecursionlimit()): 

121 # sys._getframe is the right thing to use here, as it's the best 

122 # way to walk up the call stack. 

123 globals_for_frame = sys._getframe(depth).f_globals # pylint: disable=protected-access 

124 module, module_name = get_module_object_and_name(globals_for_frame) 

125 if id(module) not in disclaim_module_ids and module_name is not None: 

126 return _ModuleObjectAndName(module, module_name) 

127 raise AssertionError('No module was found')

135def str_or_unicode(value): 

136 """Converts a value to a python string. 

137 

138 Behavior of this function is intentionally different in Python2/3. 

139 

140 In Python2, the given value is attempted to convert to a str (byte string). 

141 If it contains non-ASCII characters, it is converted to a unicode instead. 

142 

143 In Python3, the given value is always converted to a str (unicode string). 

144 

145 This behavior reflects the (bad) practice in Python2 to try to represent 

146 a string as str as long as it contains ASCII characters only. 

147 

148 Args: 

149 value: An object to be converted to a string. 

150 

151 Returns: 

152 A string representation of the given value. See the description above 

153 for its type. 

154 """ 

155 try: 

156 return str(value) 

157 except UnicodeEncodeError: 

158 return unicode(value) # Python3 should never come here

161def create_xml_dom_element(doc, name, value): 

162 """Returns an XML DOM element with name and text value. 

163 

164 Args: 

165 doc: minidom.Document, the DOM document it should create nodes from. 

166 name: str, the tag of XML element. 

167 value: object, whose string representation will be used 

168 as the value of the XML element. Illegal or highly discouraged xml 1.0 

169 characters are stripped. 

170 

171 Returns: 

172 An instance of minidom.Element. 

173 """ 

174 s = str_or_unicode(value) 

175 if six.PY2 and not isinstance(s, unicode): 

176 # Get a valid unicode string. 

177 s = s.decode('utf-8', 'ignore') 

178 if isinstance(value, bool): 

179 # Display boolean values as the C++ flag library does: no caps. 

180 s = s.lower() 

181 # Remove illegal xml characters. 

182 s = _ILLEGAL_XML_CHARS_REGEX.sub(u'', s) 

183 

184 e = doc.createElement(name) 

185 e.appendChild(doc.createTextNode(s)) 

186 return e

189def get_help_width(): 

190 """Returns the integer width of help lines that is used in TextWrap.""" 

191 if not sys.stdout.isatty() or termios is None or fcntl is None: 

192 return _DEFAULT_HELP_WIDTH 

193 try: 

194 data = fcntl.ioctl(sys.stdout, termios.TIOCGWINSZ, '1234') 

195 columns = struct.unpack('hh', data)[1] 

196 # Emacs mode returns 0. 

197 # Here we assume that any value below 40 is unreasonable. 

198 if columns >= _MIN_HELP_WIDTH: 

199 return columns 

200 # Returning an int as default is fine, int(int) just return the int. 

201 return int(os.getenv('COLUMNS', _DEFAULT_HELP_WIDTH)) 

202 

203 except (TypeError, IOError, struct.error): 

204 return _DEFAULT_HELP_WIDTH

207def get_flag_suggestions(attempt, longopt_list): 

208 """Returns helpful similar matches for an invalid flag.""" 

209 # Don't suggest on very short strings, or if no longopts are specified. 

210 if len(attempt) <= 2 or not longopt_list: 

211 return [] 

212 

213 option_names = [v.split('=')[0] for v in longopt_list] 

214 

215 # Find close approximations in flag prefixes. 

216 # This also handles the case where the flag is spelled right but ambiguous. 

217 distances = [(_damerau_levenshtein(attempt, option[0:len(attempt)]), option) 

218 for option in option_names] 

219 # t[0] is distance, and sorting by t[1] allows us to have stable output. 

220 distances.sort() 

221 

222 least_errors, _ = distances[0] 

223 # Don't suggest excessively bad matches. 

224 if least_errors >= _SUGGESTION_ERROR_RATE_THRESHOLD * len(attempt): 

225 return [] 

226 

227 suggestions = [] 

228 for errors, name in distances: 

229 if errors == least_errors: 

230 suggestions.append(name) 

231 else: 

232 break 

233 return suggestions

236def _damerau_levenshtein(a, b): 

237 """Returns Damerau-Levenshtein edit distance from a to b.""" 

238 memo = {} 

239 

240 def distance(x, y): 

241 """Recursively defined string distance with memoization.""" 

242 if (x, y) in memo: 

243 return memo[x, y] 

244 if not x: 

245 d = len(y) 

246 elif not y: 

247 d = len(x) 

248 else: 

249 d = min( 

250 distance(x[1:], y) + 1, # correct an insertion error 

251 distance(x, y[1:]) + 1, # correct a deletion error 

252 distance(x[1:], y[1:]) + (x[0] != y[0])) # correct a wrong character 

253 if len(x) >= 2 and len(y) >= 2 and x[0] == y[1] and x[1] == y[0]: 

254 # Correct a transposition. 

255 t = distance(x[2:], y[2:]) + 1 

256 if d > t: 

257 d = t 

258 

259 memo[x, y] = d 

260 return d 

261 return distance(a, b)

264def text_wrap(text, length=None, indent='', firstline_indent=None): 

265 """Wraps a given text to a maximum line length and returns it. 

266 

267 It turns lines that only contain whitespace into empty lines, keeps new lines, 

268 and expands tabs using 4 spaces. 

269 

270 Args: 

271 text: str, text to wrap. 

272 length: int, maximum length of a line, includes indentation. 

273 If this is None then use get_help_width() 

274 indent: str, indent for all but first line. 

275 firstline_indent: str, indent for first line; if None, fall back to indent. 

276 

277 Returns: 

278 str, the wrapped text. 

279 

280 Raises: 

281 ValueError: Raised if indent or firstline_indent not shorter than length. 

282 """ 

283 # Get defaults where callee used None 

284 if length is None: 

285 length = get_help_width() 

286 if indent is None: 

287 indent = '' 

288 if firstline_indent is None: 

289 firstline_indent = indent 

290 

291 if len(indent) >= length: 

292 raise ValueError('Length of indent exceeds length') 

293 if len(firstline_indent) >= length: 

294 raise ValueError('Length of first line indent exceeds length') 

295 

296 text = text.expandtabs(4) 

297 

298 result = [] 

299 # Create one wrapper for the first paragraph and one for subsequent 

300 # paragraphs that does not have the initial wrapping. 

301 wrapper = textwrap.TextWrapper( 

302 width=length, initial_indent=firstline_indent, subsequent_indent=indent) 

303 subsequent_wrapper = textwrap.TextWrapper( 

304 width=length, initial_indent=indent, subsequent_indent=indent) 

305 

306 # textwrap does not have any special treatment for newlines. From the docs: 

307 # "...newlines may appear in the middle of a line and cause strange output. 

308 # For this reason, text should be split into paragraphs (using 

309 # str.splitlines() or similar) which are wrapped separately." 

310 for paragraph in (p.strip() for p in text.splitlines()): 

311 if paragraph: 

312 result.extend(wrapper.wrap(paragraph)) 

313 else: 

314 result.append('') # Keep empty lines. 

315 # Replace initial wrapper with wrapper for subsequent paragraphs. 

316 wrapper = subsequent_wrapper 

317 

318 return '\n'.join(result)

321def flag_dict_to_args(flag_map): 

322 """Convert a dict of values into process call parameters. 

323 

324 This method is used to convert a dictionary into a sequence of parameters 

325 for a binary that parses arguments using this module. 

326 

327 Args: 

328 flag_map: dict, a mapping where the keys are flag names (strings). 

329 values are treated according to their type: 

330 * If value is None, then only the name is emitted. 

331 * If value is True, then only the name is emitted. 

332 * If value is False, then only the name prepended with 'no' is emitted. 

333 * If value is a string then --name=value is emitted. 

334 * If value is a collection, this will emit --name=value1,value2,value3. 

335 * Everything else is converted to string an passed as such. 

336 Yields: 

337 sequence of string suitable for a subprocess execution. 

338 """ 

339 for key, value in six.iteritems(flag_map): 

340 if value is None: 

341 yield '--%s' % key 

342 elif isinstance(value, bool): 

343 if value: 

344 yield '--%s' % key 

345 else: 

346 yield '--no%s' % key 

347 elif isinstance(value, (bytes, type(u''))): 

348 # We don't want strings to be handled like python collections. 

349 yield '--%s=%s' % (key, value) 

350 else: 

351 # Now we attempt to deal with collections. 

352 try: 

353 yield '--%s=%s' % (key, ','.join(str(item) for item in value)) 

354 except TypeError: 

355 # Default case. 

356 yield '--%s=%s' % (key, value)

359def trim_docstring(docstring): 

360 """Removes indentation from triple-quoted strings. 

361 

362 This is the function specified in PEP 257 to handle docstrings: 

363 https://www.python.org/dev/peps/pep-0257/. 

364 

365 Args: 

366 docstring: str, a python docstring. 

367 

368 Returns: 

369 str, docstring with indentation removed. 

370 """ 

371 if not docstring: 

372 return '' 

373 

374 # If you've got a line longer than this you have other problems... 

375 max_indent = 1 << 29 

376 

377 # Convert tabs to spaces (following the normal Python rules) 

378 # and split into a list of lines: 

379 lines = docstring.expandtabs().splitlines() 

380 

381 # Determine minimum indentation (first line doesn't count): 

382 indent = max_indent 

383 for line in lines[1:]: 

384 stripped = line.lstrip() 

385 if stripped: 

386 indent = min(indent, len(line) - len(stripped)) 

387 # Remove indentation (first line is special): 

388 trimmed = [lines[0].strip()] 

389 if indent < max_indent: 

390 for line in lines[1:]: 

391 trimmed.append(line[indent:].rstrip()) 

392 # Strip off trailing and leading blank lines: 

393 while trimmed and not trimmed[-1]: 

394 trimmed.pop() 

395 while trimmed and not trimmed[0]: 

396 trimmed.pop(0) 

397 # Return a single string: 

398 return '\n'.join(trimmed)

401def doc_to_help(doc): 

402 """Takes a __doc__ string and reformats it as help.""" 

403 

404 # Get rid of starting and ending white space. Using lstrip() or even 

405 # strip() could drop more than maximum of first line and right space 

406 # of last line. 

407 doc = doc.strip() 

408 

409 # Get rid of all empty lines. 

410 whitespace_only_line = re.compile('^[ \t]+$', re.M) 

411 doc = whitespace_only_line.sub('', doc) 

412 

413 # Cut out common space at line beginnings. 

414 doc = trim_docstring(doc) 

415 

416 # Just like this module's comment, comments tend to be aligned somehow. 

417 # In other words they all start with the same amount of white space. 

418 # 1) keep double new lines; 

419 # 2) keep ws after new lines if not empty line; 

420 # 3) all other new lines shall be changed to a space; 

421 # Solution: Match new lines between non white space and replace with space. 

422 doc = re.sub(r'(?<=\S)\n(?=\S)', ' ', doc, flags=re.M) 

423 

424 return doc
