32def solve_sylvester(a, b, q): 

33 """ 

34 Computes a solution (X) to the Sylvester equation :math:`AX + XB = Q`. 

35 

36 Parameters 

37 ---------- 

38 a : (M, M) array_like 

39 Leading matrix of the Sylvester equation 

40 b : (N, N) array_like 

41 Trailing matrix of the Sylvester equation 

42 q : (M, N) array_like 

43 Right-hand side 

44 

45 Returns 

46 ------- 

47 x : (M, N) ndarray 

48 The solution to the Sylvester equation. 

49 

50 Raises 

51 ------ 

52 LinAlgError 

53 If solution was not found 

54 

55 Notes 

56 ----- 

57 Computes a solution to the Sylvester matrix equation via the Bartels- 

58 Stewart algorithm. The A and B matrices first undergo Schur 

59 decompositions. The resulting matrices are used to construct an 

60 alternative Sylvester equation (``RY + YS^T = F``) where the R and S 

61 matrices are in quasi-triangular form (or, when R, S or F are complex, 

62 triangular form). The simplified equation is then solved using 

63 ``*TRSYL`` from LAPACK directly. 

64 

65 .. versionadded:: 0.11.0 

66 

67 Examples 

68 -------- 

69 Given `a`, `b`, and `q` solve for `x`: 

70 

71 >>> from scipy import linalg 

72 >>> a = np.array([[-3, -2, 0], [-1, -1, 3], [3, -5, -1]]) 

73 >>> b = np.array([[1]]) 

74 >>> q = np.array([[1],[2],[3]]) 

75 >>> x = linalg.solve_sylvester(a, b, q) 

76 >>> x 

77 array([[ 0.0625], 

78 [-0.5625], 

79 [ 0.6875]]) 

80 >>> np.allclose(a.dot(x) + x.dot(b), q) 

81 True 

82 

83 """ 

84 

85 # Compute the Schur decomp form of a 

86 r, u = schur(a, output='real') 

87 

88 # Compute the Schur decomp of b 

89 s, v = schur(b.conj().transpose(), output='real') 

90 

91 # Construct f = u'*q*v 

92 f = np.dot(np.dot(u.conj().transpose(), q), v) 

93 

94 # Call the Sylvester equation solver 

95 trsyl, = get_lapack_funcs(('trsyl',), (r, s, f)) 

96 if trsyl is None: 

97 raise RuntimeError('LAPACK implementation does not contain a proper ' 

98 'Sylvester equation solver (TRSYL)') 

99 y, scale, info = trsyl(r, s, f, tranb='C') 

100 

101 y = scale*y 

102 

103 if info < 0: 

104 raise LinAlgError("Illegal value encountered in " 

105 "the %d term" % (-info,)) 

106 

107 return np.dot(np.dot(u, y), v.conj().transpose())

110def solve_continuous_lyapunov(a, q): 

111 """ 

112 Solves the continuous Lyapunov equation :math:`AX + XA^H = Q`. 

113 

114 Uses the Bartels-Stewart algorithm to find :math:`X`. 

115 

116 Parameters 

117 ---------- 

118 a : array_like 

119 A square matrix 

120 

121 q : array_like 

122 Right-hand side square matrix 

123 

124 Returns 

125 ------- 

126 x : ndarray 

127 Solution to the continuous Lyapunov equation 

128 

129 See Also 

130 -------- 

131 solve_discrete_lyapunov : computes the solution to the discrete-time 

132 Lyapunov equation 

133 solve_sylvester : computes the solution to the Sylvester equation 

134 

135 Notes 

136 ----- 

137 The continuous Lyapunov equation is a special form of the Sylvester 

138 equation, hence this solver relies on LAPACK routine ?TRSYL. 

139 

140 .. versionadded:: 0.11.0 

141 

142 Examples 

143 -------- 

144 Given `a` and `q` solve for `x`: 

145 

146 >>> from scipy import linalg 

147 >>> a = np.array([[-3, -2, 0], [-1, -1, 0], [0, -5, -1]]) 

148 >>> b = np.array([2, 4, -1]) 

149 >>> q = np.eye(3) 

150 >>> x = linalg.solve_continuous_lyapunov(a, q) 

151 >>> x 

152 array([[ -0.75 , 0.875 , -3.75 ], 

153 [ 0.875 , -1.375 , 5.3125], 

154 [ -3.75 , 5.3125, -27.0625]]) 

155 >>> np.allclose(a.dot(x) + x.dot(a.T), q) 

156 True 

157 """ 

158 

159 a = np.atleast_2d(_asarray_validated(a, check_finite=True)) 

160 q = np.atleast_2d(_asarray_validated(q, check_finite=True)) 

161 

162 r_or_c = float 

163 

164 for ind, _ in enumerate((a, q)): 

165 if np.iscomplexobj(_): 

166 r_or_c = complex 

167 

168 if not np.equal(*_.shape): 

169 raise ValueError("Matrix {} should be square.".format("aq"[ind])) 

170 

171 # Shape consistency check 

172 if a.shape != q.shape: 

173 raise ValueError("Matrix a and q should have the same shape.") 

174 

175 # Compute the Schur decomp form of a 

176 r, u = schur(a, output='real') 

177 

178 # Construct f = u'*q*u 

179 f = u.conj().T.dot(q.dot(u)) 

180 

181 # Call the Sylvester equation solver 

182 trsyl = get_lapack_funcs('trsyl', (r, f)) 

183 

184 dtype_string = 'T' if r_or_c == float else 'C' 

185 y, scale, info = trsyl(r, r, f, tranb=dtype_string) 

186 

187 if info < 0: 

188 raise ValueError('?TRSYL exited with the internal error ' 

189 '"illegal value in argument number {}.". See ' 

190 'LAPACK documentation for the ?TRSYL error codes.' 

191 ''.format(-info)) 

192 elif info == 1: 

193 warnings.warn('Input "a" has an eigenvalue pair whose sum is ' 

194 'very close to or exactly zero. The solution is ' 

195 'obtained via perturbing the coefficients.', 

196 RuntimeWarning) 

197 y *= scale 

198 

199 return u.dot(y).dot(u.conj().T) 

200

203solve_lyapunov = solve_continuous_lyapunov

206def _solve_discrete_lyapunov_direct(a, q): 

207 """ 

208 Solves the discrete Lyapunov equation directly. 

209 

210 This function is called by the `solve_discrete_lyapunov` function with 

211 `method=direct`. It is not supposed to be called directly. 

212 """ 

213 

214 lhs = kron(a, a.conj()) 

215 lhs = np.eye(lhs.shape[0]) - lhs 

216 x = solve(lhs, q.flatten()) 

217 

218 return np.reshape(x, q.shape)

221def _solve_discrete_lyapunov_bilinear(a, q): 

222 """ 

223 Solves the discrete Lyapunov equation using a bilinear transformation. 

224 

225 This function is called by the `solve_discrete_lyapunov` function with 

226 `method=bilinear`. It is not supposed to be called directly. 

227 """ 

228 eye = np.eye(a.shape[0]) 

229 aH = a.conj().transpose() 

230 aHI_inv = inv(aH + eye) 

231 b = np.dot(aH - eye, aHI_inv) 

232 c = 2*np.dot(np.dot(inv(a + eye), q), aHI_inv) 

233 return solve_lyapunov(b.conj().transpose(), -c)

236def solve_discrete_lyapunov(a, q, method=None): 

237 """ 

238 Solves the discrete Lyapunov equation :math:`AXA^H - X + Q = 0`. 

239 

240 Parameters 

241 ---------- 

242 a, q : (M, M) array_like 

243 Square matrices corresponding to A and Q in the equation 

244 above respectively. Must have the same shape. 

245 

246 method : {'direct', 'bilinear'}, optional 

247 Type of solver. 

248 

249 If not given, chosen to be ``direct`` if ``M`` is less than 10 and 

250 ``bilinear`` otherwise. 

251 

252 Returns 

253 ------- 

254 x : ndarray 

255 Solution to the discrete Lyapunov equation 

256 

257 See Also 

258 -------- 

259 solve_continuous_lyapunov : computes the solution to the continuous-time 

260 Lyapunov equation 

261 

262 Notes 

263 ----- 

264 This section describes the available solvers that can be selected by the 

265 'method' parameter. The default method is *direct* if ``M`` is less than 10 

266 and ``bilinear`` otherwise. 

267 

268 Method *direct* uses a direct analytical solution to the discrete Lyapunov 

269 equation. The algorithm is given in, for example, [1]_. However it requires 

270 the linear solution of a system with dimension :math:`M^2` so that 

271 performance degrades rapidly for even moderately sized matrices. 

272 

273 Method *bilinear* uses a bilinear transformation to convert the discrete 

274 Lyapunov equation to a continuous Lyapunov equation :math:`(BX+XB'=-C)` 

275 where :math:`B=(A-I)(A+I)^{-1}` and 

276 :math:`C=2(A' + I)^{-1} Q (A + I)^{-1}`. The continuous equation can be 

277 efficiently solved since it is a special case of a Sylvester equation. 

278 The transformation algorithm is from Popov (1964) as described in [2]_. 

279 

280 .. versionadded:: 0.11.0 

281 

282 References 

283 ---------- 

284 .. [1] Hamilton, James D. Time Series Analysis, Princeton: Princeton 

285 University Press, 1994. 265. Print. 

286 http://doc1.lbfl.li/aca/FLMF037168.pdf 

287 .. [2] Gajic, Z., and M.T.J. Qureshi. 2008. 

288 Lyapunov Matrix Equation in System Stability and Control. 

289 Dover Books on Engineering Series. Dover Publications. 

290 

291 Examples 

292 -------- 

293 Given `a` and `q` solve for `x`: 

294 

295 >>> from scipy import linalg 

296 >>> a = np.array([[0.2, 0.5],[0.7, -0.9]]) 

297 >>> q = np.eye(2) 

298 >>> x = linalg.solve_discrete_lyapunov(a, q) 

299 >>> x 

300 array([[ 0.70872893, 1.43518822], 

301 [ 1.43518822, -2.4266315 ]]) 

302 >>> np.allclose(a.dot(x).dot(a.T)-x, -q) 

303 True 

304 

305 """ 

306 a = np.asarray(a) 

307 q = np.asarray(q) 

308 if method is None: 

309 # Select automatically based on size of matrices 

310 if a.shape[0] >= 10: 

311 method = 'bilinear' 

312 else: 

313 method = 'direct' 

314 

315 meth = method.lower() 

316 

317 if meth == 'direct': 

318 x = _solve_discrete_lyapunov_direct(a, q) 

319 elif meth == 'bilinear': 

320 x = _solve_discrete_lyapunov_bilinear(a, q) 

321 else: 

322 raise ValueError('Unknown solver %s' % method) 

323 

324 return x

327def solve_continuous_are(a, b, q, r, e=None, s=None, balanced=True): 

328 r""" 

329 Solves the continuous-time algebraic Riccati equation (CARE). 

330 

331 The CARE is defined as 

332 

333 .. math:: 

334 

335 X A + A^H X - X B R^{-1} B^H X + Q = 0 

336 

337 The limitations for a solution to exist are : 

338 

339 * All eigenvalues of :math:`A` on the right half plane, should be 

340 controllable. 

341 

342 * The associated hamiltonian pencil (See Notes), should have 

343 eigenvalues sufficiently away from the imaginary axis. 

344 

345 Moreover, if ``e`` or ``s`` is not precisely ``None``, then the 

346 generalized version of CARE 

347 

348 .. math:: 

349 

350 E^HXA + A^HXE - (E^HXB + S) R^{-1} (B^HXE + S^H) + Q = 0 

351 

352 is solved. When omitted, ``e`` is assumed to be the identity and ``s`` 

353 is assumed to be the zero matrix with sizes compatible with ``a`` and 

354 ``b`` respectively. 

355 

356 Parameters 

357 ---------- 

358 a : (M, M) array_like 

359 Square matrix 

360 b : (M, N) array_like 

361 Input 

362 q : (M, M) array_like 

363 Input 

364 r : (N, N) array_like 

365 Nonsingular square matrix 

366 e : (M, M) array_like, optional 

367 Nonsingular square matrix 

368 s : (M, N) array_like, optional 

369 Input 

370 balanced : bool, optional 

371 The boolean that indicates whether a balancing step is performed 

372 on the data. The default is set to True. 

373 

374 Returns 

375 ------- 

376 x : (M, M) ndarray 

377 Solution to the continuous-time algebraic Riccati equation. 

378 

379 Raises 

380 ------ 

381 LinAlgError 

382 For cases where the stable subspace of the pencil could not be 

383 isolated. See Notes section and the references for details. 

384 

385 See Also 

386 -------- 

387 solve_discrete_are : Solves the discrete-time algebraic Riccati equation 

388 

389 Notes 

390 ----- 

391 The equation is solved by forming the extended hamiltonian matrix pencil, 

392 as described in [1]_, :math:`H - \lambda J` given by the block matrices :: 

393 

394 [ A 0 B ] [ E 0 0 ] 

395 [-Q -A^H -S ] - \lambda * [ 0 E^H 0 ] 

396 [ S^H B^H R ] [ 0 0 0 ] 

397 

398 and using a QZ decomposition method. 

399 

400 In this algorithm, the fail conditions are linked to the symmetry 

401 of the product :math:`U_2 U_1^{-1}` and condition number of 

402 :math:`U_1`. Here, :math:`U` is the 2m-by-m matrix that holds the 

403 eigenvectors spanning the stable subspace with 2m rows and partitioned 

404 into two m-row matrices. See [1]_ and [2]_ for more details. 

405 

406 In order to improve the QZ decomposition accuracy, the pencil goes 

407 through a balancing step where the sum of absolute values of 

408 :math:`H` and :math:`J` entries (after removing the diagonal entries of 

409 the sum) is balanced following the recipe given in [3]_. 

410 

411 .. versionadded:: 0.11.0 

412 

413 References 

414 ---------- 

415 .. [1] P. van Dooren , "A Generalized Eigenvalue Approach For Solving 

416 Riccati Equations.", SIAM Journal on Scientific and Statistical 

417 Computing, Vol.2(2), DOI: 10.1137/0902010 

418 

419 .. [2] A.J. Laub, "A Schur Method for Solving Algebraic Riccati 

420 Equations.", Massachusetts Institute of Technology. Laboratory for 

421 Information and Decision Systems. LIDS-R ; 859. Available online : 

422 http://hdl.handle.net/1721.1/1301 

423 

424 .. [3] P. Benner, "Symplectic Balancing of Hamiltonian Matrices", 2001, 

425 SIAM J. Sci. Comput., 2001, Vol.22(5), DOI: 10.1137/S1064827500367993 

426 

427 Examples 

428 -------- 

429 Given `a`, `b`, `q`, and `r` solve for `x`: 

430 

431 >>> from scipy import linalg 

432 >>> a = np.array([[4, 3], [-4.5, -3.5]]) 

433 >>> b = np.array([[1], [-1]]) 

434 >>> q = np.array([[9, 6], [6, 4.]]) 

435 >>> r = 1 

436 >>> x = linalg.solve_continuous_are(a, b, q, r) 

437 >>> x 

438 array([[ 21.72792206, 14.48528137], 

439 [ 14.48528137, 9.65685425]]) 

440 >>> np.allclose(a.T.dot(x) + x.dot(a)-x.dot(b).dot(b.T).dot(x), -q) 

441 True 

442 

443 """ 

444 

445 # Validate input arguments 

446 a, b, q, r, e, s, m, n, r_or_c, gen_are = _are_validate_args( 

447 a, b, q, r, e, s, 'care') 

448 

449 H = np.empty((2*m+n, 2*m+n), dtype=r_or_c) 

450 H[:m, :m] = a 

451 H[:m, m:2*m] = 0. 

452 H[:m, 2*m:] = b 

453 H[m:2*m, :m] = -q 

454 H[m:2*m, m:2*m] = -a.conj().T 

455 H[m:2*m, 2*m:] = 0. if s is None else -s 

456 H[2*m:, :m] = 0. if s is None else s.conj().T 

457 H[2*m:, m:2*m] = b.conj().T 

458 H[2*m:, 2*m:] = r 

459 

460 if gen_are and e is not None: 

461 J = block_diag(e, e.conj().T, np.zeros_like(r, dtype=r_or_c)) 

462 else: 

463 J = block_diag(np.eye(2*m), np.zeros_like(r, dtype=r_or_c)) 

464 

465 if balanced: 

466 # xGEBAL does not remove the diagonals before scaling. Also 

467 # to avoid destroying the Symplectic structure, we follow Ref.3 

468 M = np.abs(H) + np.abs(J) 

469 M[np.diag_indices_from(M)] = 0. 

470 _, (sca, _) = matrix_balance(M, separate=1, permute=0) 

471 # do we need to bother? 

472 if not np.allclose(sca, np.ones_like(sca)): 

473 # Now impose diag(D,inv(D)) from Benner where D is 

474 # square root of s_i/s_(n+i) for i=0,.... 

475 sca = np.log2(sca) 

476 # NOTE: Py3 uses "Bankers Rounding: round to the nearest even" !! 

477 s = np.round((sca[m:2*m] - sca[:m])/2) 

478 sca = 2 ** np.r_[s, -s, sca[2*m:]] 

479 # Elementwise multiplication via broadcasting. 

480 elwisescale = sca[:, None] * np.reciprocal(sca) 

481 H *= elwisescale 

482 J *= elwisescale 

483 

484 # Deflate the pencil to 2m x 2m ala Ref.1, eq.(55) 

485 q, r = qr(H[:, -n:]) 

486 H = q[:, n:].conj().T.dot(H[:, :2*m]) 

487 J = q[:2*m, n:].conj().T.dot(J[:2*m, :2*m]) 

488 

489 # Decide on which output type is needed for QZ 

490 out_str = 'real' if r_or_c == float else 'complex' 

491 

492 _, _, _, _, _, u = ordqz(H, J, sort='lhp', overwrite_a=True, 

493 overwrite_b=True, check_finite=False, 

494 output=out_str) 

495 

496 # Get the relevant parts of the stable subspace basis 

497 if e is not None: 

498 u, _ = qr(np.vstack((e.dot(u[:m, :m]), u[m:, :m]))) 

499 u00 = u[:m, :m] 

500 u10 = u[m:, :m] 

501 

502 # Solve via back-substituion after checking the condition of u00 

503 up, ul, uu = lu(u00) 

504 if 1/cond(uu) < np.spacing(1.): 

505 raise LinAlgError('Failed to find a finite solution.') 

506 

507 # Exploit the triangular structure 

508 x = solve_triangular(ul.conj().T, 

509 solve_triangular(uu.conj().T, 

510 u10.conj().T, 

511 lower=True), 

512 unit_diagonal=True, 

513 ).conj().T.dot(up.conj().T) 

514 if balanced: 

515 x *= sca[:m, None] * sca[:m] 

516 

517 # Check the deviation from symmetry for lack of success 

518 # See proof of Thm.5 item 3 in [2] 

519 u_sym = u00.conj().T.dot(u10) 

520 n_u_sym = norm(u_sym, 1) 

521 u_sym = u_sym - u_sym.conj().T 

522 sym_threshold = np.max([np.spacing(1000.), 0.1*n_u_sym]) 

523 

524 if norm(u_sym, 1) > sym_threshold: 

525 raise LinAlgError('The associated Hamiltonian pencil has eigenvalues ' 

526 'too close to the imaginary axis') 

527 

528 return (x + x.conj().T)/2

531def solve_discrete_are(a, b, q, r, e=None, s=None, balanced=True): 

532 r""" 

533 Solves the discrete-time algebraic Riccati equation (DARE). 

534 

535 The DARE is defined as 

536 

537 .. math:: 

538 

539 A^HXA - X - (A^HXB) (R + B^HXB)^{-1} (B^HXA) + Q = 0 

540 

541 The limitations for a solution to exist are : 

542 

543 * All eigenvalues of :math:`A` outside the unit disc, should be 

544 controllable. 

545 

546 * The associated symplectic pencil (See Notes), should have 

547 eigenvalues sufficiently away from the unit circle. 

548 

549 Moreover, if ``e`` and ``s`` are not both precisely ``None``, then the 

550 generalized version of DARE 

551 

552 .. math:: 

553 

554 A^HXA - E^HXE - (A^HXB+S) (R+B^HXB)^{-1} (B^HXA+S^H) + Q = 0 

555 

556 is solved. When omitted, ``e`` is assumed to be the identity and ``s`` 

557 is assumed to be the zero matrix. 

558 

559 Parameters 

560 ---------- 

561 a : (M, M) array_like 

562 Square matrix 

563 b : (M, N) array_like 

564 Input 

565 q : (M, M) array_like 

566 Input 

567 r : (N, N) array_like 

568 Square matrix 

569 e : (M, M) array_like, optional 

570 Nonsingular square matrix 

571 s : (M, N) array_like, optional 

572 Input 

573 balanced : bool 

574 The boolean that indicates whether a balancing step is performed 

575 on the data. The default is set to True. 

576 

577 Returns 

578 ------- 

579 x : (M, M) ndarray 

580 Solution to the discrete algebraic Riccati equation. 

581 

582 Raises 

583 ------ 

584 LinAlgError 

585 For cases where the stable subspace of the pencil could not be 

586 isolated. See Notes section and the references for details. 

587 

588 See Also 

589 -------- 

590 solve_continuous_are : Solves the continuous algebraic Riccati equation 

591 

592 Notes 

593 ----- 

594 The equation is solved by forming the extended symplectic matrix pencil, 

595 as described in [1]_, :math:`H - \lambda J` given by the block matrices :: 

596 

597 [ A 0 B ] [ E 0 B ] 

598 [ -Q E^H -S ] - \lambda * [ 0 A^H 0 ] 

599 [ S^H 0 R ] [ 0 -B^H 0 ] 

600 

601 and using a QZ decomposition method. 

602 

603 In this algorithm, the fail conditions are linked to the symmetry 

604 of the product :math:`U_2 U_1^{-1}` and condition number of 

605 :math:`U_1`. Here, :math:`U` is the 2m-by-m matrix that holds the 

606 eigenvectors spanning the stable subspace with 2m rows and partitioned 

607 into two m-row matrices. See [1]_ and [2]_ for more details. 

608 

609 In order to improve the QZ decomposition accuracy, the pencil goes 

610 through a balancing step where the sum of absolute values of 

611 :math:`H` and :math:`J` rows/cols (after removing the diagonal entries) 

612 is balanced following the recipe given in [3]_. If the data has small 

613 numerical noise, balancing may amplify their effects and some clean up 

614 is required. 

615 

616 .. versionadded:: 0.11.0 

617 

618 References 

619 ---------- 

620 .. [1] P. van Dooren , "A Generalized Eigenvalue Approach For Solving 

621 Riccati Equations.", SIAM Journal on Scientific and Statistical 

622 Computing, Vol.2(2), DOI: 10.1137/0902010 

623 

624 .. [2] A.J. Laub, "A Schur Method for Solving Algebraic Riccati 

625 Equations.", Massachusetts Institute of Technology. Laboratory for 

626 Information and Decision Systems. LIDS-R ; 859. Available online : 

627 http://hdl.handle.net/1721.1/1301 

628 

629 .. [3] P. Benner, "Symplectic Balancing of Hamiltonian Matrices", 2001, 

630 SIAM J. Sci. Comput., 2001, Vol.22(5), DOI: 10.1137/S1064827500367993 

631 

632 Examples 

633 -------- 

634 Given `a`, `b`, `q`, and `r` solve for `x`: 

635 

636 >>> from scipy import linalg as la 

637 >>> a = np.array([[0, 1], [0, -1]]) 

638 >>> b = np.array([[1, 0], [2, 1]]) 

639 >>> q = np.array([[-4, -4], [-4, 7]]) 

640 >>> r = np.array([[9, 3], [3, 1]]) 

641 >>> x = la.solve_discrete_are(a, b, q, r) 

642 >>> x 

643 array([[-4., -4.], 

644 [-4., 7.]]) 

645 >>> R = la.solve(r + b.T.dot(x).dot(b), b.T.dot(x).dot(a)) 

646 >>> np.allclose(a.T.dot(x).dot(a) - x - a.T.dot(x).dot(b).dot(R), -q) 

647 True 

648 

649 """ 

650 

651 # Validate input arguments 

652 a, b, q, r, e, s, m, n, r_or_c, gen_are = _are_validate_args( 

653 a, b, q, r, e, s, 'dare') 

654 

655 # Form the matrix pencil 

656 H = np.zeros((2*m+n, 2*m+n), dtype=r_or_c) 

657 H[:m, :m] = a 

658 H[:m, 2*m:] = b 

659 H[m:2*m, :m] = -q 

660 H[m:2*m, m:2*m] = np.eye(m) if e is None else e.conj().T 

661 H[m:2*m, 2*m:] = 0. if s is None else -s 

662 H[2*m:, :m] = 0. if s is None else s.conj().T 

663 H[2*m:, 2*m:] = r 

664 

665 J = np.zeros_like(H, dtype=r_or_c) 

666 J[:m, :m] = np.eye(m) if e is None else e 

667 J[m:2*m, m:2*m] = a.conj().T 

668 J[2*m:, m:2*m] = -b.conj().T 

669 

670 if balanced: 

671 # xGEBAL does not remove the diagonals before scaling. Also 

672 # to avoid destroying the Symplectic structure, we follow Ref.3 

673 M = np.abs(H) + np.abs(J) 

674 M[np.diag_indices_from(M)] = 0. 

675 _, (sca, _) = matrix_balance(M, separate=1, permute=0) 

676 # do we need to bother? 

677 if not np.allclose(sca, np.ones_like(sca)): 

678 # Now impose diag(D,inv(D)) from Benner where D is 

679 # square root of s_i/s_(n+i) for i=0,.... 

680 sca = np.log2(sca) 

681 # NOTE: Py3 uses "Bankers Rounding: round to the nearest even" !! 

682 s = np.round((sca[m:2*m] - sca[:m])/2) 

683 sca = 2 ** np.r_[s, -s, sca[2*m:]] 

684 # Elementwise multiplication via broadcasting. 

685 elwisescale = sca[:, None] * np.reciprocal(sca) 

686 H *= elwisescale 

687 J *= elwisescale 

688 

689 # Deflate the pencil by the R column ala Ref.1 

690 q_of_qr, _ = qr(H[:, -n:]) 

691 H = q_of_qr[:, n:].conj().T.dot(H[:, :2*m]) 

692 J = q_of_qr[:, n:].conj().T.dot(J[:, :2*m]) 

693 

694 # Decide on which output type is needed for QZ 

695 out_str = 'real' if r_or_c == float else 'complex' 

696 

697 _, _, _, _, _, u = ordqz(H, J, sort='iuc', 

698 overwrite_a=True, 

699 overwrite_b=True, 

700 check_finite=False, 

701 output=out_str) 

702 

703 # Get the relevant parts of the stable subspace basis 

704 if e is not None: 

705 u, _ = qr(np.vstack((e.dot(u[:m, :m]), u[m:, :m]))) 

706 u00 = u[:m, :m] 

707 u10 = u[m:, :m] 

708 

709 # Solve via back-substituion after checking the condition of u00 

710 up, ul, uu = lu(u00) 

711 

712 if 1/cond(uu) < np.spacing(1.): 

713 raise LinAlgError('Failed to find a finite solution.') 

714 

715 # Exploit the triangular structure 

716 x = solve_triangular(ul.conj().T, 

717 solve_triangular(uu.conj().T, 

718 u10.conj().T, 

719 lower=True), 

720 unit_diagonal=True, 

721 ).conj().T.dot(up.conj().T) 

722 if balanced: 

723 x *= sca[:m, None] * sca[:m] 

724 

725 # Check the deviation from symmetry for lack of success 

726 # See proof of Thm.5 item 3 in [2] 

727 u_sym = u00.conj().T.dot(u10) 

728 n_u_sym = norm(u_sym, 1) 

729 u_sym = u_sym - u_sym.conj().T 

730 sym_threshold = np.max([np.spacing(1000.), 0.1*n_u_sym]) 

731 

732 if norm(u_sym, 1) > sym_threshold: 

733 raise LinAlgError('The associated symplectic pencil has eigenvalues' 

734 'too close to the unit circle') 

735 

736 return (x + x.conj().T)/2 

737

739def _are_validate_args(a, b, q, r, e, s, eq_type='care'): 

740 """ 

741 A helper function to validate the arguments supplied to the 

742 Riccati equation solvers. Any discrepancy found in the input 

743 matrices leads to a ``ValueError`` exception. 

744 

745 Essentially, it performs: 

746 

747 - a check whether the input is free of NaN and Infs. 

748 - a pass for the data through ``numpy.atleast_2d()`` 

749 - squareness check of the relevant arrays, 

750 - shape consistency check of the arrays, 

751 - singularity check of the relevant arrays, 

752 - symmetricity check of the relevant matrices, 

753 - a check whether the regular or the generalized version is asked. 

754 

755 This function is used by ``solve_continuous_are`` and 

756 ``solve_discrete_are``. 

757 

758 Parameters 

759 ---------- 

760 a, b, q, r, e, s : array_like 

761 Input data 

762 eq_type : str 

763 Accepted arguments are 'care' and 'dare'. 

764 

765 Returns 

766 ------- 

767 a, b, q, r, e, s : ndarray 

768 Regularized input data 

769 m, n : int 

770 shape of the problem 

771 r_or_c : type 

772 Data type of the problem, returns float or complex 

773 gen_or_not : bool 

774 Type of the equation, True for generalized and False for regular ARE. 

775 

776 """ 

777 

778 if not eq_type.lower() in ('dare', 'care'): 

779 raise ValueError("Equation type unknown. " 

780 "Only 'care' and 'dare' is understood") 

781 

782 a = np.atleast_2d(_asarray_validated(a, check_finite=True)) 

783 b = np.atleast_2d(_asarray_validated(b, check_finite=True)) 

784 q = np.atleast_2d(_asarray_validated(q, check_finite=True)) 

785 r = np.atleast_2d(_asarray_validated(r, check_finite=True)) 

786 

787 # Get the correct data types otherwise NumPy complains 

788 # about pushing complex numbers into real arrays. 

789 r_or_c = complex if np.iscomplexobj(b) else float 

790 

791 for ind, mat in enumerate((a, q, r)): 

792 if np.iscomplexobj(mat): 

793 r_or_c = complex 

794 

795 if not np.equal(*mat.shape): 

796 raise ValueError("Matrix {} should be square.".format("aqr"[ind])) 

797 

798 # Shape consistency checks 

799 m, n = b.shape 

800 if m != a.shape[0]: 

801 raise ValueError("Matrix a and b should have the same number of rows.") 

802 if m != q.shape[0]: 

803 raise ValueError("Matrix a and q should have the same shape.") 

804 if n != r.shape[0]: 

805 raise ValueError("Matrix b and r should have the same number of cols.") 

806 

807 # Check if the data matrices q, r are (sufficiently) hermitian 

808 for ind, mat in enumerate((q, r)): 

809 if norm(mat - mat.conj().T, 1) > np.spacing(norm(mat, 1))*100: 

810 raise ValueError("Matrix {} should be symmetric/hermitian." 

811 "".format("qr"[ind])) 

812 

813 # Continuous time ARE should have a nonsingular r matrix. 

814 if eq_type == 'care': 

815 min_sv = svd(r, compute_uv=False)[-1] 

816 if min_sv == 0. or min_sv < np.spacing(1.)*norm(r, 1): 

817 raise ValueError('Matrix r is numerically singular.') 

818 

819 # Check if the generalized case is required with omitted arguments 

820 # perform late shape checking etc. 

821 generalized_case = e is not None or s is not None 

822 

823 if generalized_case: 

824 if e is not None: 

825 e = np.atleast_2d(_asarray_validated(e, check_finite=True)) 

826 if not np.equal(*e.shape): 

827 raise ValueError("Matrix e should be square.") 

828 if m != e.shape[0]: 

829 raise ValueError("Matrix a and e should have the same shape.") 

830 # numpy.linalg.cond doesn't check for exact zeros and 

831 # emits a runtime warning. Hence the following manual check. 

832 min_sv = svd(e, compute_uv=False)[-1] 

833 if min_sv == 0. or min_sv < np.spacing(1.) * norm(e, 1): 

834 raise ValueError('Matrix e is numerically singular.') 

835 if np.iscomplexobj(e): 

836 r_or_c = complex 

837 if s is not None: 

838 s = np.atleast_2d(_asarray_validated(s, check_finite=True)) 

839 if s.shape != b.shape: 

840 raise ValueError("Matrix b and s should have the same shape.") 

841 if np.iscomplexobj(s): 

842 r_or_c = complex 

843 

844 return a, b, q, r, e, s, m, n, r_or_c, generalized_case
