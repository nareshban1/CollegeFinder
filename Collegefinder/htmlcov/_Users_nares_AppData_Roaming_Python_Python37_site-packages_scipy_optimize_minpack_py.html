27 if (output_shape is not None) and (shape(res) != output_shape): 

28 if (output_shape[0] != 1):

36 msg += " '%s'." % func_name

51 """ 

52 Find the roots of a function. 

53 

54 Return the roots of the (non-linear) equations defined by 

55 ``func(x) = 0`` given a starting estimate. 

56 

57 Parameters 

58 ----------

266def leastsq(func, x0, args=(), Dfun=None, full_output=0, 

267 col_deriv=0, ftol=1.49012e-8, xtol=1.49012e-8, 

268 gtol=0.0, maxfev=0, epsfcn=None, factor=100, diag=None): 

269 """ 

270 Minimize the sum of squares of a set of equations. 

271 

272 :: 

273 

274 x = arg min(sum(func(y)**2,axis=0)) 

275 y 

276 

277 Parameters 

278 ---------- 

279 func : callable 

280 should take at least one (possibly length N vector) argument and 

281 returns M floating point numbers. It must not return NaNs or 

282 fitting might fail. 

283 x0 : ndarray 

284 The starting estimate for the minimization. 

285 args : tuple, optional 

286 Any extra arguments to func are placed in this tuple. 

287 Dfun : callable, optional 

288 A function or method to compute the Jacobian of func with derivatives 

289 across the rows. If this is None, the Jacobian will be estimated. 

290 full_output : bool, optional 

291 non-zero to return all optional outputs. 

292 col_deriv : bool, optional 

293 non-zero to specify that the Jacobian function computes derivatives 

294 down the columns (faster, because there is no transpose operation). 

295 ftol : float, optional 

296 Relative error desired in the sum of squares. 

297 xtol : float, optional 

298 Relative error desired in the approximate solution. 

299 gtol : float, optional 

300 Orthogonality desired between the function vector and the columns of 

301 the Jacobian. 

302 maxfev : int, optional 

303 The maximum number of calls to the function. If `Dfun` is provided 

304 then the default `maxfev` is 100*(N+1) where N is the number of elements 

305 in x0, otherwise the default `maxfev` is 200*(N+1). 

306 epsfcn : float, optional 

307 A variable used in determining a suitable step length for the forward- 

308 difference approximation of the Jacobian (for Dfun=None). 

309 Normally the actual step length will be sqrt(epsfcn)*x 

310 If epsfcn is less than the machine precision, it is assumed that the 

311 relative errors are of the order of the machine precision. 

312 factor : float, optional 

313 A parameter determining the initial step bound 

314 (``factor * || diag * x||``). Should be in interval ``(0.1, 100)``. 

315 diag : sequence, optional 

316 N positive entries that serve as a scale factors for the variables. 

317 

318 Returns 

319 ------- 

320 x : ndarray 

321 The solution (or the result of the last iteration for an unsuccessful 

322 call). 

323 cov_x : ndarray 

324 The inverse of the Hessian. `fjac` and `ipvt` are used to construct an 

325 estimate of the Hessian. A value of None indicates a singular matrix, 

326 which means the curvature in parameters `x` is numerically flat. To 

327 obtain the covariance matrix of the parameters `x`, `cov_x` must be 

328 multiplied by the variance of the residuals -- see curve_fit. 

329 infodict : dict 

330 a dictionary of optional outputs with the keys: 

331 

332 ``nfev`` 

333 The number of function calls 

334 ``fvec`` 

335 The function evaluated at the output 

336 ``fjac`` 

337 A permutation of the R matrix of a QR 

338 factorization of the final approximate 

339 Jacobian matrix, stored column wise. 

340 Together with ipvt, the covariance of the 

341 estimate can be approximated. 

342 ``ipvt`` 

343 An integer array of length N which defines 

344 a permutation matrix, p, such that 

345 fjac*p = q*r, where r is upper triangular 

346 with diagonal elements of nonincreasing 

347 magnitude. Column j of p is column ipvt(j) 

348 of the identity matrix. 

349 ``qtf`` 

350 The vector (transpose(q) * fvec). 

351 

352 mesg : str 

353 A string message giving information about the cause of failure. 

354 ier : int 

355 An integer flag. If it is equal to 1, 2, 3 or 4, the solution was 

356 found. Otherwise, the solution was not found. In either case, the 

357 optional output variable 'mesg' gives more information. 

358 

359 See Also 

360 -------- 

361 least_squares : Newer interface to solve nonlinear least-squares problems 

362 with bounds on the variables. See ``method=='lm'`` in particular. 

363 

364 Notes 

365 ----- 

366 "leastsq" is a wrapper around MINPACK's lmdif and lmder algorithms. 

367 

368 cov_x is a Jacobian approximation to the Hessian of the least squares 

369 objective function. 

370 This approximation assumes that the objective function is based on the 

371 difference between some observed target data (ydata) and a (non-linear) 

372 function of the parameters `f(xdata, params)` :: 

373 

374 func(params) = ydata - f(xdata, params) 

375 

376 so that the objective function is :: 

377 

378 min sum((ydata - f(xdata, params))**2, axis=0) 

379 params 

380 

381 The solution, `x`, is always a 1D array, regardless of the shape of `x0`, 

382 or whether `x0` is a scalar. 

383 """ 

384 x0 = asarray(x0).flatten() 

385 n = len(x0) 

386 if not isinstance(args, tuple): 

387 args = (args,) 

388 shape, dtype = _check_func('leastsq', 'func', func, x0, args, n) 

389 m = shape[0] 

390 

391 if n > m: 

392 raise TypeError('Improper input: N=%s must not exceed M=%s' % (n, m)) 

393 

394 if epsfcn is None: 

395 epsfcn = finfo(dtype).eps 

396 

397 if Dfun is None: 

398 if maxfev == 0: 

399 maxfev = 200*(n + 1) 

400 retval = _minpack._lmdif(func, x0, args, full_output, ftol, xtol, 

401 gtol, maxfev, epsfcn, factor, diag) 

402 else: 

403 if col_deriv: 

404 _check_func('leastsq', 'Dfun', Dfun, x0, args, n, (n, m)) 

405 else: 

406 _check_func('leastsq', 'Dfun', Dfun, x0, args, n, (m, n)) 

407 if maxfev == 0: 

408 maxfev = 100 * (n + 1) 

409 retval = _minpack._lmder(func, Dfun, x0, args, full_output, 

410 col_deriv, ftol, xtol, gtol, maxfev, 

411 factor, diag) 

412 

413 errors = {0: ["Improper input parameters.", TypeError], 

414 1: ["Both actual and predicted relative reductions " 

415 "in the sum of squares\n are at most %f" % ftol, None], 

416 2: ["The relative error between two consecutive " 

417 "iterates is at most %f" % xtol, None], 

418 3: ["Both actual and predicted relative reductions in " 

419 "the sum of squares\n are at most %f and the " 

420 "relative error between two consecutive " 

421 "iterates is at \n most %f" % (ftol, xtol), None], 

422 4: ["The cosine of the angle between func(x) and any " 

423 "column of the\n Jacobian is at most %f in " 

424 "absolute value" % gtol, None], 

425 5: ["Number of calls to function has reached " 

426 "maxfev = %d." % maxfev, ValueError], 

427 6: ["ftol=%f is too small, no further reduction " 

428 "in the sum of squares\n is possible." % ftol, 

429 ValueError], 

430 7: ["xtol=%f is too small, no further improvement in " 

431 "the approximate\n solution is possible." % xtol, 

432 ValueError], 

433 8: ["gtol=%f is too small, func(x) is orthogonal to the " 

434 "columns of\n the Jacobian to machine " 

435 "precision." % gtol, ValueError]} 

436 

437 # The FORTRAN return value (possible return values are >= 0 and <= 8) 

438 info = retval[-1] 

439 

440 if full_output: 

441 cov_x = None 

442 if info in LEASTSQ_SUCCESS: 

443 from numpy.dual import inv 

444 perm = take(eye(n), retval[1]['ipvt'] - 1, 0) 

445 r = triu(transpose(retval[1]['fjac'])[:n, :]) 

446 R = dot(r, perm) 

447 try: 

448 cov_x = inv(dot(transpose(R), R)) 

449 except (LinAlgError, ValueError): 

450 pass 

451 return (retval[0], cov_x) + retval[1:-1] + (errors[info][0], info) 

452 else: 

453 if info in LEASTSQ_FAILURE: 

454 warnings.warn(errors[info][0], RuntimeWarning) 

455 elif info == 0: 

456 raise errors[info][1](errors[info][0]) 

457 return retval[0], info

460def _wrap_func(func, xdata, ydata, transform): 

461 if transform is None: 

462 def func_wrapped(params): 

463 return func(xdata, *params) - ydata 

464 elif transform.ndim == 1: 

465 def func_wrapped(params): 

466 return transform * (func(xdata, *params) - ydata) 

467 else: 

468 # Chisq = (y - yd)^T C^{-1} (y-yd) 

469 # transform = L such that C = L L^T 

470 # C^{-1} = L^{-T} L^{-1} 

471 # Chisq = (y - yd)^T L^{-T} L^{-1} (y-yd) 

472 # Define (y-yd)' = L^{-1} (y-yd) 

473 # by solving 

474 # L (y-yd)' = (y-yd) 

475 # and minimize (y-yd)'^T (y-yd)' 

476 def func_wrapped(params): 

477 return solve_triangular(transform, func(xdata, *params) - ydata, lower=True) 

478 return func_wrapped

494def _initialize_feasible(lb, ub): 

495 p0 = np.ones_like(lb) 

496 lb_finite = np.isfinite(lb) 

497 ub_finite = np.isfinite(ub) 

498 

499 mask = lb_finite & ub_finite 

500 p0[mask] = 0.5 * (lb[mask] + ub[mask]) 

501 

502 mask = lb_finite & ~ub_finite 

503 p0[mask] = lb[mask] + 1 

504 

505 mask = ~lb_finite & ub_finite 

506 p0[mask] = ub[mask] - 1 

507 

508 return p0

511def curve_fit(f, xdata, ydata, p0=None, sigma=None, absolute_sigma=False, 

512 check_finite=True, bounds=(-np.inf, np.inf), method=None, 

513 jac=None, **kwargs): 

514 """ 

515 Use non-linear least squares to fit a function, f, to data. 

516 

517 Assumes ``ydata = f(xdata, *params) + eps`` 

518 

519 Parameters 

520 ---------- 

521 f : callable 

522 The model function, f(x, ...). It must take the independent 

523 variable as the first argument and the parameters to fit as 

524 separate remaining arguments. 

525 xdata : array_like or object 

526 The independent variable where the data is measured. 

527 Should usually be an M-length sequence or an (k,M)-shaped array for 

528 functions with k predictors, but can actually be any object. 

529 ydata : array_like 

530 The dependent data, a length M array - nominally ``f(xdata, ...)``. 

531 p0 : array_like, optional 

532 Initial guess for the parameters (length N). If None, then the 

533 initial values will all be 1 (if the number of parameters for the 

534 function can be determined using introspection, otherwise a 

535 ValueError is raised). 

536 sigma : None or M-length sequence or MxM array, optional 

537 Determines the uncertainty in `ydata`. If we define residuals as 

538 ``r = ydata - f(xdata, *popt)``, then the interpretation of `sigma` 

539 depends on its number of dimensions: 

540 

541 - A 1-d `sigma` should contain values of standard deviations of 

542 errors in `ydata`. In this case, the optimized function is 

543 ``chisq = sum((r / sigma) ** 2)``. 

544 

545 - A 2-d `sigma` should contain the covariance matrix of 

546 errors in `ydata`. In this case, the optimized function is 

547 ``chisq = r.T @ inv(sigma) @ r``. 

548 

549 .. versionadded:: 0.19 

550 

551 None (default) is equivalent of 1-d `sigma` filled with ones. 

552 absolute_sigma : bool, optional 

553 If True, `sigma` is used in an absolute sense and the estimated parameter 

554 covariance `pcov` reflects these absolute values. 

555 

556 If False, only the relative magnitudes of the `sigma` values matter. 

557 The returned parameter covariance matrix `pcov` is based on scaling 

558 `sigma` by a constant factor. This constant is set by demanding that the 

559 reduced `chisq` for the optimal parameters `popt` when using the 

560 *scaled* `sigma` equals unity. In other words, `sigma` is scaled to 

561 match the sample variance of the residuals after the fit. 

562 Mathematically, 

563 ``pcov(absolute_sigma=False) = pcov(absolute_sigma=True) * chisq(popt)/(M-N)`` 

564 check_finite : bool, optional 

565 If True, check that the input arrays do not contain nans of infs, 

566 and raise a ValueError if they do. Setting this parameter to 

567 False may silently produce nonsensical results if the input arrays 

568 do contain nans. Default is True. 

569 bounds : 2-tuple of array_like, optional 

570 Lower and upper bounds on parameters. Defaults to no bounds. 

571 Each element of the tuple must be either an array with the length equal 

572 to the number of parameters, or a scalar (in which case the bound is 

573 taken to be the same for all parameters.) Use ``np.inf`` with an 

574 appropriate sign to disable bounds on all or some parameters. 

575 

576 .. versionadded:: 0.17 

577 method : {'lm', 'trf', 'dogbox'}, optional 

578 Method to use for optimization. See `least_squares` for more details. 

579 Default is 'lm' for unconstrained problems and 'trf' if `bounds` are 

580 provided. The method 'lm' won't work when the number of observations 

581 is less than the number of variables, use 'trf' or 'dogbox' in this 

582 case. 

583 

584 .. versionadded:: 0.17 

585 jac : callable, string or None, optional 

586 Function with signature ``jac(x, ...)`` which computes the Jacobian 

587 matrix of the model function with respect to parameters as a dense 

588 array_like structure. It will be scaled according to provided `sigma`. 

589 If None (default), the Jacobian will be estimated numerically. 

590 String keywords for 'trf' and 'dogbox' methods can be used to select 

591 a finite difference scheme, see `least_squares`. 

592 

593 .. versionadded:: 0.18 

594 kwargs 

595 Keyword arguments passed to `leastsq` for ``method='lm'`` or 

596 `least_squares` otherwise. 

597 

598 Returns 

599 ------- 

600 popt : array 

601 Optimal values for the parameters so that the sum of the squared 

602 residuals of ``f(xdata, *popt) - ydata`` is minimized 

603 pcov : 2d array 

604 The estimated covariance of popt. The diagonals provide the variance 

605 of the parameter estimate. To compute one standard deviation errors 

606 on the parameters use ``perr = np.sqrt(np.diag(pcov))``. 

607 

608 How the `sigma` parameter affects the estimated covariance 

609 depends on `absolute_sigma` argument, as described above. 

610 

611 If the Jacobian matrix at the solution doesn't have a full rank, then 

612 'lm' method returns a matrix filled with ``np.inf``, on the other hand 

613 'trf' and 'dogbox' methods use Moore-Penrose pseudoinverse to compute 

614 the covariance matrix. 

615 

616 Raises 

617 ------ 

618 ValueError 

619 if either `ydata` or `xdata` contain NaNs, or if incompatible options 

620 are used. 

621 

622 RuntimeError 

623 if the least-squares minimization fails. 

624 

625 OptimizeWarning 

626 if covariance of the parameters can not be estimated. 

627 

628 See Also 

629 -------- 

630 least_squares : Minimize the sum of squares of nonlinear functions. 

631 scipy.stats.linregress : Calculate a linear least squares regression for 

632 two sets of measurements. 

633 

634 Notes 

635 ----- 

636 With ``method='lm'``, the algorithm uses the Levenberg-Marquardt algorithm 

637 through `leastsq`. Note that this algorithm can only deal with 

638 unconstrained problems. 

639 

640 Box constraints can be handled by methods 'trf' and 'dogbox'. Refer to 

641 the docstring of `least_squares` for more information. 

642 

643 Examples 

644 -------- 

645 >>> import matplotlib.pyplot as plt 

646 >>> from scipy.optimize import curve_fit 

647 

648 >>> def func(x, a, b, c): 

649 ... return a * np.exp(-b * x) + c 

650 

651 Define the data to be fit with some noise: 

652 

653 >>> xdata = np.linspace(0, 4, 50) 

654 >>> y = func(xdata, 2.5, 1.3, 0.5) 

655 >>> np.random.seed(1729) 

656 >>> y_noise = 0.2 * np.random.normal(size=xdata.size) 

657 >>> ydata = y + y_noise 

658 >>> plt.plot(xdata, ydata, 'b-', label='data') 

659 

660 Fit for the parameters a, b, c of the function `func`: 

661 

662 >>> popt, pcov = curve_fit(func, xdata, ydata) 

663 >>> popt 

664 array([ 2.55423706, 1.35190947, 0.47450618]) 

665 >>> plt.plot(xdata, func(xdata, *popt), 'r-', 

666 ... label='fit: a=%5.3f, b=%5.3f, c=%5.3f' % tuple(popt)) 

667 

668 Constrain the optimization to the region of ``0 <= a <= 3``, 

669 ``0 <= b <= 1`` and ``0 <= c <= 0.5``: 

670 

671 >>> popt, pcov = curve_fit(func, xdata, ydata, bounds=(0, [3., 1., 0.5])) 

672 >>> popt 

673 array([ 2.43708906, 1. , 0.35015434]) 

674 >>> plt.plot(xdata, func(xdata, *popt), 'g--', 

675 ... label='fit: a=%5.3f, b=%5.3f, c=%5.3f' % tuple(popt)) 

676 

677 >>> plt.xlabel('x') 

678 >>> plt.ylabel('y') 

679 >>> plt.legend() 

680 >>> plt.show() 

681 

682 """ 

683 if p0 is None: 

684 # determine number of parameters by inspecting the function 

685 from scipy._lib._util import getargspec_no_self as _getargspec 

686 args, varargs, varkw, defaults = _getargspec(f) 

687 if len(args) < 2: 

688 raise ValueError("Unable to determine number of fit parameters.") 

689 n = len(args) - 1 

690 else: 

691 p0 = np.atleast_1d(p0) 

692 n = p0.size 

693 

694 lb, ub = prepare_bounds(bounds, n) 

695 if p0 is None: 

696 p0 = _initialize_feasible(lb, ub) 

697 

698 bounded_problem = np.any((lb > -np.inf) | (ub < np.inf)) 

699 if method is None: 

700 if bounded_problem: 

701 method = 'trf' 

702 else: 

703 method = 'lm' 

704 

705 if method == 'lm' and bounded_problem: 

706 raise ValueError("Method 'lm' only works for unconstrained problems. " 

707 "Use 'trf' or 'dogbox' instead.") 

708 

709 # optimization may produce garbage for float32 inputs, cast them to float64 

710 

711 # NaNs can not be handled 

712 if check_finite: 

713 ydata = np.asarray_chkfinite(ydata, float) 

714 else: 

715 ydata = np.asarray(ydata, float) 

716 

717 if isinstance(xdata, (list, tuple, np.ndarray)): 

718 # `xdata` is passed straight to the user-defined `f`, so allow 

719 # non-array_like `xdata`. 

720 if check_finite: 

721 xdata = np.asarray_chkfinite(xdata, float) 

722 else: 

723 xdata = np.asarray(xdata, float) 

724 

725 if ydata.size == 0: 

726 raise ValueError("`ydata` must not be empty!") 

727 

728 # Determine type of sigma 

729 if sigma is not None: 

730 sigma = np.asarray(sigma) 

731 

732 # if 1-d, sigma are errors, define transform = 1/sigma 

733 if sigma.shape == (ydata.size, ): 

734 transform = 1.0 / sigma 

735 # if 2-d, sigma is the covariance matrix, 

736 # define transform = L such that L L^T = C 

737 elif sigma.shape == (ydata.size, ydata.size): 

738 try: 

739 # scipy.linalg.cholesky requires lower=True to return L L^T = A 

740 transform = cholesky(sigma, lower=True) 

741 except LinAlgError: 

742 raise ValueError("`sigma` must be positive definite.") 

743 else: 

744 raise ValueError("`sigma` has incorrect shape.") 

745 else: 

746 transform = None 

747 

748 func = _wrap_func(f, xdata, ydata, transform) 

749 if callable(jac): 

750 jac = _wrap_jac(jac, xdata, transform) 

751 elif jac is None and method != 'lm': 

752 jac = '2-point' 

753 

754 if 'args' in kwargs: 

755 # The specification for the model function `f` does not support 

756 # additional arguments. Refer to the `curve_fit` docstring for 

757 # acceptable call signatures of `f`. 

758 raise ValueError("'args' is not a supported keyword argument.") 

759 

760 if method == 'lm': 

761 # Remove full_output from kwargs, otherwise we're passing it in twice. 

762 return_full = kwargs.pop('full_output', False) 

763 res = leastsq(func, p0, Dfun=jac, full_output=1, **kwargs) 

764 popt, pcov, infodict, errmsg, ier = res 

765 ysize = len(infodict['fvec']) 

766 cost = np.sum(infodict['fvec'] ** 2) 

767 if ier not in [1, 2, 3, 4]: 

768 raise RuntimeError("Optimal parameters not found: " + errmsg) 

769 else: 

770 # Rename maxfev (leastsq) to max_nfev (least_squares), if specified. 

771 if 'max_nfev' not in kwargs: 

772 kwargs['max_nfev'] = kwargs.pop('maxfev', None) 

773 

774 res = least_squares(func, p0, jac=jac, bounds=bounds, method=method, 

775 **kwargs) 

776 

777 if not res.success: 

778 raise RuntimeError("Optimal parameters not found: " + res.message) 

779 

780 ysize = len(res.fun) 

781 cost = 2 * res.cost # res.cost is half sum of squares! 

782 popt = res.x 

783 

784 # Do Moore-Penrose inverse discarding zero singular values. 

785 _, s, VT = svd(res.jac, full_matrices=False) 

786 threshold = np.finfo(float).eps * max(res.jac.shape) * s[0] 

787 s = s[s > threshold] 

788 VT = VT[:s.size] 

789 pcov = np.dot(VT.T / s**2, VT) 

790 return_full = False 

791 

792 warn_cov = False 

793 if pcov is None: 

794 # indeterminate covariance 

795 pcov = zeros((len(popt), len(popt)), dtype=float) 

796 pcov.fill(inf) 

797 warn_cov = True 

798 elif not absolute_sigma: 

799 if ysize > p0.size: 

800 s_sq = cost / (ysize - p0.size) 

801 pcov = pcov * s_sq 

802 else: 

803 pcov.fill(inf) 

804 warn_cov = True 

805 

806 if warn_cov: 

807 warnings.warn('Covariance of the parameters could not be estimated', 

808 category=OptimizeWarning) 

809 

810 if return_full: 

811 return popt, pcov, infodict, errmsg, ier 

812 else: 

813 return popt, pcov

816def check_gradient(fcn, Dfcn, x0, args=(), col_deriv=0): 

817 """Perform a simple check on the gradient for correctness. 

818 

819 """ 

820 

821 x = atleast_1d(x0) 

822 n = len(x) 

823 x = x.reshape((n,)) 

824 fvec = atleast_1d(fcn(x, *args)) 

825 m = len(fvec) 

826 fvec = fvec.reshape((m,)) 

827 ldfjac = m 

828 fjac = atleast_1d(Dfcn(x, *args)) 

829 fjac = fjac.reshape((m, n)) 

830 if col_deriv == 0: 

831 fjac = transpose(fjac) 

832 

833 xp = zeros((n,), float) 

834 err = zeros((m,), float) 

835 fvecp = None 

836 _minpack._chkder(m, n, x, fvec, fjac, ldfjac, xp, fvecp, 1, err) 

837 

838 fvecp = atleast_1d(fcn(xp, *args)) 

839 fvecp = fvecp.reshape((m,)) 

840 _minpack._chkder(m, n, x, fvec, fjac, ldfjac, xp, fvecp, 2, err) 

841 

842 good = (prod(greater(err, 0.5), axis=0)) 

843 

844 return (good, err)

847def _del2(p0, p1, d): 

848 return p0 - np.square(p1 - p0) / d 

849 

850 

851def _relerr(actual, desired): 

852 return (actual - desired) / desired 

853 

854 

855def _fixed_point_helper(func, x0, args, xtol, maxiter, use_accel): 

856 p0 = x0 

857 for i in range(maxiter): 

858 p1 = func(p0, *args) 

859 if use_accel: 

860 p2 = func(p1, *args) 

861 d = p2 - 2.0 * p1 + p0 

862 p = _lazywhere(d != 0, (p0, p1, d), f=_del2, fillvalue=p2) 

863 else: 

864 p = p1 

865 relerr = _lazywhere(p0 != 0, (p, p0), f=_relerr, fillvalue=p) 

866 if np.all(np.abs(relerr) < xtol): 

867 return p 

868 p0 = p 

869 msg = "Failed to converge after %d iterations, value is %s" % (maxiter, p) 

870 raise RuntimeError(msg)

873def fixed_point(func, x0, args=(), xtol=1e-8, maxiter=500, method='del2'): 

874 """ 

875 Find a fixed point of the function. 

876 

877 Given a function of one or more variables and a starting point, find a 

878 fixed-point of the function: i.e. where ``func(x0) == x0``. 

879 

880 Parameters 

881 ---------- 

882 func : function 

883 Function to evaluate. 

884 x0 : array_like 

885 Fixed point of function. 

886 args : tuple, optional 

887 Extra arguments to `func`. 

888 xtol : float, optional 

889 Convergence tolerance, defaults to 1e-08. 

890 maxiter : int, optional 

891 Maximum number of iterations, defaults to 500. 

892 method : {"del2", "iteration"}, optional 

893 Method of finding the fixed-point, defaults to "del2" 

894 which uses Steffensen's Method with Aitken's ``Del^2`` 

895 convergence acceleration [1]_. The "iteration" method simply iterates 

896 the function until convergence is detected, without attempting to 

897 accelerate the convergence. 

898 

899 References 

900 ---------- 

901 .. [1] Burden, Faires, "Numerical Analysis", 5th edition, pg. 80 

902 

903 Examples 

904 -------- 

905 >>> from scipy import optimize 

906 >>> def func(x, c1, c2): 

907 ... return np.sqrt(c1/(x+c2)) 

908 >>> c1 = np.array([10,12.]) 

909 >>> c2 = np.array([3, 5.]) 

910 >>> optimize.fixed_point(func, [1.2, 1.3], args=(c1,c2)) 

911 array([ 1.4920333 , 1.37228132]) 

912 

913 """ 

914 use_accel = {'del2': True, 'iteration': False}[method] 

915 x0 = _asarray_validated(x0, as_inexact=True) 

916 return _fixed_point_helper(func, x0, args, xtol, maxiter, use_accel)
