9def logsumexp(a, axis=None, b=None, keepdims=False, return_sign=False): 

10 """Compute the log of the sum of exponentials of input elements. 

11 

12 Parameters 

13 ---------- 

14 a : array_like 

15 Input array. 

16 axis : None or int or tuple of ints, optional 

17 Axis or axes over which the sum is taken. By default `axis` is None, 

18 and all elements are summed. 

19 

20 .. versionadded:: 0.11.0 

21 keepdims : bool, optional 

22 If this is set to True, the axes which are reduced are left in the 

23 result as dimensions with size one. With this option, the result 

24 will broadcast correctly against the original array. 

25 

26 .. versionadded:: 0.15.0 

27 b : array-like, optional 

28 Scaling factor for exp(`a`) must be of the same shape as `a` or 

29 broadcastable to `a`. These values may be negative in order to 

30 implement subtraction. 

31 

32 .. versionadded:: 0.12.0 

33 return_sign : bool, optional 

34 If this is set to True, the result will be a pair containing sign 

35 information; if False, results that are negative will be returned 

36 as NaN. Default is False (no sign information). 

37 

38 .. versionadded:: 0.16.0 

39 

40 Returns 

41 ------- 

42 res : ndarray 

43 The result, ``np.log(np.sum(np.exp(a)))`` calculated in a numerically 

44 more stable way. If `b` is given then ``np.log(np.sum(b*np.exp(a)))`` 

45 is returned. 

46 sgn : ndarray 

47 If return_sign is True, this will be an array of floating-point 

48 numbers matching res and +1, 0, or -1 depending on the sign 

49 of the result. If False, only one result is returned. 

50 

51 See Also 

52 -------- 

53 numpy.logaddexp, numpy.logaddexp2 

54 

55 Notes 

56 ----- 

57 NumPy has a logaddexp function which is very similar to `logsumexp`, but 

58 only handles two arguments. `logaddexp.reduce` is similar to this 

59 function, but may be less stable. 

60 

61 Examples 

62 -------- 

63 >>> from scipy.special import logsumexp 

64 >>> a = np.arange(10) 

65 >>> np.log(np.sum(np.exp(a))) 

66 9.4586297444267107 

67 >>> logsumexp(a) 

68 9.4586297444267107 

69 

70 With weights 

71 

72 >>> a = np.arange(10) 

73 >>> b = np.arange(10, 0, -1) 

74 >>> logsumexp(a, b=b) 

75 9.9170178533034665 

76 >>> np.log(np.sum(b*np.exp(a))) 

77 9.9170178533034647 

78 

79 Returning a sign flag 

80 

81 >>> logsumexp([1,2],b=[1,-1],return_sign=True) 

82 (1.5413248546129181, -1.0) 

83 

84 Notice that `logsumexp` does not directly support masked arrays. To use it 

85 on a masked array, convert the mask into zero weights: 

86 

87 >>> a = np.ma.array([np.log(2), 2, np.log(3)], 

88 ... mask=[False, True, False]) 

89 >>> b = (~a.mask).astype(int) 

90 >>> logsumexp(a.data, b=b), np.log(5) 

91 1.6094379124341005, 1.6094379124341005 

92 

93 """ 

94 a = _asarray_validated(a, check_finite=False) 

95 if b is not None: 

96 a, b = np.broadcast_arrays(a, b) 

97 if np.any(b == 0): 

98 a = a + 0. # promote to at least float 

99 a[b == 0] = -np.inf 

100 

101 a_max = np.amax(a, axis=axis, keepdims=True) 

102 

103 if a_max.ndim > 0: 

104 a_max[~np.isfinite(a_max)] = 0 

105 elif not np.isfinite(a_max): 

106 a_max = 0 

107 

108 if b is not None: 

109 b = np.asarray(b) 

110 tmp = b * np.exp(a - a_max) 

111 else: 

112 tmp = np.exp(a - a_max) 

113 

114 # suppress warnings about log of zero 

115 with np.errstate(divide='ignore'): 

116 s = np.sum(tmp, axis=axis, keepdims=keepdims) 

117 if return_sign: 

118 sgn = np.sign(s) 

119 s *= sgn # /= makes more sense but we need zero -> zero 

120 out = np.log(s) 

121 

122 if not keepdims: 

123 a_max = np.squeeze(a_max, axis=axis) 

124 out += a_max 

125 

126 if return_sign: 

127 return out, sgn 

128 else: 

129 return out

132def softmax(x, axis=None): 

133 r""" 

134 Softmax function 

135 

136 The softmax function transforms each element of a collection by 

137 computing the exponential of each element divided by the sum of the 

138 exponentials of all the elements. That is, if `x` is a one-dimensional 

139 numpy array:: 

140 

141 softmax(x) = np.exp(x)/sum(np.exp(x)) 

142 

143 Parameters 

144 ---------- 

145 x : array_like 

146 Input array. 

147 axis : int or tuple of ints, optional 

148 Axis to compute values along. Default is None and softmax will be 

149 computed over the entire array `x`. 

150 

151 Returns 

152 ------- 

153 s : ndarray 

154 An array the same shape as `x`. The result will sum to 1 along the 

155 specified axis. 

156 

157 Notes 

158 ----- 

159 The formula for the softmax function :math:`\sigma(x)` for a vector 

160 :math:`x = \{x_0, x_1, ..., x_{n-1}\}` is 

161 

162 .. math:: \sigma(x)_j = \frac{e^{x_j}}{\sum_k e^{x_k}} 

163 

164 The `softmax` function is the gradient of `logsumexp`. 

165 

166 .. versionadded:: 1.2.0 

167 

168 Examples 

169 -------- 

170 >>> from scipy.special import softmax 

171 >>> np.set_printoptions(precision=5) 

172 

173 >>> x = np.array([[1, 0.5, 0.2, 3], 

174 ... [1, -1, 7, 3], 

175 ... [2, 12, 13, 3]]) 

176 ... 

177 

178 Compute the softmax transformation over the entire array. 

179 

180 >>> m = softmax(x) 

181 >>> m 

182 array([[ 4.48309e-06, 2.71913e-06, 2.01438e-06, 3.31258e-05], 

183 [ 4.48309e-06, 6.06720e-07, 1.80861e-03, 3.31258e-05], 

184 [ 1.21863e-05, 2.68421e-01, 7.29644e-01, 3.31258e-05]]) 

185 

186 >>> m.sum() 

187 1.0000000000000002 

188 

189 Compute the softmax transformation along the first axis (i.e. the columns). 

190 

191 >>> m = softmax(x, axis=0) 

192 

193 >>> m 

194 array([[ 2.11942e-01, 1.01300e-05, 2.75394e-06, 3.33333e-01], 

195 [ 2.11942e-01, 2.26030e-06, 2.47262e-03, 3.33333e-01], 

196 [ 5.76117e-01, 9.99988e-01, 9.97525e-01, 3.33333e-01]]) 

197 

198 >>> m.sum(axis=0) 

199 array([ 1., 1., 1., 1.]) 

200 

201 Compute the softmax transformation along the second axis (i.e. the rows). 

202 

203 >>> m = softmax(x, axis=1) 

204 >>> m 

205 array([[ 1.05877e-01, 6.42177e-02, 4.75736e-02, 7.82332e-01], 

206 [ 2.42746e-03, 3.28521e-04, 9.79307e-01, 1.79366e-02], 

207 [ 1.22094e-05, 2.68929e-01, 7.31025e-01, 3.31885e-05]]) 

208 

209 >>> m.sum(axis=1) 

210 array([ 1., 1., 1.]) 

211 

212 """ 

213 

214 # compute in log space for numerical stability 

215 return np.exp(x - logsumexp(x, axis=axis, keepdims=True))
