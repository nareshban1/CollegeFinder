35_bad_control_character_codes = set(range(0, 0x20)) - {0x9, 0xA, 0xD}

52# When class or module level function fails, unittest/suite.py adds a

67def _escape_cdata(s): 

68 """Escapes a string to be used as XML CDATA. 

69 

70 CDATA characters are treated strictly as character data, not as XML markup, 

71 but there are still certain restrictions on them. 

72 

73 Args: 

74 s: the string to be escaped. 

75 Returns: 

76 An escaped version of the input string. 

77 """ 

78 for char, escaped in six.iteritems(_control_character_conversions): 

79 s = s.replace(char, escaped) 

80 return s.replace(']]>', ']] >')

83def _iso8601_timestamp(timestamp): 

84 """Produces an ISO8601 datetime. 

85 

86 Args: 

87 timestamp: an Epoch based timestamp in seconds. 

88 

89 Returns: 

90 A iso8601 format timestamp if the input is a valid timestamp, None otherwise 

91 """ 

92 if timestamp is None or timestamp < 0: 

93 return None 

94 # Use utcfromtimestamp in PY2 because it doesn't have a built-in UTC object 

95 if six.PY2: 

96 return '%s+00:00' % datetime.datetime.utcfromtimestamp( 

97 timestamp).isoformat() 

98 else: 

99 return datetime.datetime.fromtimestamp( 

100 timestamp, tz=datetime.timezone.utc).isoformat()

103def _print_xml_element_header(element, attributes, stream, indentation=''): 

104 """Prints an XML header of an arbitrary element. 

105 

106 Args: 

107 element: element name (testsuites, testsuite, testcase) 

108 attributes: 2-tuple list with (attributes, values) already escaped 

109 stream: output stream to write test report XML to 

110 indentation: indentation added to the element header 

111 """ 

112 stream.write('%s<%s' % (indentation, element)) 

113 for attribute in attributes: 

114 if len(attribute) == 2 \ 

115 and attribute[0] is not None and attribute[1] is not None: 

116 stream.write(' %s="%s"' % (attribute[0], attribute[1])) 

117 stream.write('>\n')

120# This prevents bad interactions with tests that stub out time. 

121_time_copy = time.time

123if hasattr(traceback, '_some_str'): 

124 # Use the traceback module str function to format safely. 

125 _safe_str = traceback._some_str

130class _TestCaseResult(object): 

131 """Private helper for _TextAndXMLTestResult that represents a test result. 

132 

133 Attributes: 

134 test: A TestCase instance of an individual test method. 

135 name: The name of the individual test method. 

136 full_class_name: The full name of the test class. 

137 run_time: The duration (in seconds) it took to run the test. 

138 start_time: Epoch relative timestamp of when test started (in seconds) 

139 errors: A list of error 4-tuples. Error tuple entries are 

140 1) a string identifier of either "failure" or "error" 

141 2) an exception_type 

142 3) an exception_message 

143 4) a string version of a sys.exc_info()-style tuple of values 

144 ('error', err[0], err[1], self._exc_info_to_string(err)) 

145 If the length of errors is 0, then the test is either passed or 

146 skipped. 

147 skip_reason: A string explaining why the test was skipped. 

148 """ 

149 

150 def __init__(self, test): 

151 self.run_time = -1 

152 self.start_time = -1 

153 self.skip_reason = None 

154 self.errors = [] 

155 self.test = test 

156 

157 # Parse the test id to get its test name and full class path. 

158 # Unfortunately there is no better way of knowning the test and class. 

159 # Worse, unittest uses _ErrorHandler instances to represent class / module 

160 # level failures. 

161 test_desc = test.id() or str(test) 

162 # Check if it's something like "setUpClass (__main__.TestCase)". 

163 match = _CLASS_OR_MODULE_LEVEL_TEST_DESC_REGEX.match(test_desc) 

164 if match: 

165 name = match.group(1) 

166 full_class_name = match.group(2) 

167 else: 

168 class_name = unittest.util.strclass(test.__class__) 

169 if ((six.PY3 and isinstance(test, unittest.case._SubTest)) or 

170 (six.PY2 and isinstance(test, unittest3_backport.case._SubTest))): 

171 # If the test case is a _SubTest, the real TestCase instance is 

172 # available as _SubTest.test_case. 

173 class_name = unittest.util.strclass(test.test_case.__class__) 

174 if test_desc.startswith(class_name + '.'): 

175 # In a typical unittest.TestCase scenario, test.id() returns with 

176 # a class name formatted using unittest.util.strclass. 

177 name = test_desc[len(class_name)+1:] 

178 full_class_name = class_name 

179 else: 

180 # Otherwise make a best effort to guess the test name and full class 

181 # path. 

182 parts = test_desc.rsplit('.', 1) 

183 name = parts[-1] 

184 full_class_name = parts[0] if len(parts) == 2 else '' 

185 self.name = _escape_xml_attr(name) 

186 self.full_class_name = _escape_xml_attr(full_class_name)

189 self.run_time = time_in_secs

194 def print_xml_summary(self, stream): 

195 """Prints an XML Summary of a TestCase. 

196 

197 Status and result are populated as per JUnit XML test result reporter. 

198 A test that has been skipped will always have a skip reason, 

199 as every skip method in Python's unittest requires the reason arg to be 

200 passed. 

201 

202 Args: 

203 stream: output stream to write test report XML to 

204 """ 

205 

206 if self.skip_reason is None: 

207 status = 'run' 

208 result = 'completed' 

209 else: 

210 status = 'notrun' 

211 result = 'suppressed' 

212 

213 test_case_attributes = [ 

214 ('name', '%s' % self.name), 

215 ('status', '%s' % status), 

216 ('result', '%s' % result), 

217 ('time', '%.1f' % self.run_time), 

218 ('classname', self.full_class_name), 

219 ('timestamp', _iso8601_timestamp(self.start_time)), 

220 ] 

221 _print_xml_element_header('testcase', test_case_attributes, stream, ' ') 

222 self._print_testcase_details(stream) 

223 stream.write(' </testcase>\n') 

224 

225 def _print_testcase_details(self, stream): 

226 for error in self.errors: 

227 outcome, exception_type, message, error_msg = error # pylint: disable=unpacking-non-sequence 

228 message = _escape_xml_attr(_safe_str(message)) 

229 exception_type = _escape_xml_attr(str(exception_type)) 

230 error_msg = _escape_cdata(error_msg) 

231 stream.write(' <%s message="%s" type="%s"><![CDATA[%s]]></%s>\n' 

232 % (outcome, message, exception_type, error_msg, outcome))

235class _TestSuiteResult(object): 

236 """Private helper for _TextAndXMLTestResult.""" 

237 

238 def __init__(self): 

239 self.suites = {} 

240 self.failure_counts = {} 

241 self.error_counts = {} 

242 self.overall_start_time = -1 

243 self.overall_end_time = -1 

244 

245 def add_test_case_result(self, test_case_result): 

246 suite_name = type(test_case_result.test).__name__ 

247 if suite_name == '_ErrorHolder': 

248 # _ErrorHolder is a special case created by unittest for class / module 

249 # level functions. 

250 suite_name = test_case_result.full_class_name.rsplit('.')[-1] 

251 if ((six.PY3 and 

252 isinstance(test_case_result.test, unittest.case._SubTest)) or 

253 (six.PY2 and 

254 isinstance(test_case_result.test, unittest3_backport.case._SubTest))): 

255 # If the test case is a _SubTest, the real TestCase instance is 

256 # available as _SubTest.test_case. 

257 suite_name = type(test_case_result.test.test_case).__name__ 

258 

259 self._setup_test_suite(suite_name) 

260 self.suites[suite_name].append(test_case_result) 

261 for error in test_case_result.errors: 

262 # Only count the first failure or error so that the sum is equal to the 

263 # total number of *testcases* that have failures or errors. 

264 if error[0] == 'failure': 

265 self.failure_counts[suite_name] += 1 

266 break 

267 elif error[0] == 'error': 

268 self.error_counts[suite_name] += 1 

269 break 

270 

271 def print_xml_summary(self, stream): 

272 overall_test_count = sum(len(x) for x in self.suites.values()) 

273 overall_failures = sum(self.failure_counts.values()) 

274 overall_errors = sum(self.error_counts.values()) 

275 overall_attributes = [ 

276 ('name', ''), 

277 ('tests', '%d' % overall_test_count), 

278 ('failures', '%d' % overall_failures), 

279 ('errors', '%d' % overall_errors), 

280 ('time', '%.1f' % (self.overall_end_time - self.overall_start_time)), 

281 ('timestamp', _iso8601_timestamp(self.overall_start_time)), 

282 ] 

283 _print_xml_element_header('testsuites', overall_attributes, stream) 

284 

285 for suite_name in self.suites: 

286 suite = self.suites[suite_name] 

287 suite_end_time = max(x.start_time + x.run_time for x in suite) 

288 suite_start_time = min(x.start_time for x in suite) 

289 failures = self.failure_counts[suite_name] 

290 errors = self.error_counts[suite_name] 

291 suite_attributes = [ 

292 ('name', '%s' % suite_name), 

293 ('tests', '%d' % len(suite)), 

294 ('failures', '%d' % failures), 

295 ('errors', '%d' % errors), 

296 ('time', '%.1f' % (suite_end_time - suite_start_time)), 

297 ('timestamp', _iso8601_timestamp(suite_start_time)), 

298 ] 

299 _print_xml_element_header('testsuite', suite_attributes, stream) 

300 

301 for test_case_result in suite: 

302 test_case_result.print_xml_summary(stream) 

303 stream.write('</testsuite>\n') 

304 stream.write('</testsuites>\n')

335class _TextAndXMLTestResult(_pretty_print_reporter.TextTestResult): 

336 """Private TestResult class that produces both formatted text results and XML. 

337 

338 Used by TextAndXMLTestRunner. 

339 """ 

340 

341 _TEST_SUITE_RESULT_CLASS = _TestSuiteResult 

342 _TEST_CASE_RESULT_CLASS = _TestCaseResult

344 def __init__(self, xml_stream, stream, descriptions, verbosity, 

345 time_getter=_time_copy): 

346 super(_TextAndXMLTestResult, self).__init__(stream, descriptions, verbosity) 

347 self.xml_stream = xml_stream 

348 self.pending_test_case_results = {} 

349 self.suite = self._TEST_SUITE_RESULT_CLASS() 

350 self.time_getter = time_getter 

351 

352 # This lock guards any mutations on pending_test_case_results. 

353 self._pending_test_case_results_lock = threading.RLock()

355 def startTest(self, test): 

356 self.start_time = self.time_getter() 

357 super(_TextAndXMLTestResult, self).startTest(test) 

358 

359 def stopTest(self, test): 

360 # Grabbing the write lock to avoid conflicting with stopTestRun. 

361 with self._pending_test_case_results_lock: 

362 super(_TextAndXMLTestResult, self).stopTest(test) 

363 result = self.get_pending_test_case_result(test) 

364 if not result: 

365 test_name = test.id() or str(test) 

366 sys.stderr.write('No pending test case: %s\n' % test_name) 

367 return 

368 test_id = id(test) 

369 run_time = self.time_getter() - self.start_time 

370 result.set_run_time(run_time) 

371 result.set_start_time(self.start_time) 

372 self.suite.add_test_case_result(result) 

373 del self.pending_test_case_results[test_id] 

374 

375 def startTestRun(self): 

376 self.suite.set_start_time(self.time_getter()) 

377 super(_TextAndXMLTestResult, self).startTestRun() 

378 

379 def stopTestRun(self): 

380 self.suite.set_end_time(self.time_getter()) 

381 # All pending_test_case_results will be added to the suite and removed from 

382 # the pending_test_case_results dictionary. Grabing the write lock to avoid 

383 # results from being added during this process to avoid duplicating adds or 

384 # accidentally erasing newly appended pending results. 

385 with self._pending_test_case_results_lock: 

386 # Errors in the test fixture (setUpModule, tearDownModule, 

387 # setUpClass, tearDownClass) can leave a pending result which 

388 # never gets added to the suite. The runner calls stopTestRun 

389 # which gives us an opportunity to add these errors for 

390 # reporting here. 

391 for test_id in self.pending_test_case_results: 

392 result = self.pending_test_case_results[test_id] 

393 if hasattr(self, 'start_time'): 

394 run_time = self.suite.overall_end_time - self.start_time 

395 result.set_run_time(run_time) 

396 result.set_start_time(self.start_time) 

397 self.suite.add_test_case_result(result) 

398 self.pending_test_case_results.clear()

417 def add_pending_test_case_result(self, test, error_summary=None, 

418 skip_reason=None): 

419 """Adds result information to a test case result which may still be running. 

420 

421 If a result entry for the test already exists, add_pending_test_case_result 

422 will add error summary tuples and/or overwrite skip_reason for the result. 

423 If it does not yet exist, a result entry will be created. 

424 Note that a test result is considered to have been run and passed 

425 only if there are no errors or skip_reason. 

426 

427 Args: 

428 test: A test method as defined by unittest 

429 error_summary: A 4-tuple with the following entries: 

430 1) a string identifier of either "failure" or "error" 

431 2) an exception_type 

432 3) an exception_message 

433 4) a string version of a sys.exc_info()-style tuple of values 

434 ('error', err[0], err[1], self._exc_info_to_string(err)) 

435 If the length of errors is 0, then the test is either passed or 

436 skipped. 

437 skip_reason: a string explaining why the test was skipped 

438 """ 

439 with self._pending_test_case_results_lock: 

440 test_id = id(test) 

441 if test_id not in self.pending_test_case_results: 

442 self.pending_test_case_results[test_id] = self._TEST_CASE_RESULT_CLASS( 

443 test) 

444 if error_summary: 

445 self.pending_test_case_results[test_id].errors.append(error_summary) 

446 if skip_reason: 

447 self.pending_test_case_results[test_id].skip_reason = skip_reason 

448 

449 def delete_pending_test_case_result(self, test): 

450 with self._pending_test_case_results_lock: 

451 test_id = id(test) 

452 del self.pending_test_case_results[test_id]

454 def get_pending_test_case_result(self, test): 

455 test_id = id(test) 

456 return self.pending_test_case_results.get(test_id, None)
