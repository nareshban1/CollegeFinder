19def prepare_input(x, y, axis, dydx=None): 

20 """Prepare input for cubic spline interpolators. 

21 

22 All data are converted to numpy arrays and checked for correctness. 

23 Axes equal to `axis` of arrays `y` and `dydx` are rolled to be the 0-th 

24 axis. The value of `axis` is converted to lie in 

25 [0, number of dimensions of `y`). 

26 """ 

27 

28 x, y = map(np.asarray, (x, y)) 

29 if np.issubdtype(x.dtype, np.complexfloating): 

30 raise ValueError("`x` must contain real values.") 

31 x = x.astype(float) 

32 

33 if np.issubdtype(y.dtype, np.complexfloating): 

34 dtype = complex 

35 else: 

36 dtype = float 

37 

38 if dydx is not None: 

39 dydx = np.asarray(dydx) 

40 if y.shape != dydx.shape: 

41 raise ValueError("The shapes of `y` and `dydx` must be identical.") 

42 if np.issubdtype(dydx.dtype, np.complexfloating): 

43 dtype = complex 

44 dydx = dydx.astype(dtype, copy=False) 

45 

46 y = y.astype(dtype, copy=False) 

47 axis = axis % y.ndim 

48 if x.ndim != 1: 

49 raise ValueError("`x` must be 1-dimensional.") 

50 if x.shape[0] < 2: 

51 raise ValueError("`x` must contain at least 2 elements.") 

52 if x.shape[0] != y.shape[axis]: 

53 raise ValueError("The length of `y` along `axis`={0} doesn't " 

54 "match the length of `x`".format(axis)) 

55 

56 if not np.all(np.isfinite(x)): 

57 raise ValueError("`x` must contain only finite values.") 

58 if not np.all(np.isfinite(y)): 

59 raise ValueError("`y` must contain only finite values.") 

60 

61 if dydx is not None and not np.all(np.isfinite(dydx)): 

62 raise ValueError("`dydx` must contain only finite values.") 

63 

64 dx = np.diff(x) 

65 if np.any(dx <= 0): 

66 raise ValueError("`x` must be strictly increasing sequence.") 

67 

68 y = np.rollaxis(y, axis) 

69 if dydx is not None: 

70 dydx = np.rollaxis(dydx, axis) 

71 

72 return x, dx, y, axis, dydx

75class CubicHermiteSpline(PPoly): 

76 """Piecewise-cubic interpolator matching values and first derivatives. 

77 

78 The result is represented as a `PPoly` instance. 

79 

80 Parameters 

81 ---------- 

82 x : array_like, shape (n,) 

83 1-d array containing values of the independent variable. 

84 Values must be real, finite and in strictly increasing order. 

85 y : array_like 

86 Array containing values of the dependent variable. It can have 

87 arbitrary number of dimensions, but the length along ``axis`` 

88 (see below) must match the length of ``x``. Values must be finite. 

89 dydx : array_like 

90 Array containing derivatives of the dependent variable. It can have 

91 arbitrary number of dimensions, but the length along ``axis`` 

92 (see below) must match the length of ``x``. Values must be finite. 

93 axis : int, optional 

94 Axis along which `y` is assumed to be varying. Meaning that for 

95 ``x[i]`` the corresponding values are ``np.take(y, i, axis=axis)``. 

96 Default is 0. 

97 extrapolate : {bool, 'periodic', None}, optional 

98 If bool, determines whether to extrapolate to out-of-bounds points 

99 based on first and last intervals, or to return NaNs. If 'periodic', 

100 periodic extrapolation is used. If None (default), it is set to True. 

101 

102 Attributes 

103 ---------- 

104 x : ndarray, shape (n,) 

105 Breakpoints. The same ``x`` which was passed to the constructor. 

106 c : ndarray, shape (4, n-1, ...) 

107 Coefficients of the polynomials on each segment. The trailing 

108 dimensions match the dimensions of `y`, excluding ``axis``. 

109 For example, if `y` is 1-d, then ``c[k, i]`` is a coefficient for 

110 ``(x-x[i])**(3-k)`` on the segment between ``x[i]`` and ``x[i+1]``. 

111 axis : int 

112 Interpolation axis. The same axis which was passed to the 

113 constructor. 

114 

115 Methods 

116 ------- 

117 __call__ 

118 derivative 

119 antiderivative 

120 integrate 

121 roots 

122 

123 See Also 

124 -------- 

125 Akima1DInterpolator 

126 PchipInterpolator 

127 CubicSpline 

128 PPoly 

129 

130 Notes 

131 ----- 

132 If you want to create a higher-order spline matching higher-order 

133 derivatives, use `BPoly.from_derivatives`. 

134 

135 References 

136 ---------- 

137 .. [1] `Cubic Hermite spline 

138 <https://en.wikipedia.org/wiki/Cubic_Hermite_spline>`_ 

139 on Wikipedia. 

140 """ 

141 def __init__(self, x, y, dydx, axis=0, extrapolate=None): 

142 if extrapolate is None: 

143 extrapolate = True 

144 

145 x, dx, y, axis, dydx = prepare_input(x, y, axis, dydx) 

146 

147 dxr = dx.reshape([dx.shape[0]] + [1] * (y.ndim - 1)) 

148 slope = np.diff(y, axis=0) / dxr 

149 t = (dydx[:-1] + dydx[1:] - 2 * slope) / dxr 

150 

151 c = np.empty((4, len(x) - 1) + y.shape[1:], dtype=t.dtype) 

152 c[0] = t / dxr 

153 c[1] = (slope - dydx[:-1]) / dxr - t 

154 c[2] = dydx[:-1] 

155 c[3] = y[:-1] 

156 

157 super(CubicHermiteSpline, self).__init__(c, x, extrapolate=extrapolate) 

158 self.axis = axis

161class PchipInterpolator(CubicHermiteSpline): 

162 r"""PCHIP 1-d monotonic cubic interpolation. 

163 

164 ``x`` and ``y`` are arrays of values used to approximate some function f, 

165 with ``y = f(x)``. The interpolant uses monotonic cubic splines 

166 to find the value of new points. (PCHIP stands for Piecewise Cubic 

167 Hermite Interpolating Polynomial). 

168 

169 Parameters 

170 ---------- 

171 x : ndarray 

172 A 1-D array of monotonically increasing real values. ``x`` cannot 

173 include duplicate values (otherwise f is overspecified) 

174 y : ndarray 

175 A 1-D array of real values. ``y``'s length along the interpolation 

176 axis must be equal to the length of ``x``. If N-D array, use ``axis`` 

177 parameter to select correct axis. 

178 axis : int, optional 

179 Axis in the y array corresponding to the x-coordinate values. 

180 extrapolate : bool, optional 

181 Whether to extrapolate to out-of-bounds points based on first 

182 and last intervals, or to return NaNs. 

183 

184 Methods 

185 ------- 

186 __call__ 

187 derivative 

188 antiderivative 

189 roots 

190 

191 See Also 

192 -------- 

193 CubicHermiteSpline 

194 Akima1DInterpolator 

195 CubicSpline 

196 PPoly 

197 

198 Notes 

199 ----- 

200 The interpolator preserves monotonicity in the interpolation data and does 

201 not overshoot if the data is not smooth. 

202 

203 The first derivatives are guaranteed to be continuous, but the second 

204 derivatives may jump at :math:`x_k`. 

205 

206 Determines the derivatives at the points :math:`x_k`, :math:`f'_k`, 

207 by using PCHIP algorithm [1]_. 

208 

209 Let :math:`h_k = x_{k+1} - x_k`, and :math:`d_k = (y_{k+1} - y_k) / h_k` 

210 are the slopes at internal points :math:`x_k`. 

211 If the signs of :math:`d_k` and :math:`d_{k-1}` are different or either of 

212 them equals zero, then :math:`f'_k = 0`. Otherwise, it is given by the 

213 weighted harmonic mean 

214 

215 .. math:: 

216 

217 \frac{w_1 + w_2}{f'_k} = \frac{w_1}{d_{k-1}} + \frac{w_2}{d_k} 

218 

219 where :math:`w_1 = 2 h_k + h_{k-1}` and :math:`w_2 = h_k + 2 h_{k-1}`. 

220 

221 The end slopes are set using a one-sided scheme [2]_. 

222 

223 

224 References 

225 ---------- 

226 .. [1] F. N. Fritsch and R. E. Carlson, Monotone Piecewise Cubic Interpolation, 

227 SIAM J. Numer. Anal., 17(2), 238 (1980). 

228 :doi:`10.1137/0717021`. 

229 .. [2] see, e.g., C. Moler, Numerical Computing with Matlab, 2004. 

230 :doi:`10.1137/1.9780898717952` 

231 

232 

233 """ 

234 def __init__(self, x, y, axis=0, extrapolate=None): 

235 x, _, y, axis, _ = prepare_input(x, y, axis) 

236 xp = x.reshape((x.shape[0],) + (1,)*(y.ndim-1)) 

237 dk = self._find_derivatives(xp, y) 

238 super(PchipInterpolator, self).__init__(x, y, dk, axis=0, 

239 extrapolate=extrapolate) 

240 self.axis = axis 

241 

242 @staticmethod 

243 def _edge_case(h0, h1, m0, m1): 

244 # one-sided three-point estimate for the derivative 

245 d = ((2*h0 + h1)*m0 - h0*m1) / (h0 + h1) 

246 

247 # try to preserve shape 

248 mask = np.sign(d) != np.sign(m0) 

249 mask2 = (np.sign(m0) != np.sign(m1)) & (np.abs(d) > 3.*np.abs(m0)) 

250 mmm = (~mask) & mask2 

251 

252 d[mask] = 0. 

253 d[mmm] = 3.*m0[mmm] 

254 

255 return d 

256 

257 @staticmethod 

258 def _find_derivatives(x, y): 

259 # Determine the derivatives at the points y_k, d_k, by using 

260 # PCHIP algorithm is: 

261 # We choose the derivatives at the point x_k by 

262 # Let m_k be the slope of the kth segment (between k and k+1) 

263 # If m_k=0 or m_{k-1}=0 or sgn(m_k) != sgn(m_{k-1}) then d_k == 0 

264 # else use weighted harmonic mean: 

265 # w_1 = 2h_k + h_{k-1}, w_2 = h_k + 2h_{k-1} 

266 # 1/d_k = 1/(w_1 + w_2)*(w_1 / m_k + w_2 / m_{k-1}) 

267 # where h_k is the spacing between x_k and x_{k+1} 

268 y_shape = y.shape 

269 if y.ndim == 1: 

270 # So that _edge_case doesn't end up assigning to scalars 

271 x = x[:, None] 

272 y = y[:, None] 

273 

274 hk = x[1:] - x[:-1] 

275 mk = (y[1:] - y[:-1]) / hk 

276 

277 if y.shape[0] == 2: 

278 # edge case: only have two points, use linear interpolation 

279 dk = np.zeros_like(y) 

280 dk[0] = mk 

281 dk[1] = mk 

282 return dk.reshape(y_shape) 

283 

284 smk = np.sign(mk) 

285 condition = (smk[1:] != smk[:-1]) | (mk[1:] == 0) | (mk[:-1] == 0) 

286 

287 w1 = 2*hk[1:] + hk[:-1] 

288 w2 = hk[1:] + 2*hk[:-1] 

289 

290 # values where division by zero occurs will be excluded 

291 # by 'condition' afterwards 

292 with np.errstate(divide='ignore'): 

293 whmean = (w1/mk[:-1] + w2/mk[1:]) / (w1 + w2) 

294 

295 dk = np.zeros_like(y) 

296 dk[1:-1][condition] = 0.0 

297 dk[1:-1][~condition] = 1.0 / whmean[~condition] 

298 

299 # special case endpoints, as suggested in 

300 # Cleve Moler, Numerical Computing with MATLAB, Chap 3.4 

301 dk[0] = PchipInterpolator._edge_case(hk[0], hk[1], mk[0], mk[1]) 

302 dk[-1] = PchipInterpolator._edge_case(hk[-1], hk[-2], mk[-1], mk[-2]) 

303 

304 return dk.reshape(y_shape)

353class Akima1DInterpolator(CubicHermiteSpline): 

354 """ 

355 Akima interpolator 

356 

357 Fit piecewise cubic polynomials, given vectors x and y. The interpolation 

358 method by Akima uses a continuously differentiable sub-spline built from 

359 piecewise cubic polynomials. The resultant curve passes through the given 

360 data points and will appear smooth and natural. 

361 

362 Parameters 

363 ---------- 

364 x : ndarray, shape (m, ) 

365 1-D array of monotonically increasing real values. 

366 y : ndarray, shape (m, ...) 

367 N-D array of real values. The length of ``y`` along the first axis 

368 must be equal to the length of ``x``. 

369 axis : int, optional 

370 Specifies the axis of ``y`` along which to interpolate. Interpolation 

371 defaults to the first axis of ``y``. 

372 

373 Methods 

374 ------- 

375 __call__ 

376 derivative 

377 antiderivative 

378 roots 

379 

380 See Also 

381 -------- 

382 PchipInterpolator 

383 CubicSpline 

384 PPoly 

385 

386 Notes 

387 ----- 

388 .. versionadded:: 0.14 

389 

390 Use only for precise data, as the fitted curve passes through the given 

391 points exactly. This routine is useful for plotting a pleasingly smooth 

392 curve through a few given points for purposes of plotting. 

393 

394 References 

395 ---------- 

396 [1] A new method of interpolation and smooth curve fitting based 

397 on local procedures. Hiroshi Akima, J. ACM, October 1970, 17(4), 

398 589-602. 

399 

400 """ 

401 

402 def __init__(self, x, y, axis=0): 

403 # Original implementation in MATLAB by N. Shamsundar (BSD licensed), see 

404 # https://www.mathworks.com/matlabcentral/fileexchange/1814-akima-interpolation 

405 x, dx, y, axis, _ = prepare_input(x, y, axis) 

406 # determine slopes between breakpoints 

407 m = np.empty((x.size + 3, ) + y.shape[1:]) 

408 dx = dx[(slice(None), ) + (None, ) * (y.ndim - 1)] 

409 m[2:-2] = np.diff(y, axis=0) / dx 

410 

411 # add two additional points on the left ... 

412 m[1] = 2. * m[2] - m[3] 

413 m[0] = 2. * m[1] - m[2] 

414 # ... and on the right 

415 m[-2] = 2. * m[-3] - m[-4] 

416 m[-1] = 2. * m[-2] - m[-3] 

417 

418 # if m1 == m2 != m3 == m4, the slope at the breakpoint is not defined. 

419 # This is the fill value: 

420 t = .5 * (m[3:] + m[:-3]) 

421 # get the denominator of the slope t 

422 dm = np.abs(np.diff(m, axis=0)) 

423 f1 = dm[2:] 

424 f2 = dm[:-2] 

425 f12 = f1 + f2 

426 # These are the mask of where the the slope at breakpoint is defined: 

427 ind = np.nonzero(f12 > 1e-9 * np.max(f12)) 

428 x_ind, y_ind = ind[0], ind[1:] 

429 # Set the slope at breakpoint 

430 t[ind] = (f1[ind] * m[(x_ind + 1,) + y_ind] + 

431 f2[ind] * m[(x_ind + 2,) + y_ind]) / f12[ind] 

432 

433 super(Akima1DInterpolator, self).__init__(x, y, t, axis=0, 

434 extrapolate=False) 

435 self.axis = axis

454class CubicSpline(CubicHermiteSpline): 

455 """Cubic spline data interpolator. 

456 

457 Interpolate data with a piecewise cubic polynomial which is twice 

458 continuously differentiable [1]_. The result is represented as a `PPoly` 

459 instance with breakpoints matching the given data. 

460 

461 Parameters 

462 ---------- 

463 x : array_like, shape (n,) 

464 1-d array containing values of the independent variable. 

465 Values must be real, finite and in strictly increasing order. 

466 y : array_like 

467 Array containing values of the dependent variable. It can have 

468 arbitrary number of dimensions, but the length along ``axis`` 

469 (see below) must match the length of ``x``. Values must be finite. 

470 axis : int, optional 

471 Axis along which `y` is assumed to be varying. Meaning that for 

472 ``x[i]`` the corresponding values are ``np.take(y, i, axis=axis)``. 

473 Default is 0. 

474 bc_type : string or 2-tuple, optional 

475 Boundary condition type. Two additional equations, given by the 

476 boundary conditions, are required to determine all coefficients of 

477 polynomials on each segment [2]_. 

478 

479 If `bc_type` is a string, then the specified condition will be applied 

480 at both ends of a spline. Available conditions are: 

481 

482 * 'not-a-knot' (default): The first and second segment at a curve end 

483 are the same polynomial. It is a good default when there is no 

484 information on boundary conditions. 

485 * 'periodic': The interpolated functions is assumed to be periodic 

486 of period ``x[-1] - x[0]``. The first and last value of `y` must be 

487 identical: ``y[0] == y[-1]``. This boundary condition will result in 

488 ``y'[0] == y'[-1]`` and ``y''[0] == y''[-1]``. 

489 * 'clamped': The first derivative at curves ends are zero. Assuming 

490 a 1D `y`, ``bc_type=((1, 0.0), (1, 0.0))`` is the same condition. 

491 * 'natural': The second derivative at curve ends are zero. Assuming 

492 a 1D `y`, ``bc_type=((2, 0.0), (2, 0.0))`` is the same condition. 

493 

494 If `bc_type` is a 2-tuple, the first and the second value will be 

495 applied at the curve start and end respectively. The tuple values can 

496 be one of the previously mentioned strings (except 'periodic') or a 

497 tuple `(order, deriv_values)` allowing to specify arbitrary 

498 derivatives at curve ends: 

499 

500 * `order`: the derivative order, 1 or 2. 

501 * `deriv_value`: array_like containing derivative values, shape must 

502 be the same as `y`, excluding ``axis`` dimension. For example, if 

503 `y` is 1D, then `deriv_value` must be a scalar. If `y` is 3D with 

504 the shape (n0, n1, n2) and axis=2, then `deriv_value` must be 2D 

505 and have the shape (n0, n1). 

506 extrapolate : {bool, 'periodic', None}, optional 

507 If bool, determines whether to extrapolate to out-of-bounds points 

508 based on first and last intervals, or to return NaNs. If 'periodic', 

509 periodic extrapolation is used. If None (default), ``extrapolate`` is 

510 set to 'periodic' for ``bc_type='periodic'`` and to True otherwise. 

511 

512 Attributes 

513 ---------- 

514 x : ndarray, shape (n,) 

515 Breakpoints. The same ``x`` which was passed to the constructor. 

516 c : ndarray, shape (4, n-1, ...) 

517 Coefficients of the polynomials on each segment. The trailing 

518 dimensions match the dimensions of `y`, excluding ``axis``. 

519 For example, if `y` is 1-d, then ``c[k, i]`` is a coefficient for 

520 ``(x-x[i])**(3-k)`` on the segment between ``x[i]`` and ``x[i+1]``. 

521 axis : int 

522 Interpolation axis. The same axis which was passed to the 

523 constructor. 

524 

525 Methods 

526 ------- 

527 __call__ 

528 derivative 

529 antiderivative 

530 integrate 

531 roots 

532 

533 See Also 

534 -------- 

535 Akima1DInterpolator 

536 PchipInterpolator 

537 PPoly 

538 

539 Notes 

540 ----- 

541 Parameters `bc_type` and ``interpolate`` work independently, i.e. the 

542 former controls only construction of a spline, and the latter only 

543 evaluation. 

544 

545 When a boundary condition is 'not-a-knot' and n = 2, it is replaced by 

546 a condition that the first derivative is equal to the linear interpolant 

547 slope. When both boundary conditions are 'not-a-knot' and n = 3, the 

548 solution is sought as a parabola passing through given points. 

549 

550 When 'not-a-knot' boundary conditions is applied to both ends, the 

551 resulting spline will be the same as returned by `splrep` (with ``s=0``) 

552 and `InterpolatedUnivariateSpline`, but these two methods use a 

553 representation in B-spline basis. 

554 

555 .. versionadded:: 0.18.0

557 Examples 

558 -------- 

559 In this example the cubic spline is used to interpolate a sampled sinusoid. 

560 You can see that the spline continuity property holds for the first and 

561 second derivatives and violates only for the third derivative. 

562 

563 >>> from scipy.interpolate import CubicSpline 

564 >>> import matplotlib.pyplot as plt 

565 >>> x = np.arange(10) 

566 >>> y = np.sin(x) 

567 >>> cs = CubicSpline(x, y) 

568 >>> xs = np.arange(-0.5, 9.6, 0.1) 

569 >>> fig, ax = plt.subplots(figsize=(6.5, 4)) 

570 >>> ax.plot(x, y, 'o', label='data') 

571 >>> ax.plot(xs, np.sin(xs), label='true') 

572 >>> ax.plot(xs, cs(xs), label="S") 

573 >>> ax.plot(xs, cs(xs, 1), label="S'") 

574 >>> ax.plot(xs, cs(xs, 2), label="S''") 

575 >>> ax.plot(xs, cs(xs, 3), label="S'''") 

576 >>> ax.set_xlim(-0.5, 9.5) 

577 >>> ax.legend(loc='lower left', ncol=2) 

578 >>> plt.show() 

579 

580 In the second example, the unit circle is interpolated with a spline. A 

581 periodic boundary condition is used. You can see that the first derivative 

582 values, ds/dx=0, ds/dy=1 at the periodic point (1, 0) are correctly 

583 computed. Note that a circle cannot be exactly represented by a cubic 

584 spline. To increase precision, more breakpoints would be required. 

585 

586 >>> theta = 2 * np.pi * np.linspace(0, 1, 5) 

587 >>> y = np.c_[np.cos(theta), np.sin(theta)] 

588 >>> cs = CubicSpline(theta, y, bc_type='periodic') 

589 >>> print("ds/dx={:.1f} ds/dy={:.1f}".format(cs(0, 1)[0], cs(0, 1)[1])) 

590 ds/dx=0.0 ds/dy=1.0 

591 >>> xs = 2 * np.pi * np.linspace(0, 1, 100) 

592 >>> fig, ax = plt.subplots(figsize=(6.5, 4)) 

593 >>> ax.plot(y[:, 0], y[:, 1], 'o', label='data') 

594 >>> ax.plot(np.cos(xs), np.sin(xs), label='true') 

595 >>> ax.plot(cs(xs)[:, 0], cs(xs)[:, 1], label='spline') 

596 >>> ax.axes.set_aspect('equal') 

597 >>> ax.legend(loc='center') 

598 >>> plt.show() 

599 

600 The third example is the interpolation of a polynomial y = x**3 on the 

601 interval 0 <= x<= 1. A cubic spline can represent this function exactly. 

602 To achieve that we need to specify values and first derivatives at 

603 endpoints of the interval. Note that y' = 3 * x**2 and thus y'(0) = 0 and 

604 y'(1) = 3. 

605 

606 >>> cs = CubicSpline([0, 1], [0, 1], bc_type=((1, 0), (1, 3))) 

607 >>> x = np.linspace(0, 1) 

608 >>> np.allclose(x**3, cs(x)) 

609 True 

610 

611 References 

612 ----------

614 <https://en.wikiversity.org/wiki/Cubic_Spline_Interpolation>`_ 

615 on Wikiversity. 

616 .. [2] Carl de Boor, "A Practical Guide to Splines", Springer-Verlag, 1978. 

617 """ 

618 def __init__(self, x, y, axis=0, bc_type='not-a-knot', extrapolate=None): 

619 x, dx, y, axis, _ = prepare_input(x, y, axis) 

620 n = len(x) 

621 

622 bc, y = self._validate_bc(bc_type, y, y.shape[1:], axis) 

623 

624 if extrapolate is None: 

625 if bc[0] == 'periodic': 

626 extrapolate = 'periodic' 

627 else: 

628 extrapolate = True 

629 

630 dxr = dx.reshape([dx.shape[0]] + [1] * (y.ndim - 1)) 

631 slope = np.diff(y, axis=0) / dxr 

632 

633 # If bc is 'not-a-knot' this change is just a convention. 

634 # If bc is 'periodic' then we already checked that y[0] == y[-1], 

635 # and the spline is just a constant, we handle this case in the same 

636 # way by setting the first derivatives to slope, which is 0. 

637 if n == 2: 

638 if bc[0] in ['not-a-knot', 'periodic']: 

639 bc[0] = (1, slope[0]) 

640 if bc[1] in ['not-a-knot', 'periodic']: 

641 bc[1] = (1, slope[0]) 

642 

643 # This is a very special case, when both conditions are 'not-a-knot' 

644 # and n == 3. In this case 'not-a-knot' can't be handled regularly 

645 # as the both conditions are identical. We handle this case by 

646 # constructing a parabola passing through given points. 

647 if n == 3 and bc[0] == 'not-a-knot' and bc[1] == 'not-a-knot': 

648 A = np.zeros((3, 3)) # This is a standard matrix. 

649 b = np.empty((3,) + y.shape[1:], dtype=y.dtype) 

650 

651 A[0, 0] = 1 

652 A[0, 1] = 1 

653 A[1, 0] = dx[1] 

654 A[1, 1] = 2 * (dx[0] + dx[1]) 

655 A[1, 2] = dx[0] 

656 A[2, 1] = 1 

657 A[2, 2] = 1 

658 

659 b[0] = 2 * slope[0] 

660 b[1] = 3 * (dxr[0] * slope[1] + dxr[1] * slope[0]) 

661 b[2] = 2 * slope[1] 

662 

663 s = solve(A, b, overwrite_a=True, overwrite_b=True, 

664 check_finite=False) 

665 else: 

666 # Find derivative values at each x[i] by solving a tridiagonal 

667 # system. 

668 A = np.zeros((3, n)) # This is a banded matrix representation. 

669 b = np.empty((n,) + y.shape[1:], dtype=y.dtype) 

670 

671 # Filling the system for i=1..n-2 

672 # (x[i-1] - x[i]) * s[i-1] +\ 

673 # 2 * ((x[i] - x[i-1]) + (x[i+1] - x[i])) * s[i] +\ 

674 # (x[i] - x[i-1]) * s[i+1] =\ 

675 # 3 * ((x[i+1] - x[i])*(y[i] - y[i-1])/(x[i] - x[i-1]) +\ 

676 # (x[i] - x[i-1])*(y[i+1] - y[i])/(x[i+1] - x[i])) 

677 

678 A[1, 1:-1] = 2 * (dx[:-1] + dx[1:]) # The diagonal 

679 A[0, 2:] = dx[:-1] # The upper diagonal 

680 A[-1, :-2] = dx[1:] # The lower diagonal 

681 

682 b[1:-1] = 3 * (dxr[1:] * slope[:-1] + dxr[:-1] * slope[1:]) 

683 

684 bc_start, bc_end = bc 

685 

686 if bc_start == 'periodic': 

687 # Due to the periodicity, and because y[-1] = y[0], the linear 

688 # system has (n-1) unknowns/equations instead of n: 

689 A = A[:, 0:-1] 

690 A[1, 0] = 2 * (dx[-1] + dx[0]) 

691 A[0, 1] = dx[-1] 

692 

693 b = b[:-1] 

694 

695 # Also, due to the periodicity, the system is not tri-diagonal. 

696 # We need to compute a "condensed" matrix of shape (n-2, n-2). 

697 # See https://web.archive.org/web/20151220180652/http://www.cfm.brown.edu/people/gk/chap6/node14.html 

698 # for more explanations. 

699 # The condensed matrix is obtained by removing the last column 

700 # and last row of the (n-1, n-1) system matrix. The removed 

701 # values are saved in scalar variables with the (n-1, n-1) 

702 # system matrix indices forming their names: 

703 a_m1_0 = dx[-2] # lower left corner value: A[-1, 0] 

704 a_m1_m2 = dx[-1] 

705 a_m1_m1 = 2 * (dx[-1] + dx[-2]) 

706 a_m2_m1 = dx[-2] 

707 a_0_m1 = dx[0] 

708 

709 b[0] = 3 * (dxr[0] * slope[-1] + dxr[-1] * slope[0]) 

710 b[-1] = 3 * (dxr[-1] * slope[-2] + dxr[-2] * slope[-1]) 

711 

712 Ac = A[:, :-1] 

713 b1 = b[:-1] 

714 b2 = np.zeros_like(b1) 

715 b2[0] = -a_0_m1 

716 b2[-1] = -a_m2_m1 

717 

718 # s1 and s2 are the solutions of (n-2, n-2) system 

719 s1 = solve_banded((1, 1), Ac, b1, overwrite_ab=False, 

720 overwrite_b=False, check_finite=False) 

721 

722 s2 = solve_banded((1, 1), Ac, b2, overwrite_ab=False, 

723 overwrite_b=False, check_finite=False) 

724 

725 # computing the s[n-2] solution: 

726 s_m1 = ((b[-1] - a_m1_0 * s1[0] - a_m1_m2 * s1[-1]) / 

727 (a_m1_m1 + a_m1_0 * s2[0] + a_m1_m2 * s2[-1])) 

728 

729 # s is the solution of the (n, n) system: 

730 s = np.empty((n,) + y.shape[1:], dtype=y.dtype) 

731 s[:-2] = s1 + s_m1 * s2 

732 s[-2] = s_m1 

733 s[-1] = s[0] 

734 else: 

735 if bc_start == 'not-a-knot': 

736 A[1, 0] = dx[1] 

737 A[0, 1] = x[2] - x[0] 

738 d = x[2] - x[0] 

739 b[0] = ((dxr[0] + 2*d) * dxr[1] * slope[0] + 

740 dxr[0]**2 * slope[1]) / d 

741 elif bc_start[0] == 1: 

742 A[1, 0] = 1 

743 A[0, 1] = 0 

744 b[0] = bc_start[1] 

745 elif bc_start[0] == 2: 

746 A[1, 0] = 2 * dx[0] 

747 A[0, 1] = dx[0] 

748 b[0] = -0.5 * bc_start[1] * dx[0]**2 + 3 * (y[1] - y[0]) 

749 

750 if bc_end == 'not-a-knot': 

751 A[1, -1] = dx[-2] 

752 A[-1, -2] = x[-1] - x[-3] 

753 d = x[-1] - x[-3] 

754 b[-1] = ((dxr[-1]**2*slope[-2] + 

755 (2*d + dxr[-1])*dxr[-2]*slope[-1]) / d) 

756 elif bc_end[0] == 1: 

757 A[1, -1] = 1 

758 A[-1, -2] = 0 

759 b[-1] = bc_end[1] 

760 elif bc_end[0] == 2: 

761 A[1, -1] = 2 * dx[-1] 

762 A[-1, -2] = dx[-1] 

763 b[-1] = 0.5 * bc_end[1] * dx[-1]**2 + 3 * (y[-1] - y[-2]) 

764 

765 s = solve_banded((1, 1), A, b, overwrite_ab=True, 

766 overwrite_b=True, check_finite=False) 

767 

768 super(CubicSpline, self).__init__(x, y, s, axis=0, 

769 extrapolate=extrapolate) 

770 self.axis = axis

774 """Validate and prepare boundary conditions.

778 validated_bc : 2-tuple 

779 Boundary conditions for a curve start and end.

785 if bc_type == 'periodic': 

786 if not np.allclose(y[0], y[-1], rtol=1e-15, atol=1e-15): 

787 raise ValueError(

789 "be identical (within machine precision) when "

793 

794 else: 

795 if len(bc_type) != 2:

806 if isinstance(bc, string_types):

808 validated_bc.append((1, np.zeros(expected_deriv_shape))) 

809 elif bc == 'natural': 

810 validated_bc.append((2, np.zeros(expected_deriv_shape))) 

811 elif bc in ['not-a-knot', 'periodic']: 

812 validated_bc.append(bc)

815 else:

831 

832 if np.issubdtype(deriv_value.dtype, np.complexfloating): 

833 y = y.astype(complex, copy=False) 

834 

835 validated_bc.append((deriv_order, deriv_value)) 

836
