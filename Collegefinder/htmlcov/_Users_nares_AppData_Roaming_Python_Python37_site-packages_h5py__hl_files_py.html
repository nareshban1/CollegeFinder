153def make_fid(name, mode, userblock_size, fapl, fcpl=None, swmr=False): 

154 """ Get a new FileID by opening or creating a file. 

155 Also validates mode argument.""" 

156 

157 if userblock_size is not None: 

158 if mode in ('r', 'r+'): 

159 raise ValueError("User block may only be specified " 

160 "when creating a file") 

161 try: 

162 userblock_size = int(userblock_size) 

163 except (TypeError, ValueError): 

164 raise ValueError("User block size must be an integer") 

165 if fcpl is None: 

166 fcpl = h5p.create(h5p.FILE_CREATE) 

167 fcpl.set_userblock(userblock_size) 

168 

169 if mode == 'r': 

170 flags = h5f.ACC_RDONLY 

171 if swmr and swmr_support: 

172 flags |= h5f.ACC_SWMR_READ 

173 fid = h5f.open(name, flags, fapl=fapl) 

174 elif mode == 'r+': 

175 fid = h5f.open(name, h5f.ACC_RDWR, fapl=fapl) 

176 elif mode in ['w-', 'x']: 

177 fid = h5f.create(name, h5f.ACC_EXCL, fapl=fapl, fcpl=fcpl) 

178 elif mode == 'w': 

179 fid = h5f.create(name, h5f.ACC_TRUNC, fapl=fapl, fcpl=fcpl) 

180 elif mode == 'a': 

181 # Open in append mode (read/write). 

182 # If that fails, create a new file only if it won't clobber an 

183 # existing one (ACC_EXCL) 

184 try: 

185 fid = h5f.open(name, h5f.ACC_RDWR, fapl=fapl) 

186 except IOError: 

187 fid = h5f.create(name, h5f.ACC_EXCL, fapl=fapl, fcpl=fcpl) 

188 elif mode is None: 

189 warn("The default file mode will change to 'r' (read-only) in h5py 3.0. " 

190 "To suppress this warning, pass the mode you need to h5py.File(), " 

191 "or set the global default h5.get_config().default_file_mode, " 

192 "or set the environment variable H5PY_DEFAULT_READONLY=1. " 

193 "Available modes are: 'r', 'r+', 'w', 'w-'/'x', 'a'. " 

194 "See the docs for details.", H5pyDeprecationWarning, stacklevel=3) 

195 # Try to open in append mode (read/write). 

196 # If that fails, try readonly, and finally create a new file only 

197 # if it won't clobber an existing file (ACC_EXCL). 

198 try: 

199 fid = h5f.open(name, h5f.ACC_RDWR, fapl=fapl) 

200 except IOError: 

201 try: 

202 fid = h5f.open(name, h5f.ACC_RDONLY, fapl=fapl) 

203 except IOError: 

204 fid = h5f.create(name, h5f.ACC_EXCL, fapl=fapl, fcpl=fcpl) 

205 else: 

206 raise ValueError("Invalid mode; must be one of r, r+, w, w-, x, a") 

207 

208 try: 

209 if userblock_size is not None: 

210 existing_fcpl = fid.get_create_plist() 

211 if existing_fcpl.get_userblock() != userblock_size: 

212 raise ValueError("Requested userblock size (%d) does not match that of existing file (%d)" % (userblock_size, existing_fcpl.get_userblock())) 

213 except: 

214 fid.close() 

215 raise 

216 

217 return fid
