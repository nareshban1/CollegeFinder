167def rosen(x): 

168 """ 

169 The Rosenbrock function. 

170 

171 The function computed is:: 

172 

173 sum(100.0*(x[1:] - x[:-1]**2.0)**2.0 + (1 - x[:-1])**2.0) 

174 

175 Parameters 

176 ---------- 

177 x : array_like 

178 1-D array of points at which the Rosenbrock function is to be computed. 

179 

180 Returns 

181 ------- 

182 f : float 

183 The value of the Rosenbrock function. 

184 

185 See Also 

186 -------- 

187 rosen_der, rosen_hess, rosen_hess_prod 

188 

189 Examples 

190 -------- 

191 >>> from scipy.optimize import rosen 

192 >>> X = 0.1 * np.arange(10) 

193 >>> rosen(X) 

194 76.56 

195 

196 """ 

197 x = asarray(x) 

198 r = numpy.sum(100.0 * (x[1:] - x[:-1]**2.0)**2.0 + (1 - x[:-1])**2.0, 

199 axis=0) 

200 return r

203def rosen_der(x): 

204 """ 

205 The derivative (i.e. gradient) of the Rosenbrock function. 

206 

207 Parameters 

208 ---------- 

209 x : array_like 

210 1-D array of points at which the derivative is to be computed. 

211 

212 Returns 

213 ------- 

214 rosen_der : (N,) ndarray 

215 The gradient of the Rosenbrock function at `x`. 

216 

217 See Also 

218 -------- 

219 rosen, rosen_hess, rosen_hess_prod 

220 

221 Examples 

222 -------- 

223 >>> from scipy.optimize import rosen_der 

224 >>> X = 0.1 * np.arange(9) 

225 >>> rosen_der(X) 

226 array([ -2. , 10.6, 15.6, 13.4, 6.4, -3. , -12.4, -19.4, 62. ]) 

227 

228 """ 

229 x = asarray(x) 

230 xm = x[1:-1] 

231 xm_m1 = x[:-2] 

232 xm_p1 = x[2:] 

233 der = numpy.zeros_like(x) 

234 der[1:-1] = (200 * (xm - xm_m1**2) - 

235 400 * (xm_p1 - xm**2) * xm - 2 * (1 - xm)) 

236 der[0] = -400 * x[0] * (x[1] - x[0]**2) - 2 * (1 - x[0]) 

237 der[-1] = 200 * (x[-1] - x[-2]**2) 

238 return der

241def rosen_hess(x): 

242 """ 

243 The Hessian matrix of the Rosenbrock function. 

244 

245 Parameters 

246 ---------- 

247 x : array_like 

248 1-D array of points at which the Hessian matrix is to be computed. 

249 

250 Returns 

251 ------- 

252 rosen_hess : ndarray 

253 The Hessian matrix of the Rosenbrock function at `x`. 

254 

255 See Also 

256 -------- 

257 rosen, rosen_der, rosen_hess_prod 

258 

259 Examples 

260 -------- 

261 >>> from scipy.optimize import rosen_hess 

262 >>> X = 0.1 * np.arange(4) 

263 >>> rosen_hess(X) 

264 array([[-38., 0., 0., 0.], 

265 [ 0., 134., -40., 0.], 

266 [ 0., -40., 130., -80.], 

267 [ 0., 0., -80., 200.]]) 

268 

269 """ 

270 x = atleast_1d(x) 

271 H = numpy.diag(-400 * x[:-1], 1) - numpy.diag(400 * x[:-1], -1) 

272 diagonal = numpy.zeros(len(x), dtype=x.dtype) 

273 diagonal[0] = 1200 * x[0]**2 - 400 * x[1] + 2 

274 diagonal[-1] = 200 

275 diagonal[1:-1] = 202 + 1200 * x[1:-1]**2 - 400 * x[2:] 

276 H = H + numpy.diag(diagonal) 

277 return H

280def rosen_hess_prod(x, p): 

281 """ 

282 Product of the Hessian matrix of the Rosenbrock function with a vector. 

283 

284 Parameters 

285 ---------- 

286 x : array_like 

287 1-D array of points at which the Hessian matrix is to be computed. 

288 p : array_like 

289 1-D array, the vector to be multiplied by the Hessian matrix. 

290 

291 Returns 

292 ------- 

293 rosen_hess_prod : ndarray 

294 The Hessian matrix of the Rosenbrock function at `x` multiplied 

295 by the vector `p`. 

296 

297 See Also 

298 -------- 

299 rosen, rosen_der, rosen_hess 

300 

301 Examples 

302 -------- 

303 >>> from scipy.optimize import rosen_hess_prod 

304 >>> X = 0.1 * np.arange(9) 

305 >>> p = 0.5 * np.arange(9) 

306 >>> rosen_hess_prod(X, p) 

307 array([ -0., 27., -10., -95., -192., -265., -278., -195., -180.]) 

308 

309 """ 

310 x = atleast_1d(x) 

311 Hp = numpy.zeros(len(x), dtype=x.dtype) 

312 Hp[0] = (1200 * x[0]**2 - 400 * x[1] + 2) * p[0] - 400 * x[0] * p[1] 

313 Hp[1:-1] = (-400 * x[:-2] * p[:-2] + 

314 (202 + 1200 * x[1:-1]**2 - 400 * x[2:]) * p[1:-1] - 

315 400 * x[1:-1] * p[2:]) 

316 Hp[-1] = -400 * x[-2] * p[-2] + 200*p[-1] 

317 return Hp

332def fmin(func, x0, args=(), xtol=1e-4, ftol=1e-4, maxiter=None, maxfun=None, 

333 full_output=0, disp=1, retall=0, callback=None, initial_simplex=None): 

334 """ 

335 Minimize a function using the downhill simplex algorithm. 

336 

337 This algorithm only uses function values, not derivatives or second 

338 derivatives. 

339 

340 Parameters 

341 ---------- 

342 func : callable func(x,*args) 

343 The objective function to be minimized. 

344 x0 : ndarray 

345 Initial guess. 

346 args : tuple, optional 

347 Extra arguments passed to func, i.e. ``f(x,*args)``. 

348 xtol : float, optional 

349 Absolute error in xopt between iterations that is acceptable for 

350 convergence. 

351 ftol : number, optional 

352 Absolute error in func(xopt) between iterations that is acceptable for 

353 convergence. 

354 maxiter : int, optional 

355 Maximum number of iterations to perform. 

356 maxfun : number, optional 

357 Maximum number of function evaluations to make. 

358 full_output : bool, optional 

359 Set to True if fopt and warnflag outputs are desired. 

360 disp : bool, optional 

361 Set to True to print convergence messages. 

362 retall : bool, optional 

363 Set to True to return list of solutions at each iteration. 

364 callback : callable, optional 

365 Called after each iteration, as callback(xk), where xk is the 

366 current parameter vector. 

367 initial_simplex : array_like of shape (N + 1, N), optional 

368 Initial simplex. If given, overrides `x0`. 

369 ``initial_simplex[j,:]`` should contain the coordinates of 

370 the j-th vertex of the ``N+1`` vertices in the simplex, where 

371 ``N`` is the dimension. 

372 

373 Returns 

374 ------- 

375 xopt : ndarray 

376 Parameter that minimizes function. 

377 fopt : float 

378 Value of function at minimum: ``fopt = func(xopt)``. 

379 iter : int 

380 Number of iterations performed. 

381 funcalls : int 

382 Number of function calls made. 

383 warnflag : int 

384 1 : Maximum number of function evaluations made. 

385 2 : Maximum number of iterations reached. 

386 allvecs : list 

387 Solution at each iteration. 

388 

389 See also 

390 -------- 

391 minimize: Interface to minimization algorithms for multivariate 

392 functions. See the 'Nelder-Mead' `method` in particular. 

393 

394 Notes 

395 ----- 

396 Uses a Nelder-Mead simplex algorithm to find the minimum of function of 

397 one or more variables. 

398 

399 This algorithm has a long history of successful use in applications. 

400 But it will usually be slower than an algorithm that uses first or 

401 second derivative information. In practice it can have poor 

402 performance in high-dimensional problems and is not robust to 

403 minimizing complicated functions. Additionally, there currently is no 

404 complete theory describing when the algorithm will successfully 

405 converge to the minimum, or how fast it will if it does. Both the ftol and 

406 xtol criteria must be met for convergence. 

407 

408 Examples 

409 -------- 

410 >>> def f(x): 

411 ... return x**2 

412 

413 >>> from scipy import optimize 

414 

415 >>> minimum = optimize.fmin(f, 1) 

416 Optimization terminated successfully. 

417 Current function value: 0.000000 

418 Iterations: 17 

419 Function evaluations: 34

421 -8.8817841970012523e-16

427

438 'maxfev': maxfun, 

439 'disp': disp,

446 if retall:

453 return res['x']

456def _minimize_neldermead(func, x0, args=(), callback=None, 

457 maxiter=None, maxfev=None, disp=False, 

458 return_all=False, initial_simplex=None, 

459 xatol=1e-4, fatol=1e-4, adaptive=False, 

460 **unknown_options): 

461 """ 

462 Minimization of scalar function of one or more variables using the 

463 Nelder-Mead algorithm. 

464 

465 Options 

466 ------- 

467 disp : bool 

468 Set to True to print convergence messages. 

469 maxiter, maxfev : int 

470 Maximum allowed number of iterations and function evaluations. 

471 Will default to ``N*200``, where ``N`` is the number of 

472 variables, if neither `maxiter` or `maxfev` is set. If both 

473 `maxiter` and `maxfev` are set, minimization will stop at the 

474 first reached. 

475 initial_simplex : array_like of shape (N + 1, N) 

476 Initial simplex. If given, overrides `x0`. 

477 ``initial_simplex[j,:]`` should contain the coordinates of 

478 the j-th vertex of the ``N+1`` vertices in the simplex, where 

479 ``N`` is the dimension. 

480 xatol : float, optional 

481 Absolute error in xopt between iterations that is acceptable for 

482 convergence. 

483 fatol : number, optional 

484 Absolute error in func(xopt) between iterations that is acceptable for 

485 convergence. 

486 adaptive : bool, optional 

487 Adapt algorithm parameters to dimensionality of problem. Useful for 

488 high-dimensional minimization [1]_. 

489 

490 References 

491 ---------- 

492 .. [1] Gao, F. and Han, L. 

493 Implementing the Nelder-Mead simplex algorithm with adaptive 

494 parameters. 2012. Computational Optimization and Applications. 

495 51:1, pp. 259-277 

496 

497 """ 

498 if 'ftol' in unknown_options: 

499 warnings.warn("ftol is deprecated for Nelder-Mead," 

500 " use fatol instead. If you specified both, only" 

501 " fatol is used.", 

502 DeprecationWarning) 

503 if (np.isclose(fatol, 1e-4) and 

504 not np.isclose(unknown_options['ftol'], 1e-4)): 

505 # only ftol was probably specified, use it. 

506 fatol = unknown_options['ftol'] 

507 unknown_options.pop('ftol') 

508 if 'xtol' in unknown_options: 

509 warnings.warn("xtol is deprecated for Nelder-Mead," 

510 " use xatol instead. If you specified both, only" 

511 " xatol is used.", 

512 DeprecationWarning) 

513 if (np.isclose(xatol, 1e-4) and 

514 not np.isclose(unknown_options['xtol'], 1e-4)): 

515 # only xtol was probably specified, use it. 

516 xatol = unknown_options['xtol'] 

517 unknown_options.pop('xtol') 

518 

519 _check_unknown_options(unknown_options) 

520 maxfun = maxfev 

521 retall = return_all 

522 

523 fcalls, func = wrap_function(func, args) 

524 

525 if adaptive: 

526 dim = float(len(x0)) 

527 rho = 1 

528 chi = 1 + 2/dim 

529 psi = 0.75 - 1/(2*dim) 

530 sigma = 1 - 1/dim 

531 else: 

532 rho = 1 

533 chi = 2 

534 psi = 0.5 

535 sigma = 0.5 

536 

537 nonzdelt = 0.05 

538 zdelt = 0.00025 

539 

540 x0 = asfarray(x0).flatten() 

541 

542 if initial_simplex is None: 

543 N = len(x0) 

544 

545 sim = numpy.zeros((N + 1, N), dtype=x0.dtype) 

546 sim[0] = x0 

547 for k in range(N): 

548 y = numpy.array(x0, copy=True) 

549 if y[k] != 0: 

550 y[k] = (1 + nonzdelt)*y[k] 

551 else: 

552 y[k] = zdelt 

553 sim[k + 1] = y 

554 else: 

555 sim = np.asfarray(initial_simplex).copy() 

556 if sim.ndim != 2 or sim.shape[0] != sim.shape[1] + 1: 

557 raise ValueError("`initial_simplex` should be an array of shape (N+1,N)") 

558 if len(x0) != sim.shape[1]: 

559 raise ValueError("Size of `initial_simplex` is not consistent with `x0`") 

560 N = sim.shape[1] 

561 

562 if retall: 

563 allvecs = [sim[0]] 

564 

565 # If neither are set, then set both to default 

566 if maxiter is None and maxfun is None: 

567 maxiter = N * 200 

568 maxfun = N * 200 

569 elif maxiter is None: 

570 # Convert remaining Nones, to np.inf, unless the other is np.inf, in 

571 # which case use the default to avoid unbounded iteration 

572 if maxfun == np.inf: 

573 maxiter = N * 200 

574 else: 

575 maxiter = np.inf 

576 elif maxfun is None: 

577 if maxiter == np.inf: 

578 maxfun = N * 200 

579 else: 

580 maxfun = np.inf 

581 

582 one2np1 = list(range(1, N + 1)) 

583 fsim = numpy.zeros((N + 1,), float) 

584 

585 for k in range(N + 1): 

586 fsim[k] = func(sim[k]) 

587 

588 ind = numpy.argsort(fsim) 

589 fsim = numpy.take(fsim, ind, 0) 

590 # sort so sim[0,:] has the lowest function value 

591 sim = numpy.take(sim, ind, 0) 

592 

593 iterations = 1 

594 

595 while (fcalls[0] < maxfun and iterations < maxiter): 

596 if (numpy.max(numpy.ravel(numpy.abs(sim[1:] - sim[0]))) <= xatol and 

597 numpy.max(numpy.abs(fsim[0] - fsim[1:])) <= fatol): 

598 break 

599 

600 xbar = numpy.add.reduce(sim[:-1], 0) / N 

601 xr = (1 + rho) * xbar - rho * sim[-1] 

602 fxr = func(xr) 

603 doshrink = 0 

604 

605 if fxr < fsim[0]: 

606 xe = (1 + rho * chi) * xbar - rho * chi * sim[-1] 

607 fxe = func(xe) 

608 

609 if fxe < fxr: 

610 sim[-1] = xe 

611 fsim[-1] = fxe 

612 else: 

613 sim[-1] = xr 

614 fsim[-1] = fxr 

615 else: # fsim[0] <= fxr 

616 if fxr < fsim[-2]: 

617 sim[-1] = xr 

618 fsim[-1] = fxr 

619 else: # fxr >= fsim[-2] 

620 # Perform contraction 

621 if fxr < fsim[-1]: 

622 xc = (1 + psi * rho) * xbar - psi * rho * sim[-1] 

623 fxc = func(xc) 

624 

625 if fxc <= fxr: 

626 sim[-1] = xc 

627 fsim[-1] = fxc 

628 else: 

629 doshrink = 1 

630 else: 

631 # Perform an inside contraction 

632 xcc = (1 - psi) * xbar + psi * sim[-1] 

633 fxcc = func(xcc) 

634 

635 if fxcc < fsim[-1]: 

636 sim[-1] = xcc 

637 fsim[-1] = fxcc 

638 else: 

639 doshrink = 1 

640 

641 if doshrink: 

642 for j in one2np1: 

643 sim[j] = sim[0] + sigma * (sim[j] - sim[0]) 

644 fsim[j] = func(sim[j]) 

645 

646 ind = numpy.argsort(fsim) 

647 sim = numpy.take(sim, ind, 0) 

648 fsim = numpy.take(fsim, ind, 0) 

649 if callback is not None: 

650 callback(sim[0]) 

651 iterations += 1 

652 if retall: 

653 allvecs.append(sim[0]) 

654 

655 x = sim[0] 

656 fval = numpy.min(fsim) 

657 warnflag = 0 

658 

659 if fcalls[0] >= maxfun: 

660 warnflag = 1 

661 msg = _status_message['maxfev'] 

662 if disp: 

663 print('Warning: ' + msg) 

664 elif iterations >= maxiter: 

665 warnflag = 2 

666 msg = _status_message['maxiter'] 

667 if disp: 

668 print('Warning: ' + msg) 

669 else: 

670 msg = _status_message['success'] 

671 if disp: 

672 print(msg) 

673 print(" Current function value: %f" % fval) 

674 print(" Iterations: %d" % iterations) 

675 print(" Function evaluations: %d" % fcalls[0]) 

676 

677 result = OptimizeResult(fun=fval, nit=iterations, nfev=fcalls[0], 

678 status=warnflag, success=(warnflag == 0),

710def approx_fprime(xk, f, epsilon, *args): 

711 """Finite-difference approximation of the gradient of a scalar function. 

712 

713 Parameters 

714 ---------- 

715 xk : array_like 

716 The coordinate vector at which to determine the gradient of `f`. 

717 f : callable 

718 The function of which to determine the gradient (partial derivatives). 

719 Should take `xk` as first argument, other arguments to `f` can be 

720 supplied in ``*args``. Should return a scalar, the value of the 

721 function at `xk`. 

722 epsilon : array_like 

723 Increment to `xk` to use for determining the function gradient. 

724 If a scalar, uses the same finite difference delta for all partial 

725 derivatives. If an array, should contain one value per element of 

726 `xk`. 

727 \\*args : args, optional 

728 Any other arguments that are to be passed to `f`. 

729 

730 Returns 

731 ------- 

732 grad : ndarray 

733 The partial derivatives of `f` to `xk`. 

734 

735 See Also 

736 -------- 

737 check_grad : Check correctness of gradient function against approx_fprime. 

738 

739 Notes 

740 ----- 

741 The function gradient is determined by the forward finite difference 

742 formula:: 

743 

744 f(xk[i] + epsilon[i]) - f(xk[i]) 

745 f'[i] = --------------------------------- 

746 epsilon[i] 

747 

748 The main use of `approx_fprime` is in scalar function optimizers like 

749 `fmin_bfgs`, to determine numerically the Jacobian of a function. 

750 

751 Examples 

752 --------

754 >>> def func(x, c0, c1):

757 

758 >>> x = np.ones(2) 

759 >>> c0, c1 = (1, 200) 

760 >>> eps = np.sqrt(np.finfo(float).eps) 

761 >>> optimize.approx_fprime(x, func, [eps, np.sqrt(200) * eps], c0, c1) 

762 array([ 2. , 400.00004198]) 

763 

764 """ 

765 return _approx_fprime_helper(xk, f, epsilon, args=args)

973 BFGS algorithm.

975 Options 

976 ------- 

977 disp : bool 

978 Set to True to print convergence messages. 

979 maxiter : int 

980 Maximum number of iterations to perform. 

981 gtol : float 

982 Gradient norm must be less than `gtol` before successful 

983 termination. 

984 norm : float 

985 Order of norm (Inf is max, -Inf is min). 

986 eps : float or ndarray 

987 If `jac` is approximated, use this value for the step size. 

988 

989 """ 

990 _check_unknown_options(unknown_options) 

991 f = fun 

992 fprime = jac 

993 epsilon = eps 

994 retall = return_all 

995 

996 x0 = asarray(x0).flatten() 

997 if x0.ndim == 0: 

998 x0.shape = (1,) 

999 if maxiter is None: 

1000 maxiter = len(x0) * 200 

1001 func_calls, f = wrap_function(f, args) 

1002 

1003 old_fval = f(x0) 

1004 

1005 if fprime is None: 

1006 grad_calls, myfprime = wrap_function(approx_fprime, (f, epsilon)) 

1007 else: 

1008 grad_calls, myfprime = wrap_function(fprime, args) 

1009 gfk = myfprime(x0) 

1010 k = 0 

1011 N = len(x0) 

1012 I = numpy.eye(N, dtype=int) 

1013 Hk = I 

1014 

1015 # Sets the initial step guess to dx ~ 1 

1016 old_old_fval = old_fval + np.linalg.norm(gfk) / 2 

1017 

1018 xk = x0 

1019 if retall: 

1020 allvecs = [x0] 

1021 warnflag = 0 

1022 gnorm = vecnorm(gfk, ord=norm) 

1023 while (gnorm > gtol) and (k < maxiter): 

1024 pk = -numpy.dot(Hk, gfk) 

1025 try: 

1026 alpha_k, fc, gc, old_fval, old_old_fval, gfkp1 = \ 

1027 _line_search_wolfe12(f, myfprime, xk, pk, gfk, 

1028 old_fval, old_old_fval, amin=1e-100, amax=1e100) 

1029 except _LineSearchError: 

1030 # Line search failed to find a better solution. 

1031 warnflag = 2 

1032 break 

1033 

1034 xkp1 = xk + alpha_k * pk 

1035 if retall: 

1036 allvecs.append(xkp1) 

1037 sk = xkp1 - xk 

1038 xk = xkp1 

1039 if gfkp1 is None: 

1040 gfkp1 = myfprime(xkp1) 

1041 

1042 yk = gfkp1 - gfk 

1043 gfk = gfkp1 

1044 if callback is not None: 

1045 callback(xk) 

1046 k += 1 

1047 gnorm = vecnorm(gfk, ord=norm) 

1048 if (gnorm <= gtol): 

1049 break 

1050 

1051 if not numpy.isfinite(old_fval): 

1052 # We correctly found +-Inf as optimal value, or something went 

1053 # wrong. 

1054 warnflag = 2 

1055 break 

1056 

1057 try: # this was handled in numeric, let it remaines for more safety 

1058 rhok = 1.0 / (numpy.dot(yk, sk)) 

1059 except ZeroDivisionError: 

1060 rhok = 1000.0 

1061 if disp: 

1062 print("Divide-by-zero encountered: rhok assumed large") 

1063 if isinf(rhok): # this is patch for numpy 

1064 rhok = 1000.0 

1065 if disp: 

1066 print("Divide-by-zero encountered: rhok assumed large") 

1067 A1 = I - sk[:, numpy.newaxis] * yk[numpy.newaxis, :] * rhok 

1068 A2 = I - yk[:, numpy.newaxis] * sk[numpy.newaxis, :] * rhok 

1069 Hk = numpy.dot(A1, numpy.dot(Hk, A2)) + (rhok * sk[:, numpy.newaxis] * 

1070 sk[numpy.newaxis, :]) 

1071 

1072 fval = old_fval 

1073 

1074 if warnflag == 2: 

1075 msg = _status_message['pr_loss'] 

1076 elif k >= maxiter: 

1077 warnflag = 1 

1078 msg = _status_message['maxiter'] 

1079 elif np.isnan(gnorm) or np.isnan(fval) or np.isnan(xk).any(): 

1080 warnflag = 3 

1081 msg = _status_message['nan'] 

1082 else: 

1083 msg = _status_message['success'] 

1084 

1085 if disp: 

1086 print("%s%s" % ("Warning: " if warnflag != 0 else "", msg)) 

1087 print(" Current function value: %f" % fval) 

1088 print(" Iterations: %d" % k) 

1089 print(" Function evaluations: %d" % func_calls[0]) 

1090 print(" Gradient evaluations: %d" % grad_calls[0]) 

1091 

1092 result = OptimizeResult(fun=fval, jac=gfk, hess_inv=Hk, nfev=func_calls[0], 

1093 njev=grad_calls[0], status=warnflag, 

1094 success=(warnflag == 0), message=msg, x=xk, 

1095 nit=k) 

1096 if retall: 

1097 result['allvecs'] = allvecs 

1098 return result

1306 func_calls, f = wrap_function(f, args) 

1307 if fprime is None: 

1308 grad_calls, myfprime = wrap_function(approx_fprime, (f, epsilon)) 

1309 else: 

1310 grad_calls, myfprime = wrap_function(fprime, args) 

1311 gfk = myfprime(x0) 

1312 k = 0 

1313 xk = x0 

1314 

1315 # Sets the initial step guess to dx ~ 1 

1316 old_fval = f(xk) 

1317 old_old_fval = old_fval + np.linalg.norm(gfk) / 2 

1318 

1319 if retall: 

1320 allvecs = [xk] 

1321 warnflag = 0 

1322 pk = -gfk 

1323 gnorm = vecnorm(gfk, ord=norm) 

1324 

1325 sigma_3 = 0.01 

1326 

1327 while (gnorm > gtol) and (k < maxiter): 

1328 deltak = numpy.dot(gfk, gfk) 

1329 

1330 cached_step = [None] 

1331 

1332 def polak_ribiere_powell_step(alpha, gfkp1=None): 

1333 xkp1 = xk + alpha * pk 

1334 if gfkp1 is None: 

1335 gfkp1 = myfprime(xkp1) 

1336 yk = gfkp1 - gfk 

1337 beta_k = max(0, numpy.dot(yk, gfkp1) / deltak) 

1338 pkp1 = -gfkp1 + beta_k * pk 

1339 gnorm = vecnorm(gfkp1, ord=norm) 

1340 return (alpha, xkp1, pkp1, gfkp1, gnorm)

1342 def descent_condition(alpha, xkp1, fp1, gfkp1): 

1343 # Polak-Ribiere+ needs an explicit check of a sufficient 

1344 # descent condition, which is not guaranteed by strong Wolfe. 

1345 # 

1346 # See Gilbert & Nocedal, "Global convergence properties of 

1347 # conjugate gradient methods for optimization", 

1348 # SIAM J. Optimization 2, 21 (1992). 

1349 cached_step[:] = polak_ribiere_powell_step(alpha, gfkp1) 

1350 alpha, xk, pk, gfk, gnorm = cached_step 

1351 

1352 # Accept step if it leads to convergence. 

1353 if gnorm <= gtol: 

1354 return True 

1355 

1356 # Accept step if sufficient descent condition applies. 

1357 return numpy.dot(pk, gfk) <= -sigma_3 * numpy.dot(gfk, gfk) 

1358 

1359 try: 

1360 alpha_k, fc, gc, old_fval, old_old_fval, gfkp1 = \ 

1361 _line_search_wolfe12(f, myfprime, xk, pk, gfk, old_fval, 

1362 old_old_fval, c2=0.4, amin=1e-100, amax=1e100, 

1363 extra_condition=descent_condition) 

1364 except _LineSearchError: 

1365 # Line search failed to find a better solution. 

1366 warnflag = 2 

1367 break 

1368 

1369 # Reuse already computed results if possible 

1370 if alpha_k == cached_step[0]: 

1371 alpha_k, xk, pk, gfk, gnorm = cached_step 

1372 else: 

1373 alpha_k, xk, pk, gfk, gnorm = polak_ribiere_powell_step(alpha_k, gfkp1) 

1374 

1375 if retall: 

1376 allvecs.append(xk) 

1377 if callback is not None: 

1378 callback(xk) 

1379 k += 1 

1380 

1381 fval = old_fval 

1382 if warnflag == 2: 

1383 msg = _status_message['pr_loss'] 

1384 elif k >= maxiter: 

1385 warnflag = 1 

1386 msg = _status_message['maxiter'] 

1387 elif np.isnan(gnorm) or np.isnan(fval) or np.isnan(xk).any(): 

1388 warnflag = 3 

1389 msg = _status_message['nan'] 

1390 else: 

1391 msg = _status_message['success'] 

1392 

1393 if disp: 

1394 print("%s%s" % ("Warning: " if warnflag != 0 else "", msg)) 

1395 print(" Current function value: %f" % fval) 

1396 print(" Iterations: %d" % k) 

1397 print(" Function evaluations: %d" % func_calls[0]) 

1398 print(" Gradient evaluations: %d" % grad_calls[0]) 

1399 

1400 result = OptimizeResult(fun=fval, jac=gfk, nfev=func_calls[0], 

1401 njev=grad_calls[0], status=warnflag, 

1402 success=(warnflag == 0), message=msg, x=xk, 

1403 nit=k) 

1404 if retall: 

1405 result['allvecs'] = allvecs 

1406 return result

1418 Objective function to be minimized.

1434 callback : callable, optional

1440 the minimizer falls below this amount. 

1441 maxiter : int, optional 

1442 Maximum number of iterations to perform. 

1443 full_output : bool, optional 

1444 If True, return the optional outputs. 

1445 disp : bool, optional

1448 If True, return a list of results at each iteration. 

1449 

1450 Returns 

1451 -------

1453 Parameters which minimize f, i.e. ``f(xopt) == fopt``. 

1454 fopt : float 

1455 Value of the function at xopt, i.e. ``fopt = f(xopt)``.

1489 and scipy while scipy.optimize.fmin_tnc calls a C function. 

1490 2. scipy.optimize.fmin_ncg is only for unconstrained minimization

1497 Wright & Nocedal, 'Numerical Optimization', 1999, pg. 140.

1522def _minimize_newtoncg(fun, x0, args=(), jac=None, hess=None, hessp=None, 

1523 callback=None, xtol=1e-5, eps=_epsilon, maxiter=None, 

1524 disp=False, return_all=False, 

1525 **unknown_options): 

1526 """ 

1527 Minimization of scalar function of one or more variables using the 

1528 Newton-CG algorithm. 

1529 

1530 Note that the `jac` parameter (Jacobian) is required. 

1531 

1532 Options 

1533 ------- 

1534 disp : bool 

1535 Set to True to print convergence messages. 

1536 xtol : float 

1537 Average relative error in solution `xopt` acceptable for 

1538 convergence. 

1539 maxiter : int 

1540 Maximum number of iterations to perform. 

1541 eps : float or ndarray

1543

1545 _check_unknown_options(unknown_options) 

1546 if jac is None: 

1547 raise ValueError('Jacobian is required for Newton-CG method')

1549 fprime = jac 

1550 fhess_p = hessp

1566 njev=gcalls[0], nhev=hcalls, status=warnflag,

1568 nit=k) 

1569 if retall: 

1570 result['allvecs'] = allvecs 

1571 return result 

1572 

1573 x0 = asarray(x0).flatten() 

1574 fcalls, f = wrap_function(f, args) 

1575 gcalls, fprime = wrap_function(fprime, args) 

1576 hcalls = 0 

1577 if maxiter is None: 

1578 maxiter = len(x0)*200 

1579 cg_maxiter = 20*len(x0) 

1580 

1581 xtol = len(x0) * avextol 

1582 update = [2 * xtol] 

1583 xk = x0 

1584 if retall: 

1585 allvecs = [xk] 

1586 k = 0 

1587 gfk = None 

1588 old_fval = f(x0) 

1589 old_old_fval = None 

1590 float64eps = numpy.finfo(numpy.float64).eps 

1591 while numpy.add.reduce(numpy.abs(update)) > xtol: 

1592 if k >= maxiter: 

1593 msg = "Warning: " + _status_message['maxiter'] 

1594 return terminate(1, msg) 

1595 # Compute a search direction pk by applying the CG method to 

1596 # del2 f(xk) p = - grad f(xk) starting from 0. 

1597 b = -fprime(xk) 

1598 maggrad = numpy.add.reduce(numpy.abs(b)) 

1599 eta = numpy.min([0.5, numpy.sqrt(maggrad)]) 

1600 termcond = eta * maggrad 

1601 xsupi = zeros(len(x0), dtype=x0.dtype) 

1602 ri = -b 

1603 psupi = -ri 

1604 i = 0 

1605 dri0 = numpy.dot(ri, ri) 

1606 

1607 if fhess is not None: # you want to compute hessian once. 

1608 A = fhess(*(xk,) + args) 

1609 hcalls = hcalls + 1 

1610 

1611 for k2 in xrange(cg_maxiter): 

1612 if numpy.add.reduce(numpy.abs(ri)) <= termcond: 

1613 break 

1614 if fhess is None: 

1615 if fhess_p is None: 

1616 Ap = approx_fhess_p(xk, psupi, fprime, epsilon) 

1617 else: 

1618 Ap = fhess_p(xk, psupi, *args) 

1619 hcalls = hcalls + 1 

1620 else: 

1621 Ap = numpy.dot(A, psupi) 

1622 # check curvature 

1623 Ap = asarray(Ap).squeeze() # get rid of matrices... 

1624 curv = numpy.dot(psupi, Ap) 

1625 if 0 <= curv <= 3 * float64eps: 

1626 break 

1627 elif curv < 0: 

1628 if (i > 0): 

1629 break 

1630 else: 

1631 # fall back to steepest descent direction 

1632 xsupi = dri0 / (-curv) * b 

1633 break 

1634 alphai = dri0 / curv 

1635 xsupi = xsupi + alphai * psupi 

1636 ri = ri + alphai * Ap 

1637 dri1 = numpy.dot(ri, ri) 

1638 betai = dri1 / dri0 

1639 psupi = -ri + betai * psupi 

1640 i = i + 1 

1641 dri0 = dri1 # update numpy.dot(ri,ri) for next time. 

1642 else: 

1643 # curvature keeps increasing, bail out 

1644 msg = ("Warning: CG iterations didn't converge. The Hessian is not " 

1645 "positive definite.") 

1646 return terminate(3, msg) 

1647 

1648 pk = xsupi # search direction is solution to system. 

1649 gfk = -b # gradient at xk 

1650 

1651 try: 

1652 alphak, fc, gc, old_fval, old_old_fval, gfkp1 = \ 

1653 _line_search_wolfe12(f, fprime, xk, pk, gfk, 

1654 old_fval, old_old_fval) 

1655 except _LineSearchError: 

1656 # Line search failed to find a better solution. 

1657 msg = "Warning: " + _status_message['pr_loss'] 

1658 return terminate(2, msg) 

1659 

1660 update = alphak * pk 

1661 xk = xk + update # upcast if necessary 

1662 if callback is not None: 

1663 callback(xk) 

1664 if retall: 

1665 allvecs.append(xk) 

1666 k += 1 

1667 else: 

1668 if np.isnan(old_fval) or np.isnan(update).any(): 

1669 return terminate(3, _status_message['nan']) 

1670 

1671 msg = _status_message['success'] 

1672 return terminate(0, msg)

1753def _minimize_scalar_bounded(func, bounds, args=(),

1759 maxiter : int

1761 disp: int, optional 

1762 If non-zero, print messages. 

1763 0 : no message printing. 

1764 1 : non-convergence notification messages only.

1782 raise ValueError("The lower bound exceeds the upper bound.")

1785 header = ' Func-count x f(x) Procedure'

1787

1789 golden_mean = 0.5 * (3.0 - sqrt(5.0))

1791 fulc = a + golden_mean * (b - a)

1793 rat = e = 0.0

1801 tol1 = sqrt_eps * numpy.abs(xf) + xatol / 3.0 

1802 tol2 = 2.0 * tol1 

1803 

1804 if disp > 2: 

1805 print(" ") 

1806 print(header) 

1807 print("%5.0f %12.6g %12.6g %s" % (fmin_data + (step,))) 

1808

1810 golden = 1 

1811 # Check for parabolic fit 

1812 if numpy.abs(e) > tol1: 

1813 golden = 0 

1814 r = (xf - nfc) * (fx - ffulc) 

1815 q = (xf - fulc) * (fx - fnfc) 

1816 p = (xf - fulc) * q - (xf - nfc) * r 

1817 q = 2.0 * (q - r) 

1818 if q > 0.0: 

1819 p = -p 

1820 q = numpy.abs(q) 

1821 r = e 

1822 e = rat 

1823

1825 if ((numpy.abs(p) < numpy.abs(0.5*q*r)) and (p > q*(a - xf)) and 

1826 (p < q * (b - xf))): 

1827 rat = (p + 0.0) / q 

1828 x = xf + rat 

1829 step = ' parabolic' 

1830 

1831 if ((x - a) < tol2) or ((b - x) < tol2): 

1832 si = numpy.sign(xm - xf) + ((xm - xf) == 0) 

1833 rat = tol1 * si 

1834 else: # do a golden section step 

1835 golden = 1 

1836 

1837 if golden: # Do a golden-section step 

1838 if xf >= xm: 

1839 e = a - xf 

1840 else: 

1841 e = b - xf 

1842 rat = golden_mean*e 

1843 step = ' golden' 

1844 

1845 si = numpy.sign(rat) + (rat == 0) 

1846 x = xf + si * numpy.max([numpy.abs(rat), tol1]) 

1847 fu = func(x, *args) 

1848 num += 1 

1849 fmin_data = (num, x, fu) 

1850 if disp > 2: 

1851 print("%5.0f %12.6g %12.6g %s" % (fmin_data + (step,))) 

1852 

1853 if fu <= fx: 

1854 if x >= xf: 

1855 a = xf 

1856 else: 

1857 b = xf 

1858 fulc, ffulc = nfc, fnfc 

1859 nfc, fnfc = xf, fx 

1860 xf, fx = x, fu 

1861 else: 

1862 if x < xf: 

1863 a = x 

1864 else: 

1865 b = x 

1866 if (fu <= fnfc) or (nfc == xf): 

1867 fulc, ffulc = nfc, fnfc 

1868 nfc, fnfc = x, fu 

1869 elif (fu <= ffulc) or (fulc == xf) or (fulc == nfc): 

1870 fulc, ffulc = x, fu 

1871 

1872 xm = 0.5 * (a + b) 

1873 tol1 = sqrt_eps * numpy.abs(xf) + xatol / 3.0 

1874 tol2 = 2.0 * tol1 

1875 

1876 if num >= maxfun: 

1877 flag = 1 

1878 break 

1879

1881 flag = 2 

1882

1884 if disp > 0: 

1885 _endprint(x, flag, fval, maxfun, xatol, disp)

1888 message={0: 'Solution found.', 

1889 1: 'Maximum number of function calls ' 

1890 'reached.', 

1891 2: _status_message['nan']}.get(flag, ''), 

1892 x=xf, nfev=num) 

1893

2319def bracket(func, xa=0.0, xb=1.0, args=(), grow_limit=110.0, maxiter=1000): 

2320 """ 

2321 Bracket the minimum of the function. 

2322 

2323 Given a function and distinct initial points, search in the 

2324 downhill direction (as defined by the initital points) and return 

2325 new points xa, xb, xc that bracket the minimum of the function 

2326 f(xa) > f(xb) < f(xc). It doesn't always mean that obtained 

2327 solution will satisfy xa<=x<=xb 

2328 

2329 Parameters 

2330 ---------- 

2331 func : callable f(x,*args) 

2332 Objective function to minimize. 

2333 xa, xb : float, optional 

2334 Bracketing interval. Defaults `xa` to 0.0, and `xb` to 1.0. 

2335 args : tuple, optional 

2336 Additional arguments (if present), passed to `func`. 

2337 grow_limit : float, optional 

2338 Maximum grow limit. Defaults to 110.0 

2339 maxiter : int, optional 

2340 Maximum number of iterations to perform. Defaults to 1000. 

2341 

2342 Returns 

2343 ------- 

2344 xa, xb, xc : float 

2345 Bracket. 

2346 fa, fb, fc : float 

2347 Objective function values in bracket. 

2348 funcalls : int 

2349 Number of function evaluations made. 

2350 

2351 """ 

2352 _gold = 1.618034 # golden ratio: (1.0+sqrt(5.0))/2.0 

2353 _verysmall_num = 1e-21 

2354 fa = func(*(xa,) + args) 

2355 fb = func(*(xb,) + args) 

2356 if (fa < fb): # Switch so fa > fb 

2357 xa, xb = xb, xa 

2358 fa, fb = fb, fa 

2359 xc = xb + _gold * (xb - xa) 

2360 fc = func(*((xc,) + args)) 

2361 funcalls = 3 

2362 iter = 0 

2363 while (fc < fb): 

2364 tmp1 = (xb - xa) * (fb - fc) 

2365 tmp2 = (xb - xc) * (fb - fa) 

2366 val = tmp2 - tmp1 

2367 if numpy.abs(val) < _verysmall_num: 

2368 denom = 2.0 * _verysmall_num 

2369 else: 

2370 denom = 2.0 * val 

2371 w = xb - ((xb - xc) * tmp2 - (xb - xa) * tmp1) / denom 

2372 wlim = xb + grow_limit * (xc - xb) 

2373 if iter > maxiter: 

2374 raise RuntimeError("Too many iterations.") 

2375 iter += 1 

2376 if (w - xc) * (xb - w) > 0.0: 

2377 fw = func(*((w,) + args)) 

2378 funcalls += 1 

2379 if (fw < fc): 

2380 xa = xb 

2381 xb = w 

2382 fa = fb 

2383 fb = fw 

2384 return xa, xb, xc, fa, fb, fc, funcalls 

2385 elif (fw > fb): 

2386 xc = w 

2387 fc = fw 

2388 return xa, xb, xc, fa, fb, fc, funcalls 

2389 w = xc + _gold * (xc - xb) 

2390 fw = func(*((w,) + args)) 

2391 funcalls += 1 

2392 elif (w - wlim)*(wlim - xc) >= 0.0: 

2393 w = wlim 

2394 fw = func(*((w,) + args)) 

2395 funcalls += 1 

2396 elif (w - wlim)*(xc - w) > 0.0: 

2397 fw = func(*((w,) + args)) 

2398 funcalls += 1 

2399 if (fw < fc): 

2400 xb = xc 

2401 xc = w 

2402 w = xc + _gold * (xc - xb) 

2403 fb = fc 

2404 fc = fw 

2405 fw = func(*((w,) + args)) 

2406 funcalls += 1 

2407 else: 

2408 w = xc + _gold * (xc - xb) 

2409 fw = func(*((w,) + args)) 

2410 funcalls += 1 

2411 xa = xb 

2412 xb = xc 

2413 xc = w 

2414 fa = fb 

2415 fb = fc 

2416 fc = fw 

2417 return xa, xb, xc, fa, fb, fc, funcalls

2731def brute(func, ranges, args=(), Ns=20, full_output=0, finish=fmin, 

2732 disp=False, workers=1): 

2733 """Minimize a function over a given range by brute force. 

2734 

2735 Uses the "brute force" method, i.e. computes the function's value 

2736 at each point of a multidimensional grid of points, to find the global 

2737 minimum of the function. 

2738 

2739 The function is evaluated everywhere in the range with the datatype of the 

2740 first call to the function, as enforced by the ``vectorize`` NumPy 

2741 function. The value and type of the function evaluation returned when 

2742 ``full_output=True`` are affected in addition by the ``finish`` argument 

2743 (see Notes). 

2744 

2745 The brute force approach is inefficient because the number of grid points 

2746 increases exponentially - the number of grid points to evaluate is 

2747 ``Ns ** len(x)``. Consequently, even with coarse grid spacing, even 

2748 moderately sized problems can take a long time to run, and/or run into 

2749 memory limitations. 

2750 

2751 Parameters 

2752 ---------- 

2753 func : callable 

2754 The objective function to be minimized. Must be in the 

2755 form ``f(x, *args)``, where ``x`` is the argument in 

2756 the form of a 1-D array and ``args`` is a tuple of any 

2757 additional fixed parameters needed to completely specify 

2758 the function. 

2759 ranges : tuple 

2760 Each component of the `ranges` tuple must be either a 

2761 "slice object" or a range tuple of the form ``(low, high)``. 

2762 The program uses these to create the grid of points on which 

2763 the objective function will be computed. See `Note 2` for 

2764 more detail. 

2765 args : tuple, optional 

2766 Any additional fixed parameters needed to completely specify 

2767 the function. 

2768 Ns : int, optional 

2769 Number of grid points along the axes, if not otherwise 

2770 specified. See `Note2`. 

2771 full_output : bool, optional 

2772 If True, return the evaluation grid and the objective function's 

2773 values on it. 

2774 finish : callable, optional 

2775 An optimization function that is called with the result of brute force 

2776 minimization as initial guess. `finish` should take `func` and 

2777 the initial guess as positional arguments, and take `args` as 

2778 keyword arguments. It may additionally take `full_output` 

2779 and/or `disp` as keyword arguments. Use None if no "polishing" 

2780 function is to be used. See Notes for more details. 

2781 disp : bool, optional 

2782 Set to True to print convergence messages from the `finish` callable. 

2783 workers : int or map-like callable, optional 

2784 If `workers` is an int the grid is subdivided into `workers` 

2785 sections and evaluated in parallel (uses 

2786 `multiprocessing.Pool <multiprocessing>`). 

2787 Supply `-1` to use all cores available to the Process. 

2788 Alternatively supply a map-like callable, such as 

2789 `multiprocessing.Pool.map` for evaluating the grid in parallel. 

2790 This evaluation is carried out as ``workers(func, iterable)``. 

2791 Requires that `func` be pickleable. 

2792 

2793 .. versionadded:: 1.3.0 

2794 

2795 Returns 

2796 ------- 

2797 x0 : ndarray 

2798 A 1-D array containing the coordinates of a point at which the 

2799 objective function had its minimum value. (See `Note 1` for 

2800 which point is returned.) 

2801 fval : float 

2802 Function value at the point `x0`. (Returned when `full_output` is 

2803 True.) 

2804 grid : tuple 

2805 Representation of the evaluation grid. It has the same 

2806 length as `x0`. (Returned when `full_output` is True.) 

2807 Jout : ndarray 

2808 Function values at each point of the evaluation 

2809 grid, `i.e.`, ``Jout = func(*grid)``. (Returned 

2810 when `full_output` is True.) 

2811 

2812 See Also 

2813 -------- 

2814 basinhopping, differential_evolution 

2815 

2816 Notes 

2817 ----- 

2818 *Note 1*: The program finds the gridpoint at which the lowest value 

2819 of the objective function occurs. If `finish` is None, that is the 

2820 point returned. When the global minimum occurs within (or not very far 

2821 outside) the grid's boundaries, and the grid is fine enough, that 

2822 point will be in the neighborhood of the global minimum. 

2823 

2824 However, users often employ some other optimization program to 

2825 "polish" the gridpoint values, `i.e.`, to seek a more precise 

2826 (local) minimum near `brute's` best gridpoint. 

2827 The `brute` function's `finish` option provides a convenient way to do 

2828 that. Any polishing program used must take `brute's` output as its 

2829 initial guess as a positional argument, and take `brute's` input values 

2830 for `args` as keyword arguments, otherwise an error will be raised. 

2831 It may additionally take `full_output` and/or `disp` as keyword arguments. 

2832 

2833 `brute` assumes that the `finish` function returns either an 

2834 `OptimizeResult` object or a tuple in the form: 

2835 ``(xmin, Jmin, ... , statuscode)``, where ``xmin`` is the minimizing 

2836 value of the argument, ``Jmin`` is the minimum value of the objective 

2837 function, "..." may be some other returned values (which are not used 

2838 by `brute`), and ``statuscode`` is the status code of the `finish` program. 

2839 

2840 Note that when `finish` is not None, the values returned are those 

2841 of the `finish` program, *not* the gridpoint ones. Consequently, 

2842 while `brute` confines its search to the input grid points, 

2843 the `finish` program's results usually will not coincide with any 

2844 gridpoint, and may fall outside the grid's boundary. Thus, if a 

2845 minimum only needs to be found over the provided grid points, make 

2846 sure to pass in `finish=None`. 

2847 

2848 *Note 2*: The grid of points is a `numpy.mgrid` object. 

2849 For `brute` the `ranges` and `Ns` inputs have the following effect. 

2850 Each component of the `ranges` tuple can be either a slice object or a 

2851 two-tuple giving a range of values, such as (0, 5). If the component is a 

2852 slice object, `brute` uses it directly. If the component is a two-tuple 

2853 range, `brute` internally converts it to a slice object that interpolates 

2854 `Ns` points from its low-value to its high-value, inclusive. 

2855 

2856 Examples 

2857 -------- 

2858 We illustrate the use of `brute` to seek the global minimum of a function 

2859 of two variables that is given as the sum of a positive-definite 

2860 quadratic and two deep "Gaussian-shaped" craters. Specifically, define 

2861 the objective function `f` as the sum of three other functions, 

2862 ``f = f1 + f2 + f3``. We suppose each of these has a signature 

2863 ``(z, *params)``, where ``z = (x, y)``, and ``params`` and the functions 

2864 are as defined below. 

2865 

2866 >>> params = (2, 3, 7, 8, 9, 10, 44, -1, 2, 26, 1, -2, 0.5) 

2867 >>> def f1(z, *params): 

2868 ... x, y = z 

2869 ... a, b, c, d, e, f, g, h, i, j, k, l, scale = params 

2870 ... return (a * x**2 + b * x * y + c * y**2 + d*x + e*y + f) 

2871 

2872 >>> def f2(z, *params): 

2873 ... x, y = z 

2874 ... a, b, c, d, e, f, g, h, i, j, k, l, scale = params 

2875 ... return (-g*np.exp(-((x-h)**2 + (y-i)**2) / scale)) 

2876 

2877 >>> def f3(z, *params): 

2878 ... x, y = z 

2879 ... a, b, c, d, e, f, g, h, i, j, k, l, scale = params 

2880 ... return (-j*np.exp(-((x-k)**2 + (y-l)**2) / scale)) 

2881 

2882 >>> def f(z, *params): 

2883 ... return f1(z, *params) + f2(z, *params) + f3(z, *params) 

2884 

2885 Thus, the objective function may have local minima near the minimum 

2886 of each of the three functions of which it is composed. To 

2887 use `fmin` to polish its gridpoint result, we may then continue as 

2888 follows: 

2889 

2890 >>> rranges = (slice(-4, 4, 0.25), slice(-4, 4, 0.25)) 

2891 >>> from scipy import optimize 

2892 >>> resbrute = optimize.brute(f, rranges, args=params, full_output=True, 

2893 ... finish=optimize.fmin) 

2894 >>> resbrute[0] # global minimum 

2895 array([-1.05665192, 1.80834843]) 

2896 >>> resbrute[1] # function value at global minimum 

2897 -3.4085818767 

2898 

2899 Note that if `finish` had been set to None, we would have gotten the 

2900 gridpoint [-1.0 1.75] where the rounded function value is -2.892. 

2901 

2902 """ 

2903 N = len(ranges) 

2904 if N > 40: 

2905 raise ValueError("Brute Force not possible with more " 

2906 "than 40 variables.") 

2907 lrange = list(ranges) 

2908 for k in range(N): 

2909 if type(lrange[k]) is not type(slice(None)): 

2910 if len(lrange[k]) < 3: 

2911 lrange[k] = tuple(lrange[k]) + (complex(Ns),) 

2912 lrange[k] = slice(*lrange[k]) 

2913 if (N == 1): 

2914 lrange = lrange[0] 

2915 

2916 grid = np.mgrid[lrange] 

2917 

2918 # obtain an array of parameters that is iterable by a map-like callable 

2919 inpt_shape = grid.shape 

2920 if (N > 1): 

2921 grid = np.reshape(grid, (inpt_shape[0], np.prod(inpt_shape[1:]))).T 

2922 

2923 wrapped_func = _Brute_Wrapper(func, args) 

2924 

2925 # iterate over input arrays, possibly in parallel 

2926 with MapWrapper(pool=workers) as mapper: 

2927 Jout = np.array(list(mapper(wrapped_func, grid))) 

2928 if (N == 1): 

2929 grid = (grid,) 

2930 Jout = np.squeeze(Jout) 

2931 elif (N > 1): 

2932 Jout = np.reshape(Jout, inpt_shape[1:]) 

2933 grid = np.reshape(grid.T, inpt_shape) 

2934 

2935 Nshape = shape(Jout) 

2936 

2937 indx = argmin(Jout.ravel(), axis=-1) 

2938 Nindx = zeros(N, int) 

2939 xmin = zeros(N, float) 

2940 for k in range(N - 1, -1, -1): 

2941 thisN = Nshape[k] 

2942 Nindx[k] = indx % Nshape[k] 

2943 indx = indx // thisN 

2944 for k in range(N): 

2945 xmin[k] = grid[k][tuple(Nindx)] 

2946 

2947 Jmin = Jout[tuple(Nindx)] 

2948 if (N == 1): 

2949 grid = grid[0] 

2950 xmin = xmin[0] 

2951 

2952 if callable(finish): 

2953 # set up kwargs for `finish` function 

2954 finish_args = _getargspec(finish).args 

2955 finish_kwargs = dict() 

2956 if 'full_output' in finish_args: 

2957 finish_kwargs['full_output'] = 1 

2958 if 'disp' in finish_args: 

2959 finish_kwargs['disp'] = disp 

2960 elif 'options' in finish_args: 

2961 # pass 'disp' as `options` 

2962 # (e.g. if `finish` is `minimize`) 

2963 finish_kwargs['options'] = {'disp': disp} 

2964 

2965 # run minimizer

2967 

2968 if isinstance(res, OptimizeResult): 

2969 xmin = res.x 

2970 Jmin = res.fun 

2971 success = res.success 

2972 else:

2974 Jmin = res[1]

2976 if not success: 

2977 if disp:

2979 "or `finish` does not return `statuscode` as its last " 

2980 "argument.") 

2981 

2982 if full_output:

2984 else:
