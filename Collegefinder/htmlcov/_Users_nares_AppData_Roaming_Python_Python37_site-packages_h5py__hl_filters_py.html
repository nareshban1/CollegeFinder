10""" 

11 Implements support for HDF5 compression filters via the high-level 

12 interface. The following types of filter are available: 

13 

14 "gzip" 

15 Standard DEFLATE-based compression, at integer levels from 0 to 9. 

16 Built-in to all public versions of HDF5. Use this if you want a 

17 decent-to-good ratio, good portability, and don't mind waiting. 

18 

19 "lzf" 

20 Custom compression filter for h5py. This filter is much, much faster 

21 than gzip (roughly 10x in compression vs. gzip level 4, and 3x faster 

22 in decompressing), but at the cost of a worse compression ratio. Use 

23 this if you want cheap compression and portability is not a concern. 

24 

25 "szip" 

26 Access to the HDF5 SZIP encoder. SZIP is a non-mainstream compression 

27 format used in space science on integer and float datasets. SZIP is 

28 subject to license requirements, which means the encoder is not 

29 guaranteed to be always available. However, it is also much faster 

30 than gzip. 

31 

32 The following constants in this module are also useful: 

33 

34 decode 

35 Tuple of available filter names for decoding 

36 

37 encode 

38 Tuple of available filter names for encoding 

39"""

59def _gen_filter_tuples(): 

60 """ Bootstrap function to figure out what filters are available. """ 

61 dec = [] 

62 enc = [] 

63 for name, code in _COMP_FILTERS.items(): 

64 if h5z.filter_avail(code): 

65 info = h5z.get_filter_info(code) 

66 if info & h5z.FILTER_CONFIG_ENCODE_ENABLED: 

67 enc.append(name) 

68 if info & h5z.FILTER_CONFIG_DECODE_ENABLED: 

69 dec.append(name) 

70 

71 return tuple(dec), tuple(enc)

75def _external_entry(name, offset=0, size=h5f.UNLIMITED): 

76 """ Check for and return a well-formed entry tuple for 

77 a call to h5p.set_external. """ 

78 if not isinstance(name, six.string_types): 

79 raise TypeError("External entry's name must be a string") 

80 if not isinstance(offset, six.integer_types): 

81 raise TypeError("External entry's offset must be an integer") 

82 if not isinstance(size, six.integer_types): 

83 raise TypeError("External entry's size must be an integer") 

84 return (filename_encode(name), offset, size)

86def _normalize_external(external): 

87 """ Normalize external into a well-formed list of tuples and return. """ 

88 if external is None: 

89 return [] 

90 elif isinstance(external, six.string_types): 

91 # accept a solitary file string 

92 return [_external_entry(external)] 

93 if not isinstance(external, (list)): 

94 raise ValueError('external should be a list of tuples of (file[, offset[, size]])') 

95 try: 

96 # accept a single entry, not in a list 

97 return [_external_entry(*external)] 

98 except TypeError: 

99 pass 

100 # check and rebuild each list entry to be well-formed 

101 return [_external_entry(entry) 

102 if isinstance(entry, six.string_types) else 

103 _external_entry(*entry) 

104 for entry in external]

246def get_filters(plist): 

247 """ Extract a dictionary of active filters from a DCPL, along with 

248 their settings. 

249 

250 Undocumented and subject to change without warning. 

251 """ 

252 

253 filters = {h5z.FILTER_DEFLATE: 'gzip', h5z.FILTER_SZIP: 'szip', 

254 h5z.FILTER_SHUFFLE: 'shuffle', h5z.FILTER_FLETCHER32: 'fletcher32', 

255 h5z.FILTER_LZF: 'lzf', h5z.FILTER_SCALEOFFSET: 'scaleoffset'} 

256 

257 pipeline = {} 

258 

259 nfilters = plist.get_nfilters() 

260 

261 for i in range(nfilters): 

262 

263 code, _, vals, _ = plist.get_filter(i) 

264 

265 if code == h5z.FILTER_DEFLATE: 

266 vals = vals[0] # gzip level 

267 

268 elif code == h5z.FILTER_SZIP: 

269 mask, pixels = vals[0:2] 

270 if mask & h5z.SZIP_EC_OPTION_MASK: 

271 mask = 'ec' 

272 elif mask & h5z.SZIP_NN_OPTION_MASK: 

273 mask = 'nn' 

274 else: 

275 raise TypeError("Unknown SZIP configuration") 

276 vals = (mask, pixels) 

277 elif code == h5z.FILTER_LZF: 

278 vals = None 

279 else: 

280 if len(vals) == 0: 

281 vals = None 

282 

283 pipeline[filters.get(code, str(code))] = vals 

284 

285 return pipeline

287CHUNK_BASE = 16*1024 # Multiplier by which chunks are adjusted 

288CHUNK_MIN = 8*1024 # Soft lower limit (8k) 

289CHUNK_MAX = 1024*1024 # Hard upper limit (1M)

291def guess_chunk(shape, maxshape, typesize): 

292 """ Guess an appropriate chunk layout for a dataset, given its shape and 

293 the size of each element in bytes. Will allocate chunks only as large 

294 as MAX_SIZE. Chunks are generally close to some power-of-2 fraction of 

295 each axis, slightly favoring bigger values for the last index. 

296 

297 Undocumented and subject to change without warning. 

298 """ 

299 # pylint: disable=unused-argument 

300 

301 # For unlimited dimensions we have to guess 1024 

302 shape = tuple((x if x!=0 else 1024) for i, x in enumerate(shape)) 

303 

304 ndims = len(shape) 

305 if ndims == 0: 

306 raise ValueError("Chunks not allowed for scalar datasets.") 

307 

308 chunks = np.array(shape, dtype='=f8') 

309 if not np.all(np.isfinite(chunks)): 

310 raise ValueError("Illegal value in chunk tuple") 

311 

312 # Determine the optimal chunk size in bytes using a PyTables expression. 

313 # This is kept as a float. 

314 dset_size = np.product(chunks)*typesize 

315 target_size = CHUNK_BASE * (2**np.log10(dset_size/(1024.*1024))) 

316 

317 if target_size > CHUNK_MAX: 

318 target_size = CHUNK_MAX 

319 elif target_size < CHUNK_MIN: 

320 target_size = CHUNK_MIN 

321 

322 idx = 0 

323 while True: 

324 # Repeatedly loop over the axes, dividing them by 2. Stop when: 

325 # 1a. We're smaller than the target chunk size, OR 

326 # 1b. We're within 50% of the target chunk size, AND 

327 # 2. The chunk is smaller than the maximum chunk size 

328 

329 chunk_bytes = np.product(chunks)*typesize 

330 

331 if (chunk_bytes < target_size or \ 

332 abs(chunk_bytes-target_size)/target_size < 0.5) and \ 

333 chunk_bytes < CHUNK_MAX: 

334 break 

335 

336 if np.product(chunks) == 1: 

337 break # Element size larger than CHUNK_MAX 

338 

339 chunks[idx%ndims] = np.ceil(chunks[idx%ndims] / 2.0) 

340 idx += 1 

341 

342 return tuple(int(x) for x in chunks)
