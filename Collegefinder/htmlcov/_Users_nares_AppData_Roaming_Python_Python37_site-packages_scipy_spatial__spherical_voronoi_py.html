23class SphericalVoronoi: 

24 """ Voronoi diagrams on the surface of a sphere. 

25 

26 .. versionadded:: 0.18.0 

27 

28 Parameters 

29 ---------- 

30 points : ndarray of floats, shape (npoints, ndim) 

31 Coordinates of points from which to construct a spherical 

32 Voronoi diagram. 

33 radius : float, optional 

34 Radius of the sphere (Default: 1) 

35 center : ndarray of floats, shape (ndim,) 

36 Center of sphere (Default: origin) 

37 threshold : float 

38 Threshold for detecting duplicate points and 

39 mismatches between points and sphere parameters. 

40 (Default: 1e-06) 

41 

42 Attributes 

43 ---------- 

44 points : double array of shape (npoints, ndim) 

45 the points in `ndim` dimensions to generate the Voronoi diagram from 

46 radius : double 

47 radius of the sphere 

48 center : double array of shape (ndim,) 

49 center of the sphere 

50 vertices : double array of shape (nvertices, ndim) 

51 Voronoi vertices corresponding to points 

52 regions : list of list of integers of shape (npoints, _ ) 

53 the n-th entry is a list consisting of the indices 

54 of the vertices belonging to the n-th point in points 

55 

56 Raises 

57 ------ 

58 ValueError 

59 If there are duplicates in `points`. 

60 If the provided `radius` is not consistent with `points`. 

61 

62 Notes 

63 ----- 

64 The spherical Voronoi diagram algorithm proceeds as follows. The Convex 

65 Hull of the input points (generators) is calculated, and is equivalent to 

66 their Delaunay triangulation on the surface of the sphere [Caroli]_. 

67 The Convex Hull neighbour information is then used to 

68 order the Voronoi region vertices around each generator. The latter 

69 approach is substantially less sensitive to floating point issues than 

70 angle-based methods of Voronoi region vertex sorting. 

71 

72 Empirical assessment of spherical Voronoi algorithm performance suggests 

73 quadratic time complexity (loglinear is optimal, but algorithms are more 

74 challenging to implement). 

75 

76 References 

77 ---------- 

78 .. [Caroli] Caroli et al. Robust and Efficient Delaunay triangulations of 

79 points on or close to a sphere. Research Report RR-7004, 2009. 

80 

81 See Also 

82 -------- 

83 Voronoi : Conventional Voronoi diagrams in N dimensions. 

84 

85 Examples 

86 -------- 

87 Do some imports and take some points on a cube: 

88 

89 >>> from matplotlib import colors 

90 >>> from mpl_toolkits.mplot3d.art3d import Poly3DCollection 

91 >>> import matplotlib.pyplot as plt 

92 >>> from scipy.spatial import SphericalVoronoi 

93 >>> from mpl_toolkits.mplot3d import proj3d 

94 >>> # set input data 

95 >>> points = np.array([[0, 0, 1], [0, 0, -1], [1, 0, 0], 

96 ... [0, 1, 0], [0, -1, 0], [-1, 0, 0], ]) 

97 

98 Calculate the spherical Voronoi diagram: 

99 

100 >>> radius = 1 

101 >>> center = np.array([0, 0, 0]) 

102 >>> sv = SphericalVoronoi(points, radius, center) 

103 

104 Generate plot: 

105 

106 >>> # sort vertices (optional, helpful for plotting) 

107 >>> sv.sort_vertices_of_regions() 

108 >>> fig = plt.figure() 

109 >>> ax = fig.add_subplot(111, projection='3d') 

110 >>> # plot the unit sphere for reference (optional) 

111 >>> u = np.linspace(0, 2 * np.pi, 100) 

112 >>> v = np.linspace(0, np.pi, 100) 

113 >>> x = np.outer(np.cos(u), np.sin(v)) 

114 >>> y = np.outer(np.sin(u), np.sin(v)) 

115 >>> z = np.outer(np.ones(np.size(u)), np.cos(v)) 

116 >>> ax.plot_surface(x, y, z, color='y', alpha=0.1) 

117 >>> # plot generator points 

118 >>> ax.scatter(points[:, 0], points[:, 1], points[:, 2], c='b') 

119 >>> # plot Voronoi vertices 

120 >>> ax.scatter(sv.vertices[:, 0], sv.vertices[:, 1], sv.vertices[:, 2], 

121 ... c='g') 

122 >>> # indicate Voronoi regions (as Euclidean polygons) 

123 >>> for region in sv.regions: 

124 ... random_color = colors.rgb2hex(np.random.rand(3)) 

125 ... polygon = Poly3DCollection([sv.vertices[region]], alpha=1.0) 

126 ... polygon.set_color(random_color) 

127 ... ax.add_collection3d(polygon) 

128 >>> plt.show() 

129 

130 """ 

131 def __init__(self, points, radius=1, center=None, threshold=1e-06): 

132 

133 if radius is None: 

134 radius = 1. 

135 warnings.warn('`radius` is `None`. ' 

136 'This will raise an error in a future version. ' 

137 'Please provide a floating point number ' 

138 '(i.e. `radius=1`).', 

139 DeprecationWarning) 

140 

141 self.points = points 

142 self.radius = radius 

143 self.threshold = threshold 

144 self._dim = len(points[0]) 

145 if center is None: 

146 self.center = np.zeros(self._dim) 

147 else: 

148 self.center = np.array(center) 

149 

150 # test degenerate input 

151 self._rank = np.linalg.matrix_rank(self.points - self.center, 

152 tol=self.threshold * self.radius) 

153 if self._rank <= 1: 

154 raise ValueError("Rank of input points must be at least 2") 

155 

156 if cKDTree(self.points).query_pairs(self.threshold * self.radius): 

157 raise ValueError("Duplicate generators present.") 

158 

159 radii = np.linalg.norm(self.points - self.center, axis=1) 

160 max_discrepancy = np.abs(radii - self.radius).max() 

161 if max_discrepancy >= self.threshold * self.radius: 

162 raise ValueError("Radius inconsistent with generators.") 

163 self.vertices = None 

164 self.regions = None 

165 self._tri = None 

166 self._calc_vertices_regions()

168 def _handle_geodesic_input(self): 

169 

170 # center the points 

171 centered = self.points - self.center 

172 

173 # calculate an orthogonal transformation using SVD 

174 _, _, vh = np.linalg.svd(centered) 

175 

176 # calculate the north and south poles in this basis 

177 poles = [[0, 0, self.radius], [0, 0, -self.radius]] @ vh 

178 

179 # project points into inverse basis (such that z-components are zero) 

180 circle = centered @ vh.T[:, :2] 

181 

182 # simplicial neighbors are adjacent on the circle 

183 angles = np.arctan2(circle[:, 1], circle[:, 0]) 

184 indices = np.argsort(angles) 

185 

186 # Voronoi vertices lie halfway between neighboring pairs 

187 vertices = centered[indices] + centered[np.roll(indices, 1)] 

188 vertices /= np.linalg.norm(vertices, axis=1)[:, np.newaxis] 

189 vertices *= self.radius 

190 

191 # north and south poles are also Voronoi vertices 

192 vertices = np.concatenate((vertices, poles)) 

193 

194 # each region contains two vertices from the plane and the north and 

195 # south poles 

196 invf = np.argsort(indices) 

197 invb = np.argsort(np.roll(indices, 1)) 

198 

199 n = len(self.points) 

200 regions = np.vstack([invf, # forward neighbor 

201 [n] * n, # north pole 

202 invb, # backward neighbor 

203 [n + 1] * n]).T # south pole 

204 

205 self.regions = [list(region) for region in regions] 

206 self.vertices = vertices + self.center

208 def _calc_vertices_regions(self): 

209 """ 

210 Calculates the Voronoi vertices and regions of the generators stored 

211 in self.points. The vertices will be stored in self.vertices and the 

212 regions in self.regions. 

213 

214 This algorithm was discussed at PyData London 2015 by 

215 Tyler Reddy, Ross Hemsley and Nikolai Nowaczyk 

216 """ 

217 if self._dim == 3 and self._rank == 2: 

218 self._handle_geodesic_input() 

219 return 

220 

221 # get Convex Hull 

222 self._tri = scipy.spatial.ConvexHull(self.points) 

223 # get circumcenters of Convex Hull triangles from facet equations 

224 # for 3D input circumcenters will have shape: (2N-4, 3) 

225 self.vertices = self.radius * self._tri.equations[:, :-1] + self.center 

226 # calculate regions from triangulation 

227 # for 3D input simplex_indices will have shape: (2N-4,) 

228 simplex_indices = np.arange(self._tri.simplices.shape[0]) 

229 # for 3D input tri_indices will have shape: (6N-12,) 

230 tri_indices = np.column_stack([simplex_indices] * self._dim).ravel() 

231 # for 3D input point_indices will have shape: (6N-12,) 

232 point_indices = self._tri.simplices.ravel() 

233 # for 3D input indices will have shape: (6N-12,) 

234 indices = np.argsort(point_indices, kind='mergesort') 

235 # for 3D input flattened_groups will have shape: (6N-12,) 

236 flattened_groups = tri_indices[indices].astype(np.intp) 

237 # intervals will have shape: (N+1,) 

238 intervals = np.cumsum(np.bincount(point_indices + 1)) 

239 

240 # split flattened groups to get nested list of unsorted regions 

241 groups = [list(flattened_groups[intervals[i]:intervals[i + 1]]) 

242 for i in range(len(intervals) - 1)] 

243 self.regions = groups

245 def sort_vertices_of_regions(self): 

246 """Sort indices of the vertices to be (counter-)clockwise ordered. 

247 

248 Raises 

249 ------ 

250 TypeError 

251 If the points are not three-dimensional. 

252 

253 Notes 

254 ----- 

255 For each region in regions, it sorts the indices of the Voronoi 

256 vertices such that the resulting points are in a clockwise or 

257 counterclockwise order around the generator point. 

258 

259 This is done as follows: Recall that the n-th region in regions 

260 surrounds the n-th generator in points and that the k-th 

261 Voronoi vertex in vertices is the circumcenter of the k-th triangle 

262 in _tri.simplices. For each region n, we choose the first triangle 

263 (=Voronoi vertex) in _tri.simplices and a vertex of that triangle 

264 not equal to the center n. These determine a unique neighbor of that 

265 triangle, which is then chosen as the second triangle. The second 

266 triangle will have a unique vertex not equal to the current vertex or 

267 the center. This determines a unique neighbor of the second triangle, 

268 which is then chosen as the third triangle and so forth. We proceed 

269 through all the triangles (=Voronoi vertices) belonging to the 

270 generator in points and obtain a sorted version of the vertices 

271 of its surrounding region. 

272 """ 

273 if self._dim != 3: 

274 raise TypeError("Only supported for three-dimensional point sets") 

275 if self._rank == 2: 

276 return # regions are sorted by construction 

277 _voronoi.sort_vertices_of_regions(self._tri.simplices, self.regions)
